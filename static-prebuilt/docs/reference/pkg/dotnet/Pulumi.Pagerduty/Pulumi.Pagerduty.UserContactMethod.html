Class UserContactMethod

A contact method is a contact method for a PagerDuty user (email, phone or SMS).

Example Usage

using Pulumi;
using Pagerduty = Pulumi.Pagerduty;

class MyStack : Stack
{
public MyStack()
{
    var example = new Pagerduty.User(&quot;example&quot;, new Pagerduty.UserArgs
    {
        Email = &quot;125.greenholt.earline@graham.name&quot;,
        Teams = 
        {
            pagerduty_team.Example.Id,
        },
    });
    var email = new Pagerduty.UserContactMethod(&quot;email&quot;, new Pagerduty.UserContactMethodArgs
    {
        Address = &quot;foo@bar.com&quot;,
        Label = &quot;Work&quot;,
        Type = &quot;email_contact_method&quot;,
        UserId = example.Id,
    });
    var phone = new Pagerduty.UserContactMethod(&quot;phone&quot;, new Pagerduty.UserContactMethodArgs
    {
        Address = &quot;2025550199&quot;,
        CountryCode = &quot;+1&quot;,
        Label = &quot;Work&quot;,
        Type = &quot;phone_contact_method&quot;,
        UserId = example.Id,
    });
    var sms = new Pagerduty.UserContactMethod(&quot;sms&quot;, new Pagerduty.UserContactMethodArgs
    {
        Address = &quot;2025550199&quot;,
        CountryCode = &quot;+1&quot;,
        Label = &quot;Work&quot;,
        Type = &quot;sms_contact_method&quot;,
        UserId = example.Id,
    });
}

}
