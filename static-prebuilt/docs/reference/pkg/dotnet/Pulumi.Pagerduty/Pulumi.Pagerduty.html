Addon

With add-ons, third-party developers can write their own add-ons to PagerDuty's UI. Given a configuration containing a src parameter, that URL will be embedded in an iframe on a page that's available to users from a drop-down menu.

Example Usage

using Pulumi;
using Pagerduty = Pulumi.Pagerduty;

class MyStack : Stack
{
public MyStack()
{
    var example = new Pagerduty.Addon(&quot;example&quot;, new Pagerduty.AddonArgs
    {
        Src = &quot;https://intranet.example.com/status&quot;,
    });
}

}

EscalationPolicy

An escalation policy determines what user or schedule will be notified first, second, and so on when an incident is triggered. Escalation policies are used by one or more services.

Example Usage

using Pulumi;
using Pagerduty = Pulumi.Pagerduty;

class MyStack : Stack
{
public MyStack()
{
    var exampleTeam = new Pagerduty.Team(&quot;exampleTeam&quot;, new Pagerduty.TeamArgs
    {
        Description = &quot;All engineering&quot;,
    });
    var exampleUser = new Pagerduty.User(&quot;exampleUser&quot;, new Pagerduty.UserArgs
    {
        Email = &quot;125.greenholt.earline@graham.name&quot;,
        Teams = 
        {
            exampleTeam.Id,
        },
    });
    var exampleEscalationPolicy = new Pagerduty.EscalationPolicy(&quot;exampleEscalationPolicy&quot;, new Pagerduty.EscalationPolicyArgs
    {
        NumLoops = 2,
        Rules = 
        {
            new Pagerduty.Inputs.EscalationPolicyRuleArgs
            {
                EscalationDelayInMinutes = 10,
                Target = 
                {

                    {
                        { &quot;id&quot;, exampleUser.Id },
                        { &quot;type&quot;, &quot;user&quot; },
                    },
                },
            },
        },
        Teams = 
        {
            exampleTeam.Id,
        },
    });
}

}

MaintenanceWindow

A maintenance window is used to temporarily disable one or more services for a set period of time. No incidents will be triggered and no notifications will be received while a service is disabled by a maintenance window.

Maintenance windows are specified to start at a certain time and end after they have begun. Once started, a maintenance window cannot be deleted; it can only be ended immediately to re-enable the service.

Example Usage

using Pulumi;
using Pagerduty = Pulumi.Pagerduty;

class MyStack : Stack
{
public MyStack()
{
    var example = new Pagerduty.MaintenanceWindow(&quot;example&quot;, new Pagerduty.MaintenanceWindowArgs
    {
        EndTime = &quot;2015-11-09T22:00:00-05:00&quot;,
        Services = 
        {
            pagerduty_service.Example.Id,
        },
        StartTime = &quot;2015-11-09T20:00:00-05:00&quot;,
    });
}

}

        },
    });

        Layers =

            new Pagerduty.Inputs.ScheduleLayerArgs

UserNotificationRule

A notification rule configures where and when a PagerDuty user is notified when a triggered incident is assigned to him. Unique notification rules can be created for both high and low-urgency incidents.

Example Usage

using Pulumi;
using Pagerduty = Pulumi.Pagerduty;

class MyStack : Stack
{
public MyStack()
{
    var example = new Pagerduty.User(&quot;example&quot;, new Pagerduty.UserArgs
    {
        Email = &quot;125.greenholt.earline@graham.name&quot;,
    });
    var email = new Pagerduty.UserContactMethod(&quot;email&quot;, new Pagerduty.UserContactMethodArgs
    {
        UserId = example.Id,
        Type = &quot;email_contact_method&quot;,
        Address = &quot;foo@bar.com&quot;,
        Label = &quot;Work&quot;,
    });
    var phone = new Pagerduty.UserContactMethod(&quot;phone&quot;, new Pagerduty.UserContactMethodArgs
    {
        UserId = example.Id,
        Type = &quot;phone_contact_method&quot;,
        CountryCode = &quot;+1&quot;,
        Address = &quot;2025550199&quot;,
        Label = &quot;Work&quot;,
    });
    var sms = new Pagerduty.UserContactMethod(&quot;sms&quot;, new Pagerduty.UserContactMethodArgs
    {
        UserId = example.Id,
        Type = &quot;sms_contact_method&quot;,
        CountryCode = &quot;+1&quot;,
        Address = &quot;2025550199&quot;,
        Label = &quot;Work&quot;,
    });
    var highUrgencyPhone = new Pagerduty.UserNotificationRule(&quot;highUrgencyPhone&quot;, new Pagerduty.UserNotificationRuleArgs
    {
        UserId = example.Id,
        StartDelayInMinutes = 1,
        Urgency = &quot;high&quot;,
        ContactMethod = new Pagerduty.Inputs.UserNotificationRuleContactMethodArgs
        {
            Type = &quot;phone_contact_method&quot;,
            Id = phone.Id,
        },
    });
    var lowUrgencyEmail = new Pagerduty.UserNotificationRule(&quot;lowUrgencyEmail&quot;, new Pagerduty.UserNotificationRuleArgs
    {
        UserId = example.Id,
        StartDelayInMinutes = 1,
        Urgency = &quot;low&quot;,
        ContactMethod = new Pagerduty.Inputs.UserNotificationRuleContactMethodArgs
        {
            Type = &quot;email_contact_method&quot;,
            Id = email.Id,
        },
    });
    var lowUrgencySms = new Pagerduty.UserNotificationRule(&quot;lowUrgencySms&quot;, new Pagerduty.UserNotificationRuleArgs
    {
        UserId = example.Id,
        StartDelayInMinutes = 10,
        Urgency = &quot;low&quot;,
        ContactMethod = new Pagerduty.Inputs.UserNotificationRuleContactMethodArgs
        {
            Type = &quot;sms_contact_method&quot;,
            Id = sms.Id,
        },
    });
}
