Cluster

Provides a Redshift Cluster Resource.

Note: All arguments including the username and password will be stored in the raw state as plain-text. Read more about sensitive data in state.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var @default = new Aws.RedShift.Cluster(&quot;default&quot;, new Aws.RedShift.ClusterArgs
    {
        ClusterIdentifier = &quot;tf-redshift-cluster&quot;,
        ClusterType = &quot;single-node&quot;,
        DatabaseName = &quot;mydb&quot;,
        MasterPassword = &quot;Mustbe8characters&quot;,
        MasterUsername = &quot;foo&quot;,
        NodeType = &quot;dc1.large&quot;,
    });
}

}

EventSubscription

Provides a Redshift event subscription resource.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var defaultCluster = new Aws.RedShift.Cluster(&quot;defaultCluster&quot;, new Aws.RedShift.ClusterArgs
    {
        ClusterIdentifier = &quot;default&quot;,
        DatabaseName = &quot;default&quot;,
    });
    var defaultTopic = new Aws.Sns.Topic(&quot;defaultTopic&quot;, new Aws.Sns.TopicArgs
    {
    });
    var defaultEventSubscription = new Aws.RedShift.EventSubscription(&quot;defaultEventSubscription&quot;, new Aws.RedShift.EventSubscriptionArgs
    {
        EventCategories = 
        {
            &quot;configuration&quot;,
            &quot;management&quot;,
            &quot;monitoring&quot;,
            &quot;security&quot;,
        },
        Severity = &quot;INFO&quot;,
        SnsTopicArn = defaultTopic.Arn,
        SourceIds = 
        {
            defaultCluster.Id,
        },
        SourceType = &quot;cluster&quot;,
        Tags = 
        {
            { &quot;Name&quot;, &quot;default&quot; },
        },
    });
}

}

Attributes

The following additional atttributes are provided:

  • arn - Amazon Resource Name (ARN) of the Redshift event notification subscription
  • id - The name of the Redshift event notification subscription
  • customer_aws_id - The AWS customer account associated with the Redshift event notification subscription

SnapshotCopyGrant

Creates a snapshot copy grant that allows AWS Redshift to encrypt copied snapshots with a customer master key from AWS KMS in a destination region.

Note that the grant must exist in the destination region, and not in the region of the cluster.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var testSnapshotCopyGrant = new Aws.RedShift.SnapshotCopyGrant(&quot;testSnapshotCopyGrant&quot;, new Aws.RedShift.SnapshotCopyGrantArgs
    {
        SnapshotCopyGrantName = &quot;my-grant&quot;,
    });
    var testCluster = new Aws.RedShift.Cluster(&quot;testCluster&quot;, new Aws.RedShift.ClusterArgs
    {
        SnapshotCopy = new Aws.RedShift.Inputs.ClusterSnapshotCopyArgs
        {
            DestinationRegion = &quot;us-east-2&quot;,
            GrantName = testSnapshotCopyGrant.SnapshotCopyGrantName,
        },
    });
}

}

SubnetGroup

Creates a new Amazon Redshift subnet group. You must provide a list of one or more subnets in your existing Amazon Virtual Private Cloud (Amazon VPC) when creating Amazon Redshift subnet group.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var fooVpc = new Aws.Ec2.Vpc(&quot;fooVpc&quot;, new Aws.Ec2.VpcArgs
    {
        CidrBlock = &quot;10.1.0.0/16&quot;,
    });
    var fooSubnet = new Aws.Ec2.Subnet(&quot;fooSubnet&quot;, new Aws.Ec2.SubnetArgs
    {
        AvailabilityZone = &quot;us-west-2a&quot;,
        CidrBlock = &quot;10.1.1.0/24&quot;,
        Tags = 
        {
            { &quot;Name&quot;, &quot;tf-dbsubnet-test-1&quot; },
        },
        VpcId = fooVpc.Id,
    });
    var bar = new Aws.Ec2.Subnet(&quot;bar&quot;, new Aws.Ec2.SubnetArgs
    {
        AvailabilityZone = &quot;us-west-2b&quot;,
        CidrBlock = &quot;10.1.2.0/24&quot;,
        Tags = 
        {
            { &quot;Name&quot;, &quot;tf-dbsubnet-test-2&quot; },
        },
        VpcId = fooVpc.Id,
    });
    var fooSubnetGroup = new Aws.RedShift.SubnetGroup(&quot;fooSubnetGroup&quot;, new Aws.RedShift.SubnetGroupArgs
    {
        SubnetIds = 
        {
            fooSubnet.Id,
            bar.Id,
        },
        Tags = 
        {
            { &quot;environment&quot;, &quot;Production&quot; },
        },
    });
}

}
