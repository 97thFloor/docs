IdentityPool

Provides an AWS Cognito Identity Pool.

Example Usage

using System.IO;
using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var @default = new Aws.Iam.SamlProvider(&quot;default&quot;, new Aws.Iam.SamlProviderArgs
    {
        SamlMetadataDocument = File.ReadAllText(&quot;saml-metadata.xml&quot;),
    });
    var main = new Aws.Cognito.IdentityPool(&quot;main&quot;, new Aws.Cognito.IdentityPoolArgs
    {
        AllowUnauthenticatedIdentities = false,
        CognitoIdentityProviders = 
        {
            new Aws.Cognito.Inputs.IdentityPoolCognitoIdentityProviderArgs
            {
                ClientId = &quot;6lhlkkfbfb4q5kpp90urffae&quot;,
                ProviderName = &quot;cognito-idp.us-east-1.amazonaws.com/us-east-1_Tv0493apJ&quot;,
                ServerSideTokenCheck = false,
            },
            new Aws.Cognito.Inputs.IdentityPoolCognitoIdentityProviderArgs
            {
                ClientId = &quot;7kodkvfqfb4qfkp39eurffae&quot;,
                ProviderName = &quot;cognito-idp.us-east-1.amazonaws.com/eu-west-1_Zr231apJu&quot;,
                ServerSideTokenCheck = false,
            },
        },
        IdentityPoolName = &quot;identity pool&quot;,
        OpenidConnectProviderArns = 
        {
            &quot;arn:aws:iam::123456789012:oidc-provider/foo.example.com&quot;,
        },
        SamlProviderArns = 
        {
            @default.Arn,
        },
        SupportedLoginProviders = 
        {
            { &quot;accounts.google.com&quot;, &quot;123456789012.apps.googleusercontent.com&quot; },
            { &quot;graph.facebook.com&quot;, &quot;7346241598935552&quot; },
        },
    });
}

      &quot;&quot;cognito-identity.amazonaws.com:amr&quot;&quot;: &quot;&quot;authenticated&quot;&quot;

    &quot;&quot;*&quot;&quot;

                    },
                },
                Type = &quot;Rules&quot;,
            },
        },

            { &quot;authenticated&quot;, authenticatedRole.Arn },

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var example = new Aws.Cognito.UserPool(&quot;example&quot;, new Aws.Cognito.UserPoolArgs
    {
        AutoVerifiedAttributes = 
        {
            &quot;email&quot;,
        },
    });
    var exampleProvider = new Aws.Cognito.IdentityProvider(&quot;exampleProvider&quot;, new Aws.Cognito.IdentityProviderArgs
    {
        AttributeMapping = 
        {
            { &quot;email&quot;, &quot;email&quot; },
            { &quot;username&quot;, &quot;sub&quot; },
        },
        ProviderDetails = 
        {
            { &quot;authorize_scopes&quot;, &quot;email&quot; },
            { &quot;client_id&quot;, &quot;your client_id&quot; },
            { &quot;client_secret&quot;, &quot;your client_secret&quot; },
        },
        ProviderName = &quot;Google&quot;,
        ProviderType = &quot;Google&quot;,
        UserPoolId = example.Id,
    });
}
