Plan

Provides an AWS Backup plan resource.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var example = new Aws.Backup.Plan(&quot;example&quot;, new Aws.Backup.PlanArgs
    {
        Rules = 
        {
            new Aws.Backup.Inputs.PlanRuleArgs
            {
                RuleName = &quot;tf_example_backup_rule&quot;,
                Schedule = &quot;cron(0 12 * * ? *)&quot;,
                TargetVaultName = aws_backup_vault.Test.Name,
            },
        },
    });
}

}

Selection

Manages selection conditions for AWS Backup plan resources.

Example Usage

IAM Role

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var exampleRole = new Aws.Iam.Role(&quot;exampleRole&quot;, new Aws.Iam.RoleArgs
    {
        AssumeRolePolicy = @&quot;{
&quot;&quot;Version&quot;&quot;: &quot;&quot;2012-10-17&quot;&quot;,
&quot;&quot;Statement&quot;&quot;: [
{
  &quot;&quot;Action&quot;&quot;: [&quot;&quot;sts:AssumeRole&quot;&quot;],
  &quot;&quot;Effect&quot;&quot;: &quot;&quot;allow&quot;&quot;,
  &quot;&quot;Principal&quot;&quot;: {
    &quot;&quot;Service&quot;&quot;: [&quot;&quot;backup.amazonaws.com&quot;&quot;]
  }
}
]
}

&quot;,
    });
    var exampleRolePolicyAttachment = new Aws.Iam.RolePolicyAttachment(&quot;exampleRolePolicyAttachment&quot;, new Aws.Iam.RolePolicyAttachmentArgs
    {
        PolicyArn = &quot;arn:aws:iam::aws:policy/service-role/AWSBackupServiceRolePolicyForBackup&quot;,
        Role = exampleRole.Name,
    });
    var exampleSelection = new Aws.Backup.Selection(&quot;exampleSelection&quot;, new Aws.Backup.SelectionArgs
    {
        IamRoleArn = exampleRole.Arn,
    });
}

}
