using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var mytopic = new Aws.Sns.Topic(&quot;mytopic&quot;, new Aws.Sns.TopicArgs
    {
    });
    var role = new Aws.Iam.Role(&quot;role&quot;, new Aws.Iam.RoleArgs
    {
        AssumeRolePolicy = @&quot;{
&quot;&quot;Version&quot;&quot;: &quot;&quot;2012-10-17&quot;&quot;,
&quot;&quot;Statement&quot;&quot;: [
{

    &quot;&quot;Service&quot;&quot;: &quot;&quot;iot.amazonaws.com&quot;&quot;
  },
  &quot;&quot;Action&quot;&quot;: &quot;&quot;sts:AssumeRole&quot;&quot;
}
]
}

&quot;,
    });
    var rule = new Aws.Iot.TopicRule(&quot;rule&quot;, new Aws.Iot.TopicRuleArgs

        Description = &quot;Example rule&quot;,
        Enabled = true,
        Sns = new Aws.Iot.Inputs.TopicRuleSnsArgs
        {
            Sns = &quot;RAW&quot;,
            Sns = role.Arn,
            Sns = mytopic.Arn,
        },
        Sql = &quot;SELECT * FROM &apos;topic/test&apos;&quot;,
        SqlVersion = &quot;2016-03-23&quot;,
    });
    var iamPolicyForLambda = new Aws.Iam.RolePolicy(&quot;iamPolicyForLambda&quot;, new Aws.Iam.RolePolicyArgs

{{

        &quot;&quot;sns:Publish&quot;&quot;
    ],
    &quot;&quot;Resource&quot;&quot;: &quot;&quot;{arn}&quot;&quot;
}}
]
}}

&quot;),
        Role = role.Id,
    });
