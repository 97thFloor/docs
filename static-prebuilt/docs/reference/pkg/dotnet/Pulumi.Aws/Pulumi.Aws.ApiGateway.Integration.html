Class Integration

Provides an HTTP Method Integration for an API Gateway Integration.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var myDemoAPI = new Aws.ApiGateway.RestApi(&quot;myDemoAPI&quot;, new Aws.ApiGateway.RestApiArgs
    {
        Description = &quot;This is my API for demonstration purposes&quot;,
    });
    var myDemoResource = new Aws.ApiGateway.Resource(&quot;myDemoResource&quot;, new Aws.ApiGateway.ResourceArgs
    {
        ParentId = myDemoAPI.RootResourceId,
        PathPart = &quot;mydemoresource&quot;,
        RestApi = myDemoAPI.Id,
    });
    var myDemoMethod = new Aws.ApiGateway.Method(&quot;myDemoMethod&quot;, new Aws.ApiGateway.MethodArgs
    {
        Authorization = &quot;NONE&quot;,
        HttpMethod = &quot;GET&quot;,
        ResourceId = myDemoResource.Id,
        RestApi = myDemoAPI.Id,
    });
    var myDemoIntegration = new Aws.ApiGateway.Integration(&quot;myDemoIntegration&quot;, new Aws.ApiGateway.IntegrationArgs
    {
        CacheKeyParameters = 
        {
            &quot;method.request.path.param&quot;,
        },
        CacheNamespace = &quot;foobar&quot;,
        HttpMethod = myDemoMethod.HttpMethod,
        RequestParameters = 
        {
            { &quot;integration.request.header.X-Authorization&quot;, &quot;&apos;static&apos;&quot; },
        },
        RequestTemplates = 
        {
            { &quot;application/xml&quot;, @&quot;{
&quot;&quot;body&quot;&quot; : $$input.json(&apos;$$&apos;)
}

&quot; },
        },
        ResourceId = myDemoResource.Id,
        RestApi = myDemoAPI.Id,
        TimeoutMilliseconds = 29000,
        Type = &quot;MOCK&quot;,
    });
}

}

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var config = new Config();
    var name = config.RequireObject&lt;dynamic>(&quot;name&quot;);
    var subnetId = config.RequireObject&lt;dynamic>(&quot;subnetId&quot;);
    var testLoadBalancer = new Aws.LB.LoadBalancer(&quot;testLoadBalancer&quot;, new Aws.LB.LoadBalancerArgs
    {
        Internal = true,
        LoadBalancerType = &quot;network&quot;,
        Subnets = 
        {
            subnetId,
        },
    });
    var testVpcLink = new Aws.ApiGateway.VpcLink(&quot;testVpcLink&quot;, new Aws.ApiGateway.VpcLinkArgs
    {
        TargetArn = testLoadBalancer.Arn,
    });
    var testRestApi = new Aws.ApiGateway.RestApi(&quot;testRestApi&quot;, new Aws.ApiGateway.RestApiArgs
    {
    });
    var testResource = new Aws.ApiGateway.Resource(&quot;testResource&quot;, new Aws.ApiGateway.ResourceArgs
    {
        ParentId = testRestApi.RootResourceId,
        PathPart = &quot;test&quot;,
        RestApi = testRestApi.Id,
    });
    var testMethod = new Aws.ApiGateway.Method(&quot;testMethod&quot;, new Aws.ApiGateway.MethodArgs
    {
        Authorization = &quot;NONE&quot;,
        HttpMethod = &quot;GET&quot;,
        RequestModels = 
        {
            { &quot;application/json&quot;, &quot;Error&quot; },
        },
        ResourceId = testResource.Id,
        RestApi = testRestApi.Id,
    });
    var testIntegration = new Aws.ApiGateway.Integration(&quot;testIntegration&quot;, new Aws.ApiGateway.IntegrationArgs
    {
        ConnectionId = testVpcLink.Id,
        ConnectionType = &quot;VPC_LINK&quot;,
        ContentHandling = &quot;CONVERT_TO_TEXT&quot;,
        HttpMethod = testMethod.HttpMethod,
        IntegrationHttpMethod = &quot;GET&quot;,
        PassthroughBehavior = &quot;WHEN_NO_MATCH&quot;,
        RequestParameters = 
        {
            { &quot;integration.request.header.X-Authorization&quot;, &quot;&apos;static&apos;&quot; },
            { &quot;integration.request.header.X-Foo&quot;, &quot;&apos;Bar&apos;&quot; },
        },
        RequestTemplates = 
        {
            { &quot;application/json&quot;, &quot;&quot; },
            { &quot;application/xml&quot;, @&quot;#set($$inputRoot = $$input.path(&apos;$$&apos;))
{ }
&quot; },
        },
        ResourceId = testResource.Id,
        RestApi = testRestApi.Id,
        Type = &quot;HTTP&quot;,
        Uri = &quot;https://www.google.de&quot;,
    });
}

Type

The integration input's type. Valid values are HTTP (for HTTP backends), MOCK (not calling any real backend), AWS (for AWS services), AWS_PROXY (for Lambda proxy integration) and HTTP_PROXY (for HTTP proxy integration). An HTTP or HTTP_PROXY integration with a connection_type of VPC_LINK is referred to as a private integration and uses a VpcLink to connect API Gateway to a network load balancer of a VPC.

Uri

The input's URI. Required if type is AWS, AWS_PROXY, HTTP or HTTP_PROXY. For HTTP integrations, the URI must be a fully formed, encoded HTTP(S) URL according to the RFC-3986 specification . For AWS integrations, the URI should be of the form arn:aws:apigateway:{region}:{subdomain.service|service}:{path|action}/{service_api}. region, subdomain and service are used to determine the right endpoint. e.g. arn:aws:apigateway:eu-west-1:lambda:path/2015-03-31/functions/arn:aws:lambda:eu-west-1:012345678901:function:my-func/invocations
