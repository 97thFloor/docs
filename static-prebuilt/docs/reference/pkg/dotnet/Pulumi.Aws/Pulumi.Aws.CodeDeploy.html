DeploymentConfig

Provides a CodeDeploy deployment config for an application

Example Usage

Server Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var fooDeploymentConfig = new Aws.CodeDeploy.DeploymentConfig(&quot;fooDeploymentConfig&quot;, new Aws.CodeDeploy.DeploymentConfigArgs
    {
        DeploymentConfigName = &quot;test-deployment-config&quot;,
        MinimumHealthyHosts = new Aws.CodeDeploy.Inputs.DeploymentConfigMinimumHealthyHostsArgs
        {
            Type = &quot;HOST_COUNT&quot;,
            Value = 2,
        },
    });
    var fooDeploymentGroup = new Aws.CodeDeploy.DeploymentGroup(&quot;fooDeploymentGroup&quot;, new Aws.CodeDeploy.DeploymentGroupArgs
    {
        AlarmConfiguration = new Aws.CodeDeploy.Inputs.DeploymentGroupAlarmConfigurationArgs
        {
            Alarms = 
            {
                &quot;my-alarm-name&quot;,
            },
            Enabled = true,
        },
        AppName = aws_codedeploy_app.Foo_app.Name,
        AutoRollbackConfiguration = new Aws.CodeDeploy.Inputs.DeploymentGroupAutoRollbackConfigurationArgs
        {
            Enabled = true,
            Events = 
            {
                &quot;DEPLOYMENT_FAILURE&quot;,
            },
        },
        DeploymentConfigName = fooDeploymentConfig.Id,
        DeploymentGroupName = &quot;bar&quot;,
        Ec2TagFilters = 
        {
            new Aws.CodeDeploy.Inputs.DeploymentGroupEc2TagFilterArgs
            {
                Key = &quot;filterkey&quot;,
                Type = &quot;KEY_AND_VALUE&quot;,
                Value = &quot;filtervalue&quot;,
            },
        },
        ServiceRoleArn = aws_iam_role.Foo_role.Arn,
        TriggerConfigurations = 
        {
            new Aws.CodeDeploy.Inputs.DeploymentGroupTriggerConfigurationArgs
            {
                TriggerEvents = 
                {
                    &quot;DeploymentFailure&quot;,
                },
                TriggerName = &quot;foo-trigger&quot;,
                TriggerTargetArn = &quot;foo-topic-arn&quot;,
            },
        },
    });
}

}

DeploymentGroup

Provides a CodeDeploy Deployment Group for a CodeDeploy Application

NOTE on blue/green deployments: When using green_fleet_provisioning_option with the COPY_AUTO_SCALING_GROUP action, CodeDeploy will create a new ASG with a different name. This ASG is not managed by this provider and will conflict with existing configuration and state. You may want to use a different approach to managing deployments that involve multiple ASG, such as DISCOVER_EXISTING with separate blue and green ASG.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var exampleRole = new Aws.Iam.Role(&quot;exampleRole&quot;, new Aws.Iam.RoleArgs
    {
        AssumeRolePolicy = @&quot;{
&quot;&quot;Version&quot;&quot;: &quot;&quot;2012-10-17&quot;&quot;,

{
  &quot;&quot;Sid&quot;&quot;: &quot;&quot;&quot;&quot;,
  &quot;&quot;Effect&quot;&quot;: &quot;&quot;Allow&quot;&quot;,
  &quot;&quot;Principal&quot;&quot;: {
    &quot;&quot;Service&quot;&quot;: &quot;&quot;codedeploy.amazonaws.com&quot;&quot;
  },
  &quot;&quot;Action&quot;&quot;: &quot;&quot;sts:AssumeRole&quot;&quot;
}
]
}

&quot;,
    });
    var aWSCodeDeployRole = new Aws.Iam.RolePolicyAttachment(&quot;aWSCodeDeployRole&quot;, new Aws.Iam.RolePolicyAttachmentArgs
    {
        PolicyArn = &quot;arn:aws:iam::aws:policy/service-role/AWSCodeDeployRole&quot;,
        Role = exampleRole.Name,
    });
    var exampleApplication = new Aws.CodeDeploy.Application(&quot;exampleApplication&quot;, new Aws.CodeDeploy.ApplicationArgs
    {
    });
    var exampleTopic = new Aws.Sns.Topic(&quot;exampleTopic&quot;, new Aws.Sns.TopicArgs
    {
    });
    var exampleDeploymentGroup = new Aws.CodeDeploy.DeploymentGroup(&quot;exampleDeploymentGroup&quot;, new Aws.CodeDeploy.DeploymentGroupArgs

            Enabled = true,
        },
        AppName = exampleApplication.Name,
        AutoRollbackConfiguration = new Aws.CodeDeploy.Inputs.DeploymentGroupAutoRollbackConfigurationArgs
        {
            Enabled = true,

                &quot;DEPLOYMENT_FAILURE&quot;,

                        { &quot;key&quot;, &quot;filterkey2&quot; },
                        { &quot;type&quot;, &quot;KEY_AND_VALUE&quot; },
                        { &quot;value&quot;, &quot;filtervalue&quot; },
                    },
                },
            },
        },
        ServiceRoleArn = exampleRole.Arn,
        TriggerConfigurations = 
        {
            new Aws.CodeDeploy.Inputs.DeploymentGroupTriggerConfigurationArgs

                TriggerEvents = 
                {
                    &quot;DeploymentFailure&quot;,

Blue Green Deployments with ECS

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var exampleApplication = new Aws.CodeDeploy.Application(&quot;exampleApplication&quot;, new Aws.CodeDeploy.ApplicationArgs
    {
        ComputePlatform = &quot;ECS&quot;,
    });
    var exampleDeploymentGroup = new Aws.CodeDeploy.DeploymentGroup(&quot;exampleDeploymentGroup&quot;, new Aws.CodeDeploy.DeploymentGroupArgs
    {
        AppName = exampleApplication.Name,
        AutoRollbackConfiguration = new Aws.CodeDeploy.Inputs.DeploymentGroupAutoRollbackConfigurationArgs
        {
            Enabled = true,
            Events = 
            {
                &quot;DEPLOYMENT_FAILURE&quot;,
            },
        },
        BlueGreenDeploymentConfig = new Aws.CodeDeploy.Inputs.DeploymentGroupBlueGreenDeploymentConfigArgs
        {
            DeploymentReadyOption = new Aws.CodeDeploy.Inputs.DeploymentGroupBlueGreenDeploymentConfigDeploymentReadyOptionArgs
            {
                ActionOnTimeout = &quot;CONTINUE_DEPLOYMENT&quot;,
            },
            TerminateBlueInstancesOnDeploymentSuccess = new Aws.CodeDeploy.Inputs.DeploymentGroupBlueGreenDeploymentConfigTerminateBlueInstancesOnDeploymentSuccessArgs
            {
                Action = &quot;TERMINATE&quot;,
                TerminationWaitTimeInMinutes = 5,
            },
        },
        DeploymentConfigName = &quot;CodeDeployDefault.ECSAllAtOnce&quot;,
        DeploymentGroupName = &quot;example&quot;,
        DeploymentStyle = new Aws.CodeDeploy.Inputs.DeploymentGroupDeploymentStyleArgs
        {
            DeploymentOption = &quot;WITH_TRAFFIC_CONTROL&quot;,
            DeploymentType = &quot;BLUE_GREEN&quot;,
        },
        EcsService = new Aws.CodeDeploy.Inputs.DeploymentGroupEcsServiceArgs
        {
            ClusterName = aws_ecs_cluster.Example.Name,
            ServiceName = aws_ecs_service.Example.Name,
        },
        LoadBalancerInfo = new Aws.CodeDeploy.Inputs.DeploymentGroupLoadBalancerInfoArgs
        {
            TargetGroupPairInfo = new Aws.CodeDeploy.Inputs.DeploymentGroupLoadBalancerInfoTargetGroupPairInfoArgs
            {
                ProdTrafficRoute = new Aws.CodeDeploy.Inputs.DeploymentGroupLoadBalancerInfoTargetGroupPairInfoProdTrafficRouteArgs
                {
                    ListenerArns = 
                    {
                        aws_lb_listener.Example.Arn,
                    },
                },
                TargetGroup = 
                {

                    {
                        { &quot;name&quot;, aws_lb_target_group.Blue.Name },
                    },

                    {
                        { &quot;name&quot;, aws_lb_target_group.Green.Name },
                    },
                },
            },
        },
        ServiceRoleArn = aws_iam_role.Example.Arn,
    });
}

}
