Listener

Provides a Load Balancer Listener resource.

Note: aws.alb.Listener is known as aws.lb.Listener. The functionality is identical.

Example Usage

Forward Action

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var frontEndLoadBalancer = new Aws.LB.LoadBalancer(&quot;frontEndLoadBalancer&quot;, new Aws.LB.LoadBalancerArgs
    {
    });
    var frontEndTargetGroup = new Aws.LB.TargetGroup(&quot;frontEndTargetGroup&quot;, new Aws.LB.TargetGroupArgs
    {
    });
    var frontEndListener = new Aws.LB.Listener(&quot;frontEndListener&quot;, new Aws.LB.ListenerArgs
    {
        CertificateArn = &quot;arn:aws:iam::187416307283:server-certificate/test_cert_rab3wuqwgja25ct3n4jdj2tzu4&quot;,
        DefaultActions = 
        {
            new Aws.LB.Inputs.ListenerDefaultActionArgs
            {
                TargetGroupArn = frontEndTargetGroup.Arn,
                Type = &quot;forward&quot;,
            },
        },
        LoadBalancerArn = frontEndLoadBalancer.Arn,
        Port = &quot;443&quot;,
        Protocol = &quot;HTTPS&quot;,
        SslPolicy = &quot;ELBSecurityPolicy-2016-08&quot;,
    });
}

}

ListenerCertificate

Provides a Load Balancer Listener Certificate resource.

This resource is for additional certificates and does not replace the default certificate on the listener.

Note: aws.alb.ListenerCertificate is known as aws.lb.ListenerCertificate. The functionality is identical.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var exampleCertificate = new Aws.Acm.Certificate(&quot;exampleCertificate&quot;, new Aws.Acm.CertificateArgs
    {
    });
    var frontEndLoadBalancer = new Aws.LB.LoadBalancer(&quot;frontEndLoadBalancer&quot;, new Aws.LB.LoadBalancerArgs
    {
    });
    var frontEndListener = new Aws.LB.Listener(&quot;frontEndListener&quot;, new Aws.LB.ListenerArgs
    {
    });
    var exampleListenerCertificate = new Aws.LB.ListenerCertificate(&quot;exampleListenerCertificate&quot;, new Aws.LB.ListenerCertificateArgs
    {
        CertificateArn = exampleCertificate.Arn,
        ListenerArn = frontEndListener.Arn,
    });
}

    {
    });

    {
    });
    var @static = new Aws.LB.ListenerRule(&quot;static&quot;, new Aws.LB.ListenerRuleArgs
    {
        Actions = 
        {
            new Aws.LB.Inputs.ListenerRuleActionArgs

                TargetGroupArn = aws_lb_target_group.Static.Arn,
                Type = &quot;forward&quot;,
            },
        },
        Conditions = 
        {
            new Aws.LB.Inputs.ListenerRuleConditionArgs
            {
                PathPattern = new Aws.LB.Inputs.ListenerRuleConditionPathPatternArgs
                {
                    Values = 
                    {
                        &quot;/static/*&quot;,
                    },
                },
            },
            new Aws.LB.Inputs.ListenerRuleConditionArgs
            {
                HostHeader = new Aws.LB.Inputs.ListenerRuleConditionHostHeaderArgs

                    },

Example Usage

Application Load Balancer

using System.Linq;
using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var test = new Aws.LB.LoadBalancer(&quot;test&quot;, new Aws.LB.LoadBalancerArgs
    {
        AccessLogs = new Aws.LB.Inputs.LoadBalancerAccessLogsArgs
        {
            Bucket = aws_s3_bucket.Lb_logs.Bucket,
            Enabled = true,
            Prefix = &quot;test-lb&quot;,
        },
        EnableDeletionProtection = true,
        Internal = false,
        LoadBalancerType = &quot;application&quot;,
        SecurityGroups = 
        {
            aws_security_group.Lb_sg.Id,
        },
        Subnets = 
        {
            aws_subnet.Public.Select(__item => __item.Id).ToList(),
        },
        Tags = 
        {
            { &quot;Environment&quot;, &quot;production&quot; },
        },
    });
}

}

TargetGroupAttachment

Provides the ability to register instances and containers with an Application Load Balancer (ALB) or Network Load Balancer (NLB) target group. For attaching resources with Elastic Load Balancer (ELB), see the aws.elb.Attachment resource.

Note: aws.alb.TargetGroupAttachment is known as aws.lb.TargetGroupAttachment. The functionality is identical.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var testTargetGroup = new Aws.LB.TargetGroup(&quot;testTargetGroup&quot;, new Aws.LB.TargetGroupArgs
    {
    });
    var testInstance = new Aws.Ec2.Instance(&quot;testInstance&quot;, new Aws.Ec2.InstanceArgs
    {
    });
    var testTargetGroupAttachment = new Aws.LB.TargetGroupAttachment(&quot;testTargetGroupAttachment&quot;, new Aws.LB.TargetGroupAttachmentArgs
    {
        Port = 80,
        TargetGroupArn = testTargetGroup.Arn,
        TargetId = testInstance.Id,
    });
}

}

Usage with lambda

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var testTargetGroup = new Aws.LB.TargetGroup(&quot;testTargetGroup&quot;, new Aws.LB.TargetGroupArgs
    {
        TargetType = &quot;lambda&quot;,
    });
    var testFunction = new Aws.Lambda.Function(&quot;testFunction&quot;, new Aws.Lambda.FunctionArgs
    {
    });
    var withLb = new Aws.Lambda.Permission(&quot;withLb&quot;, new Aws.Lambda.PermissionArgs
    {
        Action = &quot;lambda:InvokeFunction&quot;,
        Function = testFunction.Arn,
        Principal = &quot;elasticloadbalancing.amazonaws.com&quot;,
        SourceArn = testTargetGroup.Arn,
    });
    var testTargetGroupAttachment = new Aws.LB.TargetGroupAttachment(&quot;testTargetGroupAttachment&quot;, new Aws.LB.TargetGroupAttachmentArgs
    {
        TargetGroupArn = testTargetGroup.Arn,
        TargetId = testFunction.Arn,
    });
}

}
