Certificate

Provides a DMS (Data Migration Service) certificate resource. DMS certificates can be created, deleted, and imported.

Note: All arguments including the PEM encoded certificate will be stored in the raw state as plain-text. Read more about sensitive data in state.

using Aws = Pulumi.Aws;

{
    // Create a new certificate

    {
        CertificateId = &quot;test-dms-certificate-tf&quot;,
        CertificatePem = &quot;...&quot;,

}

ReplicationInstance

Provides a DMS (Data Migration Service) replication instance resource. DMS replication instances can be created, updated, deleted, and imported.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var dmsAssumeRole = Output.Create(Aws.Iam.GetPolicyDocument.InvokeAsync(new Aws.Iam.GetPolicyDocumentArgs
    {
        Statements = 
        {
            new Aws.Iam.Inputs.GetPolicyDocumentStatementArgs
            {
                Actions = 
                {
                    &quot;sts:AssumeRole&quot;,
                },
                Principals = 
                {
                    new Aws.Iam.Inputs.GetPolicyDocumentStatementPrincipalArgs
                    {
                        Identifiers = 
                        {
                            &quot;dms.amazonaws.com&quot;,
                        },
                        Type = &quot;Service&quot;,
                    },
                },
            },
        },
    }));
    var dms_access_for_endpoint = new Aws.Iam.Role(&quot;dms-access-for-endpoint&quot;, new Aws.Iam.RoleArgs
    {
        AssumeRolePolicy = dmsAssumeRole.Apply(dmsAssumeRole => dmsAssumeRole.Json),
    });
    var dms_access_for_endpoint_AmazonDMSRedshiftS3Role = new Aws.Iam.RolePolicyAttachment(&quot;dms-access-for-endpoint-AmazonDMSRedshiftS3Role&quot;, new Aws.Iam.RolePolicyAttachmentArgs
    {
        PolicyArn = &quot;arn:aws:iam::aws:policy/service-role/AmazonDMSRedshiftS3Role&quot;,
        Role = dms_access_for_endpoint.Name,
    });
    var dms_cloudwatch_logs_role = new Aws.Iam.Role(&quot;dms-cloudwatch-logs-role&quot;, new Aws.Iam.RoleArgs
    {
        AssumeRolePolicy = dmsAssumeRole.Apply(dmsAssumeRole => dmsAssumeRole.Json),
    });
    var dms_cloudwatch_logs_role_AmazonDMSCloudWatchLogsRole = new Aws.Iam.RolePolicyAttachment(&quot;dms-cloudwatch-logs-role-AmazonDMSCloudWatchLogsRole&quot;, new Aws.Iam.RolePolicyAttachmentArgs
    {
        PolicyArn = &quot;arn:aws:iam::aws:policy/service-role/AmazonDMSCloudWatchLogsRole&quot;,
        Role = dms_cloudwatch_logs_role.Name,
    });
    var dms_vpc_role = new Aws.Iam.Role(&quot;dms-vpc-role&quot;, new Aws.Iam.RoleArgs
    {
        AssumeRolePolicy = dmsAssumeRole.Apply(dmsAssumeRole => dmsAssumeRole.Json),
    });
    var dms_vpc_role_AmazonDMSVPCManagementRole = new Aws.Iam.RolePolicyAttachment(&quot;dms-vpc-role-AmazonDMSVPCManagementRole&quot;, new Aws.Iam.RolePolicyAttachmentArgs
    {
        PolicyArn = &quot;arn:aws:iam::aws:policy/service-role/AmazonDMSVPCManagementRole&quot;,
        Role = dms_vpc_role.Name,
    });
    // Create a new replication instance
    var test = new Aws.Dms.ReplicationInstance(&quot;test&quot;, new Aws.Dms.ReplicationInstanceArgs
    {
        AllocatedStorage = 20,
        ApplyImmediately = true,
        AutoMinorVersionUpgrade = true,
        AvailabilityZone = &quot;us-west-2c&quot;,
        EngineVersion = &quot;3.1.4&quot;,
        KmsKeyArn = &quot;arn:aws:kms:us-east-1:123456789012:key/12345678-1234-1234-1234-123456789012&quot;,
        MultiAz = false,
        PreferredMaintenanceWindow = &quot;sun:10:30-sun:14:30&quot;,
        PubliclyAccessible = true,
        ReplicationInstanceClass = &quot;dms.t2.micro&quot;,
        ReplicationInstanceId = &quot;test-dms-replication-instance-tf&quot;,
        ReplicationSubnetGroupId = aws_dms_replication_subnet_group.Test_dms_replication_subnet_group_tf.Id,
        Tags = 
        {
            { &quot;Name&quot;, &quot;test&quot; },
        },
        VpcSecurityGroupIds = 
        {
            &quot;sg-12345678&quot;,
        },
    });
}

}
