Class ListenerRule

Provides a Load Balancer Listener Rule resource.

Note: aws.alb.ListenerRule is known as aws.lb.ListenerRule. The functionality is identical.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var frontEndLoadBalancer = new Aws.LB.LoadBalancer(&quot;frontEndLoadBalancer&quot;, new Aws.LB.LoadBalancerArgs
    {
    });
    var frontEndListener = new Aws.LB.Listener(&quot;frontEndListener&quot;, new Aws.LB.ListenerArgs
    {
    });
    var @static = new Aws.LB.ListenerRule(&quot;static&quot;, new Aws.LB.ListenerRuleArgs
    {
        Actions = 
        {
            new Aws.LB.Inputs.ListenerRuleActionArgs
            {
                TargetGroupArn = aws_lb_target_group.Static.Arn,
                Type = &quot;forward&quot;,
            },
        },
        Conditions = 
        {
            new Aws.LB.Inputs.ListenerRuleConditionArgs
            {
                PathPattern = new Aws.LB.Inputs.ListenerRuleConditionPathPatternArgs
                {
                    Values = 
                    {
                        &quot;/static/*&quot;,
                    },
                },
            },
            new Aws.LB.Inputs.ListenerRuleConditionArgs
            {
                HostHeader = new Aws.LB.Inputs.ListenerRuleConditionHostHeaderArgs
                {
                    Values = 
                    {
                        &quot;example.com&quot;,
                    },
                },
            },
        },
        ListenerArn = frontEndListener.Arn,
        Priority = 100,
    });
    var hostBasedRouting = new Aws.LB.ListenerRule(&quot;hostBasedRouting&quot;, new Aws.LB.ListenerRuleArgs
    {
        Actions = 
        {
            new Aws.LB.Inputs.ListenerRuleActionArgs
            {
                TargetGroupArn = aws_lb_target_group.Static.Arn,
                Type = &quot;forward&quot;,
            },
        },
        Conditions = 
        {
            new Aws.LB.Inputs.ListenerRuleConditionArgs
            {
                HostHeader = new Aws.LB.Inputs.ListenerRuleConditionHostHeaderArgs
                {
                    Values = 
                    {
                        &quot;my-service.*.mydomain.io&quot;,
                    },
                },
            },
        },
        ListenerArn = frontEndListener.Arn,
        Priority = 99,
    });
    var redirectHttpToHttps = new Aws.LB.ListenerRule(&quot;redirectHttpToHttps&quot;, new Aws.LB.ListenerRuleArgs
    {
        Actions = 
        {
            new Aws.LB.Inputs.ListenerRuleActionArgs
            {
                Redirect = new Aws.LB.Inputs.ListenerRuleActionRedirectArgs
                {
                    Port = &quot;443&quot;,
                    Protocol = &quot;HTTPS&quot;,
                    StatusCode = &quot;HTTP_301&quot;,
                },
                Type = &quot;redirect&quot;,
            },
        },
        Conditions = 
        {
            new Aws.LB.Inputs.ListenerRuleConditionArgs
            {
                HttpHeader = new Aws.LB.Inputs.ListenerRuleConditionHttpHeaderArgs
                {
                    HttpHeaderName = &quot;X-Forwarded-For&quot;,
                    Values = 
                    {
                        &quot;192.168.1.*&quot;,
                    },
                },
            },
        },
        ListenerArn = frontEndListener.Arn,
    });
    var healthCheck = new Aws.LB.ListenerRule(&quot;healthCheck&quot;, new Aws.LB.ListenerRuleArgs
    {
        Actions = 
        {
            new Aws.LB.Inputs.ListenerRuleActionArgs
            {
                FixedResponse = new Aws.LB.Inputs.ListenerRuleActionFixedResponseArgs
                {
                    ContentType = &quot;text/plain&quot;,
                    MessageBody = &quot;HEALTHY&quot;,
                    StatusCode = &quot;200&quot;,
                },
                Type = &quot;fixed-response&quot;,
            },
        },
        Conditions = 
        {
            new Aws.LB.Inputs.ListenerRuleConditionArgs
            {
                QueryString = 
                {

                    {
                        { &quot;key&quot;, &quot;health&quot; },
                        { &quot;value&quot;, &quot;check&quot; },
                    },

                    {
                        { &quot;value&quot;, &quot;bar&quot; },
                    },
                },
            },
        },
        ListenerArn = frontEndListener.Arn,
    });
    var pool = new Aws.Cognito.UserPool(&quot;pool&quot;, new Aws.Cognito.UserPoolArgs
    {
    });
    var client = new Aws.Cognito.UserPoolClient(&quot;client&quot;, new Aws.Cognito.UserPoolClientArgs
    {
    });
    var domain = new Aws.Cognito.UserPoolDomain(&quot;domain&quot;, new Aws.Cognito.UserPoolDomainArgs
    {
    });
    var admin = new Aws.LB.ListenerRule(&quot;admin&quot;, new Aws.LB.ListenerRuleArgs
    {
        Actions = 
        {
            new Aws.LB.Inputs.ListenerRuleActionArgs
            {
                AuthenticateOidc = new Aws.LB.Inputs.ListenerRuleActionAuthenticateOidcArgs
                {
                    AuthorizationEndpoint = &quot;https://example.com/authorization_endpoint&quot;,
                    ClientId = &quot;client_id&quot;,
                    ClientSecret = &quot;client_secret&quot;,
                    Issuer = &quot;https://example.com&quot;,
                    TokenEndpoint = &quot;https://example.com/token_endpoint&quot;,
                    UserInfoEndpoint = &quot;https://example.com/user_info_endpoint&quot;,
                },
                Type = &quot;authenticate-oidc&quot;,
            },
            new Aws.LB.Inputs.ListenerRuleActionArgs
            {
                TargetGroupArn = aws_lb_target_group.Static.Arn,
                Type = &quot;forward&quot;,
            },
        },
        ListenerArn = frontEndListener.Arn,
    });
}

}
