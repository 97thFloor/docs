Service

Provides a Service Discovery Service resource.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var exampleVpc = new Aws.Ec2.Vpc(&quot;exampleVpc&quot;, new Aws.Ec2.VpcArgs
    {
        CidrBlock = &quot;10.0.0.0/16&quot;,
        EnableDnsHostnames = true,
        EnableDnsSupport = true,
    });
    var examplePrivateDnsNamespace = new Aws.ServiceDiscovery.PrivateDnsNamespace(&quot;examplePrivateDnsNamespace&quot;, new Aws.ServiceDiscovery.PrivateDnsNamespaceArgs
    {
        Description = &quot;example&quot;,
        Vpc = exampleVpc.Id,
    });
    var exampleService = new Aws.ServiceDiscovery.Service(&quot;exampleService&quot;, new Aws.ServiceDiscovery.ServiceArgs
    {
        DnsConfig = new Aws.ServiceDiscovery.Inputs.ServiceDnsConfigArgs
        {
            DnsRecords = 
            {
                new Aws.ServiceDiscovery.Inputs.ServiceDnsConfigDnsRecordArgs
                {
                    Ttl = 10,
                    Type = &quot;A&quot;,
                },
            },
            NamespaceId = examplePrivateDnsNamespace.Id,
            RoutingPolicy = &quot;MULTIVALUE&quot;,
        },
        HealthCheckCustomConfig = new Aws.ServiceDiscovery.Inputs.ServiceHealthCheckCustomConfigArgs
        {
            FailureThreshold = 1,
        },
    });
}

}
