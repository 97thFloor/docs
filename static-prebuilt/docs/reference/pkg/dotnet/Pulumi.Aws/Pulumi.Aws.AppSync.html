{

    });

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var exampleTable = new Aws.DynamoDB.Table(&quot;exampleTable&quot;, new Aws.DynamoDB.TableArgs
    {
        Attributes = 
        {
            new Aws.DynamoDB.Inputs.TableAttributeArgs
            {
                Name = &quot;UserId&quot;,
                Type = &quot;S&quot;,
            },
        },
        HashKey = &quot;UserId&quot;,
        ReadCapacity = 1,
        WriteCapacity = 1,
    });
    var exampleRole = new Aws.Iam.Role(&quot;exampleRole&quot;, new Aws.Iam.RoleArgs
    {
        AssumeRolePolicy = @&quot;{
&quot;&quot;Version&quot;&quot;: &quot;&quot;2012-10-17&quot;&quot;,
&quot;&quot;Statement&quot;&quot;: [
{
  &quot;&quot;Action&quot;&quot;: &quot;&quot;sts:AssumeRole&quot;&quot;,
  &quot;&quot;Principal&quot;&quot;: {
    &quot;&quot;Service&quot;&quot;: &quot;&quot;appsync.amazonaws.com&quot;&quot;
  },
  &quot;&quot;Effect&quot;&quot;: &quot;&quot;Allow&quot;&quot;
}
]
}

&quot;,
    });
    var exampleRolePolicy = new Aws.Iam.RolePolicy(&quot;exampleRolePolicy&quot;, new Aws.Iam.RolePolicyArgs
    {
        Policy = exampleTable.Arn.Apply(arn => @$&quot;{{
&quot;&quot;Version&quot;&quot;: &quot;&quot;2012-10-17&quot;&quot;,
&quot;&quot;Statement&quot;&quot;: [
{{
  &quot;&quot;Action&quot;&quot;: [
    &quot;&quot;dynamodb:*&quot;&quot;
  ],
  &quot;&quot;Effect&quot;&quot;: &quot;&quot;Allow&quot;&quot;,
  &quot;&quot;Resource&quot;&quot;: [
    &quot;&quot;{arn}&quot;&quot;
  ]
}}
]
}}

&quot;),
        Role = exampleRole.Id,
    });
    var exampleGraphQLApi = new Aws.AppSync.GraphQLApi(&quot;exampleGraphQLApi&quot;, new Aws.AppSync.GraphQLApiArgs
    {
        AuthenticationType = &quot;API_KEY&quot;,
    });
    var exampleDataSource = new Aws.AppSync.DataSource(&quot;exampleDataSource&quot;, new Aws.AppSync.DataSourceArgs
    {
        ApiId = exampleGraphQLApi.Id,
        DynamodbConfig = new Aws.AppSync.Inputs.DataSourceDynamodbConfigArgs
        {
            TableName = exampleTable.Name,
        },
        ServiceRoleArn = exampleRole.Arn,
        Type = &quot;AMAZON_DYNAMODB&quot;,
    });
}

    {
        ApiId = testGraphQLApi.Id,
        DataSource = testDataSource.Name,
        Name = &quot;tf_example&quot;,
        RequestMappingTemplate = @&quot;{
&quot;&quot;version&quot;&quot;: &quot;&quot;2018-05-29&quot;&quot;,
&quot;&quot;method&quot;&quot;: &quot;&quot;GET&quot;&quot;,
&quot;&quot;resourcePath&quot;&quot;: &quot;&quot;/&quot;&quot;,

&quot;,

$$ctx.result.body

Resolver

Provides an AppSync Resolver.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var testGraphQLApi = new Aws.AppSync.GraphQLApi(&quot;testGraphQLApi&quot;, new Aws.AppSync.GraphQLApiArgs
    {
        AuthenticationType = &quot;API_KEY&quot;,
        Schema = @&quot;type Mutation {
putPost(id: ID!, title: String!): Post
}

type Post {
id: ID!
title: String!
}

type Query {
singlePost(id: ID!): Post
}

schema {
query: Query
mutation: Mutation
}

&quot;,
    });
    var testDataSource = new Aws.AppSync.DataSource(&quot;testDataSource&quot;, new Aws.AppSync.DataSourceArgs
    {
        ApiId = testGraphQLApi.Id,
        HttpConfig = new Aws.AppSync.Inputs.DataSourceHttpConfigArgs
        {
            Endpoint = &quot;http://example.com&quot;,
        },
        Type = &quot;HTTP&quot;,
    });
    // UNIT type resolver (default)
    var testResolver = new Aws.AppSync.Resolver(&quot;testResolver&quot;, new Aws.AppSync.ResolverArgs
    {
        ApiId = testGraphQLApi.Id,
        CachingConfig = new Aws.AppSync.Inputs.ResolverCachingConfigArgs
        {
            CachingKeys = 
            {
                &quot;$$context.identity.sub&quot;,
                &quot;$$context.arguments.id&quot;,
            },
            Ttl = 60,
        },
        DataSource = testDataSource.Name,
        Field = &quot;singlePost&quot;,
        RequestTemplate = @&quot;{
&quot;&quot;version&quot;&quot;: &quot;&quot;2018-05-29&quot;&quot;,
&quot;&quot;method&quot;&quot;: &quot;&quot;GET&quot;&quot;,
&quot;&quot;resourcePath&quot;&quot;: &quot;&quot;/&quot;&quot;,
&quot;&quot;params&quot;&quot;:{
    &quot;&quot;headers&quot;&quot;: $$utils.http.copyheaders($$ctx.request.headers)
}
}

&quot;,
        ResponseTemplate = @&quot;#if($$ctx.result.statusCode == 200)
$$ctx.result.body
#else
$$utils.appendError($$ctx.result.body, $$ctx.result.statusCode)
#end

&quot;,
        Type = &quot;Query&quot;,
    });
    // PIPELINE type resolver
    var mutationPipelineTest = new Aws.AppSync.Resolver(&quot;mutationPipelineTest&quot;, new Aws.AppSync.ResolverArgs
    {
        ApiId = testGraphQLApi.Id,
        Field = &quot;pipelineTest&quot;,
        Kind = &quot;PIPELINE&quot;,
        PipelineConfig = new Aws.AppSync.Inputs.ResolverPipelineConfigArgs
        {
            Functions = 
            {
                aws_appsync_function.Test1.Function_id,
                aws_appsync_function.Test2.Function_id,
                aws_appsync_function.Test3.Function_id,
            },
        },
        RequestTemplate = &quot;{}&quot;,
        ResponseTemplate = &quot;$$util.toJson($$ctx.result)&quot;,
        Type = &quot;Mutation&quot;,
    });
}
