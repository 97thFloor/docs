Generates an IAM policy document in JSON format.

This is a data source which can be used to construct a JSON representation of an IAM policy document, for use with resources which expect policy documents, such as the aws.iam.Policy resource.

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var examplePolicyDocument = Output.Create(Aws.Iam.GetPolicyDocument.InvokeAsync(new Aws.Iam.GetPolicyDocumentArgs
    {
        Statements = 
        {
            new Aws.Iam.Inputs.GetPolicyDocumentStatementArgs
            {
                Actions = 
                {
                    &quot;s3:ListAllMyBuckets&quot;,
                    &quot;s3:GetBucketLocation&quot;,
                },
                Resources = 
                {
                    &quot;arn:aws:s3:::*&quot;,
                },
                Sid = &quot;1&quot;,
            },
            new Aws.Iam.Inputs.GetPolicyDocumentStatementArgs
            {
                Actions = 
                {
                    &quot;s3:ListBucket&quot;,
                },
                Condition = 
                {

                    {
                        { &quot;test&quot;, &quot;StringLike&quot; },
                        { &quot;values&quot;, 
                        {
                            &quot;&quot;,
                            &quot;home/&quot;,
                            &quot;home/&amp;{aws:username}/&quot;,
                        } },
                        { &quot;variable&quot;, &quot;s3:prefix&quot; },
                    },
                },
                Resources = 
                {
                    $&quot;arn:aws:s3:::{@var.S3_bucket_name}&quot;,
                },
            },
            new Aws.Iam.Inputs.GetPolicyDocumentStatementArgs
            {
                Actions = 
                {
                    &quot;s3:*&quot;,
                },
                Resources = 
                {
                    $&quot;arn:aws:s3:::{@var.S3_bucket_name}/home/&amp;{{aws:username}}&quot;,
                    $&quot;arn:aws:s3:::{@var.S3_bucket_name}/home/&amp;{{aws:username}}/*&quot;,
                },
            },
        },
    }));
    var examplePolicy = new Aws.Iam.Policy(&quot;examplePolicy&quot;, new Aws.Iam.PolicyArgs
    {
        Path = &quot;/&quot;,
        Policy = examplePolicyDocument.Apply(examplePolicyDocument => examplePolicyDocument.Json),
    });
}

}

Using this data source to generate policy documents is optional. It is also valid to use literal JSON strings within your configuration, or to use the file interpolation function to read a raw JSON policy document from a file.

Context Variable Interpolation

The IAM policy document format allows context variables to be interpolated into various strings within a statement. The native IAM policy document format uses ${...}-style syntax that is in conflict with interpolation syntax, so this data source instead uses &amp;{...} syntax for interpolations that should be processed by AWS rather than by this provider.

Wildcard Principal

In order to define wildcard principal (a.k.a. anonymous user) use type = &quot;*&quot; and identifiers = [&quot;*&quot;]. In that case the rendered json will contain &quot;Principal&quot;: &quot;*&quot;. Note, that even though the IAM Documentation states that &quot;Principal&quot;: &quot;*&quot; and &quot;Principal&quot;: {&quot;AWS&quot;: &quot;*&quot;} are equivalent, those principals have different behavior for IAM Role Trust Policy. Therefore this provider will normalize the principal field only in above-mentioned case and principals like type = &quot;AWS&quot; and identifiers = [&quot;*&quot;] will be rendered as &quot;Principal&quot;: {&quot;AWS&quot;: &quot;*&quot;}.

Example with Multiple Principals

Showing how you can use this as an assume role policy as well as showing how you can specify multiple principal blocks with different types.

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var eventStreamBucketRoleAssumeRolePolicy = Output.Create(Aws.Iam.GetPolicyDocument.InvokeAsync(new Aws.Iam.GetPolicyDocumentArgs
    {
        Statements = 
        {
            new Aws.Iam.Inputs.GetPolicyDocumentStatementArgs
            {
                Actions = 
                {
                    &quot;sts:AssumeRole&quot;,
                },
                Principals = 
                {
                    new Aws.Iam.Inputs.GetPolicyDocumentStatementPrincipalArgs
                    {
                        Identifiers = 
                        {
                            &quot;firehose.amazonaws.com&quot;,
                        },
                        Type = &quot;Service&quot;,
                    },
                    new Aws.Iam.Inputs.GetPolicyDocumentStatementPrincipalArgs
                    {
                        Identifiers = 
                        {
                            @var.Trusted_role_arn,
                        },
                        Type = &quot;AWS&quot;,
                    },
                    new Aws.Iam.Inputs.GetPolicyDocumentStatementPrincipalArgs
                    {
                        Identifiers = 
                        {
                            $&quot;arn:aws:iam::{@var.Account_id}:saml-provider/{@var.Provider_name}&quot;,
                            &quot;cognito-identity.amazonaws.com&quot;,
                        },
                        Type = &quot;Federated&quot;,
                    },
                },
            },
        },
    }));
}

}

Example with Source and Override

Showing how you can use source_json and override_json

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var source = Output.Create(Aws.Iam.GetPolicyDocument.InvokeAsync(new Aws.Iam.GetPolicyDocumentArgs
    {
        Statements = 
        {
            new Aws.Iam.Inputs.GetPolicyDocumentStatementArgs
            {
                Actions = 
                {
                    &quot;ec2:*&quot;,
                },
                Resources = 
                {
                    &quot;*&quot;,
                },
            },
            new Aws.Iam.Inputs.GetPolicyDocumentStatementArgs
            {
                Actions = 
                {
                    &quot;s3:*&quot;,
                },
                Resources = 
                {
                    &quot;*&quot;,
                },
                Sid = &quot;SidToOverwrite&quot;,
            },
        },
    }));
    var sourceJsonExample = source.Apply(source => Output.Create(Aws.Iam.GetPolicyDocument.InvokeAsync(new Aws.Iam.GetPolicyDocumentArgs
    {
        SourceJson = source.Json,
        Statements = 
        {
            new Aws.Iam.Inputs.GetPolicyDocumentStatementArgs
            {
                Actions = 
                {
                    &quot;s3:*&quot;,
                },
                Resources = 
                {
                    &quot;arn:aws:s3:::somebucket&quot;,
                    &quot;arn:aws:s3:::somebucket/*&quot;,
                },
                Sid = &quot;SidToOverwrite&quot;,
            },
        },
    })));
    var @override = Output.Create(Aws.Iam.GetPolicyDocument.InvokeAsync(new Aws.Iam.GetPolicyDocumentArgs
    {
        Statements = 
        {
            new Aws.Iam.Inputs.GetPolicyDocumentStatementArgs
            {
                Actions = 
                {
                    &quot;s3:*&quot;,
                },
                Resources = 
                {
                    &quot;*&quot;,
                },
                Sid = &quot;SidToOverwrite&quot;,
            },
        },
    }));
    var overrideJsonExample = @override.Apply(@override => Output.Create(Aws.Iam.GetPolicyDocument.InvokeAsync(new Aws.Iam.GetPolicyDocumentArgs
    {
        OverrideJson = @override.Json,
        Statements = 
        {
            new Aws.Iam.Inputs.GetPolicyDocumentStatementArgs
            {
                Actions = 
                {
                    &quot;ec2:*&quot;,
                },
                Resources = 
                {
                    &quot;*&quot;,
                },
            },
            new Aws.Iam.Inputs.GetPolicyDocumentStatementArgs
            {
                Actions = 
                {
                    &quot;s3:*&quot;,
                },
                Resources = 
                {
                    &quot;arn:aws:s3:::somebucket&quot;,
                    &quot;arn:aws:s3:::somebucket/*&quot;,
                },
                Sid = &quot;SidToOverwrite&quot;,
            },
        },
    })));
}

}

Example without Statement

Use without a statement:

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var source = Output.Create(Aws.Iam.GetPolicyDocument.InvokeAsync(new Aws.Iam.GetPolicyDocumentArgs
    {
        Statements = 
        {
            new Aws.Iam.Inputs.GetPolicyDocumentStatementArgs
            {
                Actions = 
                {
                    &quot;ec2:DescribeAccountAttributes&quot;,
                },
                Resources = 
                {
                    &quot;*&quot;,
                },
                Sid = &quot;OverridePlaceholder&quot;,
            },
        },
    }));
    var @override = Output.Create(Aws.Iam.GetPolicyDocument.InvokeAsync(new Aws.Iam.GetPolicyDocumentArgs
    {
        Statements = 
        {
            new Aws.Iam.Inputs.GetPolicyDocumentStatementArgs
            {
                Actions = 
                {
                    &quot;s3:GetObject&quot;,
                },
                Resources = 
                {
                    &quot;*&quot;,
                },
                Sid = &quot;OverridePlaceholder&quot;,
            },
        },
    }));
    var politik = Output.Tuple(@override, source).Apply(values =>
    {
        var @override = values.Item1;
        var source = values.Item2;
        return Output.Create(Aws.Iam.GetPolicyDocument.InvokeAsync(new Aws.Iam.GetPolicyDocumentArgs
        {
            OverrideJson = @override.Json,
            SourceJson = source.Json,
        }));
    });
}

}
