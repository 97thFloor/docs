Account

Enables Security Hub for this AWS account.

NOTE: Destroying this resource will disable Security Hub for this AWS account.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var example = new Aws.SecurityHub.Account(&quot;example&quot;, new Aws.SecurityHub.AccountArgs
    {
    });
}

}

Member

Provides a Security Hub member resource.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var exampleAccount = new Aws.SecurityHub.Account(&quot;exampleAccount&quot;, new Aws.SecurityHub.AccountArgs
    {
    });
    var exampleMember = new Aws.SecurityHub.Member(&quot;exampleMember&quot;, new Aws.SecurityHub.MemberArgs
    {
        AccountId = &quot;123456789012&quot;,
        Email = &quot;example@example.com&quot;,
        Invite = true,
    });

ProductSubscription

Subscribes to a Security Hub product.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var exampleAccount = new Aws.SecurityHub.Account(&quot;exampleAccount&quot;, new Aws.SecurityHub.AccountArgs
    {
    });
    var current = Output.Create(Aws.GetRegion.InvokeAsync());
    var exampleProductSubscription = new Aws.SecurityHub.ProductSubscription(&quot;exampleProductSubscription&quot;, new Aws.SecurityHub.ProductSubscriptionArgs
    {
        ProductArn = current.Apply(current => $&quot;arn:aws:securityhub:{current.Name}:733251395267:product/alertlogic/althreatmanagement&quot;),
    });
}

}

StandardsSubscription

Subscribes to a Security Hub standard.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var example = new Aws.SecurityHub.Account(&quot;example&quot;, new Aws.SecurityHub.AccountArgs
    {
    });
    var cis = new Aws.SecurityHub.StandardsSubscription(&quot;cis&quot;, new Aws.SecurityHub.StandardsSubscriptionArgs
    {
        StandardsArn = &quot;arn:aws:securityhub:::ruleset/cis-aws-foundations-benchmark/v/1.2.0&quot;,
    });
    var pci321 = new Aws.SecurityHub.StandardsSubscription(&quot;pci321&quot;, new Aws.SecurityHub.StandardsSubscriptionArgs
    {
        StandardsArn = &quot;arn:aws:securityhub:us-east-1::standards/pci-dss/v/3.2.1&quot;,
    });
}

}
