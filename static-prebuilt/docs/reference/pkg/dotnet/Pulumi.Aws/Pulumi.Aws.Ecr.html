LifecyclePolicy

Manages an ECR repository lifecycle policy.

NOTE: Only one aws.ecr.LifecyclePolicy resource can be used with the same ECR repository. To apply multiple rules, they must be combined in the policy JSON.

NOTE: The AWS ECR API seems to reorder rules based on rulePriority. If you define multiple rules that are not sorted in ascending rulePriority order in the this provider code, the resource will be flagged for recreation every deployment.

Example Usage

Policy on untagged image

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var foo = new Aws.Ecr.Repository(&quot;foo&quot;, new Aws.Ecr.RepositoryArgs
    {
    });
    var foopolicy = new Aws.Ecr.LifecyclePolicy(&quot;foopolicy&quot;, new Aws.Ecr.LifecyclePolicyArgs
    {
        Policy = @&quot;{
&quot;&quot;rules&quot;&quot;: [
    {
        &quot;&quot;rulePriority&quot;&quot;: 1,
        &quot;&quot;description&quot;&quot;: &quot;&quot;Expire images older than 14 days&quot;&quot;,
        &quot;&quot;selection&quot;&quot;: {
            &quot;&quot;tagStatus&quot;&quot;: &quot;&quot;untagged&quot;&quot;,
            &quot;&quot;countType&quot;&quot;: &quot;&quot;sinceImagePushed&quot;&quot;,
            &quot;&quot;countUnit&quot;&quot;: &quot;&quot;days&quot;&quot;,
            &quot;&quot;countNumber&quot;&quot;: 14
        },
        &quot;&quot;action&quot;&quot;: {
            &quot;&quot;type&quot;&quot;: &quot;&quot;expire&quot;&quot;
        }
    }
]
}

&quot;,
        Repository = foo.Name,
    });
}

}

Repository

Provides an Elastic Container Registry Repository.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var foo = new Aws.Ecr.Repository(&quot;foo&quot;, new Aws.Ecr.RepositoryArgs
    {
        ImageScanningConfiguration = new Aws.Ecr.Inputs.RepositoryImageScanningConfigurationArgs
        {
            ScanOnPush = true,
        },
        ImageTagMutability = &quot;MUTABLE&quot;,
    });
}

}

RepositoryPolicy

Provides an Elastic Container Registry Repository Policy.

Note that currently only one policy may be applied to a repository.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var foo = new Aws.Ecr.Repository(&quot;foo&quot;, new Aws.Ecr.RepositoryArgs
    {
    });
    var foopolicy = new Aws.Ecr.RepositoryPolicy(&quot;foopolicy&quot;, new Aws.Ecr.RepositoryPolicyArgs
    {
        Policy = @&quot;{
&quot;&quot;Version&quot;&quot;: &quot;&quot;2008-10-17&quot;&quot;,
&quot;&quot;Statement&quot;&quot;: [
    {
        &quot;&quot;Sid&quot;&quot;: &quot;&quot;new policy&quot;&quot;,
        &quot;&quot;Effect&quot;&quot;: &quot;&quot;Allow&quot;&quot;,
        &quot;&quot;Principal&quot;&quot;: &quot;&quot;*&quot;&quot;,
        &quot;&quot;Action&quot;&quot;: [
            &quot;&quot;ecr:GetDownloadUrlForLayer&quot;&quot;,
            &quot;&quot;ecr:BatchGetImage&quot;&quot;,
            &quot;&quot;ecr:BatchCheckLayerAvailability&quot;&quot;,
            &quot;&quot;ecr:PutImage&quot;&quot;,
            &quot;&quot;ecr:InitiateLayerUpload&quot;&quot;,
            &quot;&quot;ecr:UploadLayerPart&quot;&quot;,
            &quot;&quot;ecr:CompleteLayerUpload&quot;&quot;,
            &quot;&quot;ecr:DescribeRepositories&quot;&quot;,
            &quot;&quot;ecr:GetRepositoryPolicy&quot;&quot;,
            &quot;&quot;ecr:ListImages&quot;&quot;,
            &quot;&quot;ecr:DeleteRepository&quot;&quot;,
            &quot;&quot;ecr:BatchDeleteImage&quot;&quot;,
            &quot;&quot;ecr:SetRepositoryPolicy&quot;&quot;,
            &quot;&quot;ecr:DeleteRepositoryPolicy&quot;&quot;
        ]
    }
]
}

&quot;,
        Repository = foo.Name,
    });
}

}
