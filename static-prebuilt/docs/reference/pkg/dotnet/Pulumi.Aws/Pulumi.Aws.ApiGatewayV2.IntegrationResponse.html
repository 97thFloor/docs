Class IntegrationResponse

Manages an Amazon API Gateway Version 2 integration response. More information can be found in the Amazon API Gateway Developer Guide.

Example Usage

Basic

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var example = new Aws.ApiGatewayV2.IntegrationResponse(&quot;example&quot;, new Aws.ApiGatewayV2.IntegrationResponseArgs
    {
        ApiId = aws_apigatewayv2_api.Example.Id,
        IntegrationId = aws_apigatewayv2_integration.Example.Id,
        IntegrationResponseKey = &quot;/200/&quot;,
    });
}

}
