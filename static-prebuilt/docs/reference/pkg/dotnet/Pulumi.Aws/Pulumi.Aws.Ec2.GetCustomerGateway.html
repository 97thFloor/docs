using Pulumi;

        Filters =

            new Aws.Ec2.Inputs.GetCustomerGatewayFilterArgs
            {
                Name = &quot;tag:Name&quot;,

                    &quot;foo-prod&quot;,
                },
            },

    {
        AmazonSideAsn = 7224,
        VpcId = aws_vpc.Main.Id,

    {
        CustomerGatewayId = foo.Apply(foo => foo.Id),
        StaticRoutesOnly = false,
        Type = foo.Apply(foo => foo.Type),
        VpnGatewayId = main.Id,
    });

}
