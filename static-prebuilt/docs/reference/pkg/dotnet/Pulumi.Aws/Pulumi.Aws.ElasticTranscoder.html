Pipeline

Provides an Elastic Transcoder pipeline resource.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var bar = new Aws.ElasticTranscoder.Pipeline(&quot;bar&quot;, new Aws.ElasticTranscoder.PipelineArgs
    {
        ContentConfig = new Aws.ElasticTranscoder.Inputs.PipelineContentConfigArgs
        {
            Bucket = aws_s3_bucket.Content_bucket.Bucket,
            StorageClass = &quot;Standard&quot;,
        },
        InputBucket = aws_s3_bucket.Input_bucket.Bucket,
        Role = aws_iam_role.Test_role.Arn,
        ThumbnailConfig = new Aws.ElasticTranscoder.Inputs.PipelineThumbnailConfigArgs
        {
            Bucket = aws_s3_bucket.Thumb_bucket.Bucket,
            StorageClass = &quot;Standard&quot;,
        },
    });
}

Preset

Provides an Elastic Transcoder preset resource.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var bar = new Aws.ElasticTranscoder.Preset(&quot;bar&quot;, new Aws.ElasticTranscoder.PresetArgs
    {
        Audio = new Aws.ElasticTranscoder.Inputs.PresetAudioArgs
        {
            AudioPackingMode = &quot;SingleTrack&quot;,
            BitRate = 96,
            Channels = 2,
            Codec = &quot;AAC&quot;,
            SampleRate = 44100,
        },
        AudioCodecOptions = new Aws.ElasticTranscoder.Inputs.PresetAudioCodecOptionsArgs
        {
            Profile = &quot;AAC-LC&quot;,
        },
        Container = &quot;mp4&quot;,
        Description = &quot;Sample Preset&quot;,
        Thumbnails = new Aws.ElasticTranscoder.Inputs.PresetThumbnailsArgs
        {
            Format = &quot;png&quot;,
            Interval = 120,
            MaxHeight = &quot;auto&quot;,
            MaxWidth = &quot;auto&quot;,
            PaddingPolicy = &quot;Pad&quot;,
            SizingPolicy = &quot;Fit&quot;,
        },
        Video = new Aws.ElasticTranscoder.Inputs.PresetVideoArgs
        {
            BitRate = &quot;1600&quot;,
            Codec = &quot;H.264&quot;,
            DisplayAspectRatio = &quot;16:9&quot;,
            FixedGop = &quot;false&quot;,
            FrameRate = &quot;auto&quot;,
            KeyframesMaxDist = 240,
            MaxFrameRate = &quot;60&quot;,
            MaxHeight = &quot;auto&quot;,
            MaxWidth = &quot;auto&quot;,
            PaddingPolicy = &quot;Pad&quot;,
            SizingPolicy = &quot;Fit&quot;,
        },
        VideoCodecOptions = 
        {
            { &quot;ColorSpaceConversionMode&quot;, &quot;None&quot; },
            { &quot;InterlacedMode&quot;, &quot;Progressive&quot; },
            { &quot;Level&quot;, &quot;2.2&quot; },
            { &quot;MaxReferenceFrames&quot;, 3 },
            { &quot;Profile&quot;, &quot;main&quot; },
        },
        VideoWatermarks = 
        {
            new Aws.ElasticTranscoder.Inputs.PresetVideoWatermarkArgs
            {
                HorizontalAlign = &quot;Right&quot;,
                HorizontalOffset = &quot;10px&quot;,
                Id = &quot;Test&quot;,
                MaxHeight = &quot;20%&quot;,
                MaxWidth = &quot;20%&quot;,
                Opacity = &quot;55.5&quot;,
                SizingPolicy = &quot;ShrinkToFit&quot;,
                Target = &quot;Content&quot;,
                VerticalAlign = &quot;Bottom&quot;,
                VerticalOffset = &quot;10px&quot;,
            },
        },
    });
}

}
