DeliveryChannel

Provides an AWS Config Delivery Channel.

Note: Delivery Channel requires a Configuration Recorder to be present. Use of depends_on (as shown below) is recommended to avoid race conditions.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var bucket = new Aws.S3.Bucket(&quot;bucket&quot;, new Aws.S3.BucketArgs
    {
        ForceDestroy = true,
    });
    var fooDeliveryChannel = new Aws.Cfg.DeliveryChannel(&quot;fooDeliveryChannel&quot;, new Aws.Cfg.DeliveryChannelArgs
    {
        S3BucketName = bucket.BucketName,
    });
    var role = new Aws.Iam.Role(&quot;role&quot;, new Aws.Iam.RoleArgs
    {
        AssumeRolePolicy = @&quot;{
&quot;&quot;Version&quot;&quot;: &quot;&quot;2012-10-17&quot;&quot;,
&quot;&quot;Statement&quot;&quot;: [
{
  &quot;&quot;Action&quot;&quot;: &quot;&quot;sts:AssumeRole&quot;&quot;,
  &quot;&quot;Principal&quot;&quot;: {
    &quot;&quot;Service&quot;&quot;: &quot;&quot;config.amazonaws.com&quot;&quot;
  },
  &quot;&quot;Effect&quot;&quot;: &quot;&quot;Allow&quot;&quot;,
  &quot;&quot;Sid&quot;&quot;: &quot;&quot;&quot;&quot;
}
]
}

&quot;,
    });
    var fooRecorder = new Aws.Cfg.Recorder(&quot;fooRecorder&quot;, new Aws.Cfg.RecorderArgs
    {
        RoleArn = role.Arn,
    });
    var rolePolicy = new Aws.Iam.RolePolicy(&quot;rolePolicy&quot;, new Aws.Iam.RolePolicyArgs
    {
        Policy = Output.Tuple(bucket.Arn, bucket.Arn).Apply(values =>
        {
            var bucketArn = values.Item1;
            var bucketArn1 = values.Item2;
            return @$&quot;{{
&quot;&quot;Version&quot;&quot;: &quot;&quot;2012-10-17&quot;&quot;,
&quot;&quot;Statement&quot;&quot;: [
{{
  &quot;&quot;Action&quot;&quot;: [
    &quot;&quot;s3:*&quot;&quot;
  ],
  &quot;&quot;Effect&quot;&quot;: &quot;&quot;Allow&quot;&quot;,
  &quot;&quot;Resource&quot;&quot;: [
    &quot;&quot;{bucketArn}&quot;&quot;,
    &quot;&quot;{bucketArn1}/*&quot;&quot;
  ]
}}
]
}}

&quot;;
        }),
        Role = role.Id,
    });
}

}

OrganizationCustomRule

Manages a Config Organization Custom Rule. More information about these rules can be found in the Enabling AWS Config Rules Across all Accounts in Your Organization and AWS Config Managed Rules documentation. For working with Organization Managed Rules (those invoking an AWS managed rule), see the aws_config_organization_managed__rule resource.

NOTE: This resource must be created in the Organization master account and rules will include the master account unless its ID is added to the excluded_accounts argument.

NOTE: The proper Lambda permission to allow the AWS Config service invoke the Lambda Function must be in place before the rule will successfully create or update. See also the aws.lambda.Permission resource.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var examplePermission = new Aws.Lambda.Permission(&quot;examplePermission&quot;, new Aws.Lambda.PermissionArgs
    {
        Action = &quot;lambda:InvokeFunction&quot;,
        Function = aws_lambda_function.Example.Arn,
        Principal = &quot;config.amazonaws.com&quot;,
    });
    var exampleOrganization = new Aws.Organizations.Organization(&quot;exampleOrganization&quot;, new Aws.Organizations.OrganizationArgs
    {
        AwsServiceAccessPrincipals = 
        {
            &quot;config-multiaccountsetup.amazonaws.com&quot;,
        },
        FeatureSet = &quot;ALL&quot;,
    });
    var exampleOrganizationCustomRule = new Aws.Cfg.OrganizationCustomRule(&quot;exampleOrganizationCustomRule&quot;, new Aws.Cfg.OrganizationCustomRuleArgs
    {
        LambdaFunctionArn = aws_lambda_function.Example.Arn,
        TriggerTypes = 
        {
            &quot;ConfigurationItemChangeNotification&quot;,
        },
    });
}

}

OrganizationManagedRule

Manages a Config Organization Managed Rule. More information about these rules can be found in the Enabling AWS Config Rules Across all Accounts in Your Organization and AWS Config Managed Rules documentation. For working with Organization Custom Rules (those invoking a custom Lambda Function), see the aws.cfg.OrganizationCustomRule resource.

NOTE: This resource must be created in the Organization master account and rules will include the master account unless its ID is added to the excluded_accounts argument.

NOTE: Every Organization account except those configured in the excluded_accounts argument must have a Configuration Recorder with proper IAM permissions before the rule will successfully create or update. See also the aws.cfg.Recorder resource.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var exampleOrganization = new Aws.Organizations.Organization(&quot;exampleOrganization&quot;, new Aws.Organizations.OrganizationArgs
    {
        AwsServiceAccessPrincipals = 
        {
            &quot;config-multiaccountsetup.amazonaws.com&quot;,
        },
        FeatureSet = &quot;ALL&quot;,
    });
    var exampleOrganizationManagedRule = new Aws.Cfg.OrganizationManagedRule(&quot;exampleOrganizationManagedRule&quot;, new Aws.Cfg.OrganizationManagedRuleArgs
    {
        RuleIdentifier = &quot;IAM_PASSWORD_POLICY&quot;,
    });
}

}

Recorder

Provides an AWS Config Configuration Recorder. Please note that this resource does not start the created recorder automatically.

Note: Starting the Configuration Recorder requires a delivery channel (while delivery channel creation requires Configuration Recorder). This is why aws.cfg.RecorderStatus is a separate resource.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var role = new Aws.Iam.Role(&quot;role&quot;, new Aws.Iam.RoleArgs
    {
        AssumeRolePolicy = @&quot;{
&quot;&quot;Version&quot;&quot;: &quot;&quot;2012-10-17&quot;&quot;,
&quot;&quot;Statement&quot;&quot;: [
{
  &quot;&quot;Action&quot;&quot;: &quot;&quot;sts:AssumeRole&quot;&quot;,
  &quot;&quot;Principal&quot;&quot;: {
    &quot;&quot;Service&quot;&quot;: &quot;&quot;config.amazonaws.com&quot;&quot;
  },
  &quot;&quot;Effect&quot;&quot;: &quot;&quot;Allow&quot;&quot;,
  &quot;&quot;Sid&quot;&quot;: &quot;&quot;&quot;&quot;
}
]
}

&quot;,
    });
    var foo = new Aws.Cfg.Recorder(&quot;foo&quot;, new Aws.Cfg.RecorderArgs
    {
        RoleArn = role.Arn,
    });
}

}
