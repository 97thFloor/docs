Class IpSet

Provides a WAFv2 IP Set Resource

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var example = new Aws.WafV2.IpSet(&quot;example&quot;, new Aws.WafV2.IpSetArgs
    {
        Addresses = 
        {
            &quot;1.2.3.4/32&quot;,
            &quot;5.6.7.8/32&quot;,
        },
        Description = &quot;Example IP set&quot;,
        IpAddressVersion = &quot;IPV4&quot;,
        Scope = &quot;REGIONAL&quot;,
        Tags = 
        {
            { &quot;Tag1&quot;, &quot;Value1&quot; },
            { &quot;Tag2&quot;, &quot;Value2&quot; },
        },
    });
}

}
