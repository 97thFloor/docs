Class Integration

Manages an Amazon API Gateway Version 2 integration. More information can be found in the Amazon API Gateway Developer Guide.

Example Usage

Basic

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var example = new Aws.ApiGatewayV2.Integration(&quot;example&quot;, new Aws.ApiGatewayV2.IntegrationArgs
    {
        ApiId = aws_apigatewayv2_api.Example.Id,
        IntegrationType = &quot;MOCK&quot;,
    });
}

}

Lambda Integration

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var exampleFunction = new Aws.Lambda.Function(&quot;exampleFunction&quot;, new Aws.Lambda.FunctionArgs
    {
        Code = new FileArchive(&quot;example.zip&quot;),
        Handler = &quot;index.handler&quot;,
        Role = aws_iam_role.Example.Arn,
        Runtime = &quot;nodejs10.x&quot;,
    });
    var exampleIntegration = new Aws.ApiGatewayV2.Integration(&quot;exampleIntegration&quot;, new Aws.ApiGatewayV2.IntegrationArgs
    {
        ApiId = aws_apigatewayv2_api.Example.Id,
        ConnectionType = &quot;INTERNET&quot;,
        ContentHandlingStrategy = &quot;CONVERT_TO_TEXT&quot;,
        Description = &quot;Lambda example&quot;,
        IntegrationMethod = &quot;POST&quot;,
        IntegrationType = &quot;AWS&quot;,
        IntegrationUri = exampleFunction.InvokeArn,
        PassthroughBehavior = &quot;WHEN_NO_MATCH&quot;,
    });
}

}
