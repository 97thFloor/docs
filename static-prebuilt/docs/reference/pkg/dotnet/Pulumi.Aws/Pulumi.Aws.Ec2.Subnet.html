Class Subnet

Provides an VPC subnet resource.

NOTE: Due to AWS Lambda improved VPC networking changes that began deploying in September 2019, subnets associated with Lambda Functions can take up to 45 minutes to successfully delete.

Example Usage

Basic Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var main = new Aws.Ec2.Subnet(&quot;main&quot;, new Aws.Ec2.SubnetArgs
    {
        CidrBlock = &quot;10.0.1.0/24&quot;,
        Tags = 
        {
            { &quot;Name&quot;, &quot;Main&quot; },
        },
        VpcId = aws_vpc.Main.Id,
    });
}

}

Subnets In Secondary VPC CIDR Blocks

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var secondaryCidr = new Aws.Ec2.VpcIpv4CidrBlockAssociation(&quot;secondaryCidr&quot;, new Aws.Ec2.VpcIpv4CidrBlockAssociationArgs
    {
        CidrBlock = &quot;172.2.0.0/16&quot;,
        VpcId = aws_vpc.Main.Id,
    });
    var inSecondaryCidr = new Aws.Ec2.Subnet(&quot;inSecondaryCidr&quot;, new Aws.Ec2.SubnetArgs
    {
        CidrBlock = &quot;172.2.0.0/24&quot;,
        VpcId = secondaryCidr.VpcId,
    });
}

}
