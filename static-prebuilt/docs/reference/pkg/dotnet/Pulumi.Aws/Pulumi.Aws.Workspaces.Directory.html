Class Directory

Provides a directory registration in AWS WorkSpaces Service

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var mainVpc = new Aws.Ec2.Vpc(&quot;mainVpc&quot;, new Aws.Ec2.VpcArgs
    {
        CidrBlock = &quot;10.0.0.0/16&quot;,
    });
    var private_a = new Aws.Ec2.Subnet(&quot;private-a&quot;, new Aws.Ec2.SubnetArgs
    {
        AvailabilityZone = &quot;us-east-1a&quot;,
        CidrBlock = &quot;10.0.0.0/24&quot;,
        VpcId = mainVpc.Id,
    });
    var private_b = new Aws.Ec2.Subnet(&quot;private-b&quot;, new Aws.Ec2.SubnetArgs
    {
        AvailabilityZone = &quot;us-east-1b&quot;,
        CidrBlock = &quot;10.0.1.0/24&quot;,
        VpcId = mainVpc.Id,
    });
    var mainDirectory = new Aws.DirectoryService.Directory(&quot;mainDirectory&quot;, new Aws.DirectoryService.DirectoryArgs
    {
        Password = &quot;#S1ncerely&quot;,
        Size = &quot;Small&quot;,
        VpcSettings = new Aws.DirectoryService.Inputs.DirectoryVpcSettingsArgs
        {
            SubnetIds = 
            {
                private_a.Id,
                private_b.Id,
            },
            VpcId = mainVpc.Id,
        },
    });
    var mainWorkspaces_directoryDirectory = new Aws.Workspaces.Directory(&quot;mainWorkspaces/directoryDirectory&quot;, new Aws.Workspaces.DirectoryArgs
    {
        DirectoryId = mainDirectory.Id,
        SelfServicePermissions = new Aws.Workspaces.Inputs.DirectorySelfServicePermissionsArgs
        {
            IncreaseVolumeSize = true,
            RebuildWorkspace = true,
        },
    });
}

}
