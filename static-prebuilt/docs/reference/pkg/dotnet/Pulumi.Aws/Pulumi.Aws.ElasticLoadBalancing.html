AppCookieStickinessPolicy

Provides an application cookie stickiness policy, which allows an ELB to wed its sticky cookie's expiration to a cookie generated by your application.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var lb = new Aws.Elb.LoadBalancer(&quot;lb&quot;, new Aws.Elb.LoadBalancerArgs
    {
        AvailabilityZones = 
        {
            &quot;us-east-1a&quot;,
        },
        Listeners = 
        {
            new Aws.Elb.Inputs.LoadBalancerListenerArgs
            {
                InstancePort = 8000,
                InstanceProtocol = &quot;http&quot;,
                LbPort = 80,
                LbProtocol = &quot;http&quot;,
            },
        },
    });
    var foo = new Aws.Elb.AppCookieStickinessPolicy(&quot;foo&quot;, new Aws.Elb.AppCookieStickinessPolicyArgs
    {
        CookieName = &quot;MyAppCookie&quot;,
        LbPort = 80,
        LoadBalancer = lb.Name,
    });
}

Attachment

Attaches an EC2 instance to an Elastic Load Balancer (ELB). For attaching resources with Application Load Balancer (ALB) or Network Load Balancer (NLB), see the aws.lb.TargetGroupAttachment resource.

NOTE on ELB Instances and ELB Attachments: This provider currently provides both a standalone ELB Attachment resource (describing an instance attached to an ELB), and an Elastic Load Balancer resource with instances defined in-line. At this time you cannot use an ELB with in-line instances in conjunction with an ELB Attachment resource. Doing so will cause a conflict and will overwrite attachments.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    // Create a new load balancer attachment
    var baz = new Aws.Elb.Attachment(&quot;baz&quot;, new Aws.Elb.AttachmentArgs
    {
        Elb = aws_elb.Bar.Id,
        Instance = aws_instance.Foo.Id,
    });
}

}

ListenerPolicy

Attaches a load balancer policy to an ELB Listener.

Example Usage for Custom Policy

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var wu_tang = new Aws.Elb.LoadBalancer(&quot;wu-tang&quot;, new Aws.Elb.LoadBalancerArgs
    {
        AvailabilityZones = 
        {
            &quot;us-east-1a&quot;,
        },
        Listeners = 
        {
            new Aws.Elb.Inputs.LoadBalancerListenerArgs
            {
                InstancePort = 443,
                InstanceProtocol = &quot;http&quot;,
                LbPort = 443,
                LbProtocol = &quot;https&quot;,
                SslCertificateId = &quot;arn:aws:iam::000000000000:server-certificate/wu-tang.net&quot;,
            },
        },
        Tags = 
        {
            { &quot;Name&quot;, &quot;wu-tang&quot; },
        },
    });
    var wu_tang_ssl = new Aws.Elb.LoadBalancerPolicy(&quot;wu-tang-ssl&quot;, new Aws.Elb.LoadBalancerPolicyArgs
    {
        LoadBalancerName = wu_tang.Name,
        PolicyAttributes = 
        {
            new Aws.Elb.Inputs.LoadBalancerPolicyPolicyAttributeArgs
            {
                Name = &quot;ECDHE-ECDSA-AES128-GCM-SHA256&quot;,
                Value = &quot;true&quot;,
            },
            new Aws.Elb.Inputs.LoadBalancerPolicyPolicyAttributeArgs
            {
                Name = &quot;Protocol-TLSv1.2&quot;,
                Value = &quot;true&quot;,
            },
        },
        PolicyName = &quot;wu-tang-ssl&quot;,
        PolicyTypeName = &quot;SSLNegotiationPolicyType&quot;,
    });
    var wu_tang_listener_policies_443 = new Aws.Elb.ListenerPolicy(&quot;wu-tang-listener-policies-443&quot;, new Aws.Elb.ListenerPolicyArgs
    {
        LoadBalancerName = wu_tang.Name,
        LoadBalancerPort = 443,
        PolicyNames = 
        {
            wu_tang_ssl.PolicyName,
        },
    });
}

}

LoadBalancer

Provides an Elastic Load Balancer resource, also known as a "Classic Load Balancer" after the release of Application/Network Load Balancers.

NOTE on ELB Instances and ELB Attachments: This provider currently provides both a standalone ELB Attachment resource (describing an instance attached to an ELB), and an ELB resource with instances defined in-line. At this time you cannot use an ELB with in-line instances in conjunction with a ELB Attachment resources. Doing so will cause a conflict and will overwrite attachments.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    // Create a new load balancer
    var bar = new Aws.Elb.LoadBalancer(&quot;bar&quot;, new Aws.Elb.LoadBalancerArgs
    {
        AccessLogs = new Aws.Elb.Inputs.LoadBalancerAccessLogsArgs
        {
            Bucket = &quot;foo&quot;,
            BucketPrefix = &quot;bar&quot;,
            Interval = 60,
        },
        AvailabilityZones = 
        {
            &quot;us-west-2a&quot;,
            &quot;us-west-2b&quot;,
            &quot;us-west-2c&quot;,
        },
        ConnectionDraining = true,
        ConnectionDrainingTimeout = 400,
        CrossZoneLoadBalancing = true,
        HealthCheck = new Aws.Elb.Inputs.LoadBalancerHealthCheckArgs
        {
            HealthyThreshold = 2,
            Interval = 30,
            Target = &quot;HTTP:8000/&quot;,
            Timeout = 3,
            UnhealthyThreshold = 2,
        },
        IdleTimeout = 400,
        Instances = 
        {
            aws_instance.Foo.Id,
        },
        Listeners = 
        {
            new Aws.Elb.Inputs.LoadBalancerListenerArgs
            {
                InstancePort = 8000,
                InstanceProtocol = &quot;http&quot;,
                LbPort = 80,
                LbProtocol = &quot;http&quot;,
            },
            new Aws.Elb.Inputs.LoadBalancerListenerArgs
            {
                InstancePort = 8000,
                InstanceProtocol = &quot;http&quot;,
                LbPort = 443,
                LbProtocol = &quot;https&quot;,
                SslCertificateId = &quot;arn:aws:iam::123456789012:server-certificate/certName&quot;,
            },
        },
        Tags = 
        {
            { &quot;Name&quot;, &quot;foobar-elb&quot; },
        },
    });
}

}

Note on ECDSA Key Algorithm

If the ARN of the ssl_certificate_id that is pointed to references a certificate that was signed by an ECDSA key, note that ELB only supports the P256 and P384 curves. Using a certificate signed by a key using a different curve could produce the error ERR_SSL_VERSION_OR_CIPHER_MISMATCH in your browser.

SslNegotiationPolicy

Provides a load balancer SSL negotiation policy, which allows an ELB to control the ciphers and protocols that are supported during SSL negotiations between a client and a load balancer.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var lb = new Aws.Elb.LoadBalancer(&quot;lb&quot;, new Aws.Elb.LoadBalancerArgs
    {
        AvailabilityZones = 
        {
            &quot;us-east-1a&quot;,
        },
        Listeners = 
        {
            new Aws.Elb.Inputs.LoadBalancerListenerArgs
            {
                InstancePort = 8000,
                InstanceProtocol = &quot;https&quot;,
                LbPort = 443,
                LbProtocol = &quot;https&quot;,
                SslCertificateId = &quot;arn:aws:iam::123456789012:server-certificate/certName&quot;,
            },
        },
    });
    var foo = new Aws.Elb.SslNegotiationPolicy(&quot;foo&quot;, new Aws.Elb.SslNegotiationPolicyArgs
    {
        Attributes = 
        {
            new Aws.Elb.Inputs.SslNegotiationPolicyAttributeArgs
            {
                Name = &quot;Protocol-TLSv1&quot;,
                Value = &quot;false&quot;,
            },
            new Aws.Elb.Inputs.SslNegotiationPolicyAttributeArgs
            {
                Name = &quot;Protocol-TLSv1.1&quot;,
                Value = &quot;false&quot;,
            },
            new Aws.Elb.Inputs.SslNegotiationPolicyAttributeArgs
            {
                Name = &quot;Protocol-TLSv1.2&quot;,
                Value = &quot;true&quot;,
            },
            new Aws.Elb.Inputs.SslNegotiationPolicyAttributeArgs
            {
                Name = &quot;Server-Defined-Cipher-Order&quot;,
                Value = &quot;true&quot;,
            },
            new Aws.Elb.Inputs.SslNegotiationPolicyAttributeArgs
            {
                Name = &quot;ECDHE-RSA-AES128-GCM-SHA256&quot;,
                Value = &quot;true&quot;,
            },
            new Aws.Elb.Inputs.SslNegotiationPolicyAttributeArgs
            {
                Name = &quot;AES128-GCM-SHA256&quot;,
                Value = &quot;true&quot;,
            },
            new Aws.Elb.Inputs.SslNegotiationPolicyAttributeArgs
            {
                Name = &quot;EDH-RSA-DES-CBC3-SHA&quot;,
                Value = &quot;false&quot;,
            },
        },
        LbPort = 443,
        LoadBalancer = lb.Id,
    });
}

}
