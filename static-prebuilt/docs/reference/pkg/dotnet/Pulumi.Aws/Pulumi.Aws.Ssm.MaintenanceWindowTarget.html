Class MaintenanceWindowTarget

Provides an SSM Maintenance Window Target resource

Instance Target Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var window = new Aws.Ssm.MaintenanceWindow(&quot;window&quot;, new Aws.Ssm.MaintenanceWindowArgs
    {
        Cutoff = 1,
        Duration = 3,
        Schedule = &quot;cron(0 16 ? * TUE *)&quot;,
    });
    var target1 = new Aws.Ssm.MaintenanceWindowTarget(&quot;target1&quot;, new Aws.Ssm.MaintenanceWindowTargetArgs
    {
        Description = &quot;This is a maintenance window target&quot;,
        ResourceType = &quot;INSTANCE&quot;,
        Targets = 
        {
            new Aws.Ssm.Inputs.MaintenanceWindowTargetTargetArgs
            {
                Key = &quot;tag:Name&quot;,
                Values = 
                {
                    &quot;acceptance_test&quot;,
                },
            },
        },
        WindowId = window.Id,
    });
}

}
