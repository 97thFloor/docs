Class Bucket

Provides a S3 bucket resource.

Example Usage

Private Bucket w/ Tags

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var bucket = new Aws.S3.Bucket(&quot;bucket&quot;, new Aws.S3.BucketArgs
    {
        Acl = &quot;private&quot;,
        Tags = 
        {
            { &quot;Environment&quot;, &quot;Dev&quot; },
            { &quot;Name&quot;, &quot;My bucket&quot; },
        },
    });
}

}

Using object lifecycle

        Acl = &quot;private&quot;,
        LifecycleRules = 
        {
            new Aws.S3.Inputs.BucketLifecycleRuleArgs
            {
                Enabled = true,
                Expiration = new Aws.S3.Inputs.BucketLifecycleRuleExpirationArgs
                {
                    Days = 90,
                },
                Id = &quot;log&quot;,
                Prefix = &quot;log/&quot;,
                Tags = 
                {
                    { &quot;autoclean&quot;, &quot;true&quot; },
                    { &quot;rule&quot;, &quot;log&quot; },
                },
                Transition = 
                {

                    {
                        { &quot;days&quot;, 30 },
                        { &quot;storageClass&quot;, &quot;STANDARD_IA&quot; },
                    },

                    {
                        { &quot;days&quot;, 60 },
                        { &quot;storageClass&quot;, &quot;GLACIER&quot; },
                    },
                },
            },
            new Aws.S3.Inputs.BucketLifecycleRuleArgs
            {
                Enabled = true,
                Expiration = new Aws.S3.Inputs.BucketLifecycleRuleExpirationArgs
                {
                    Date = &quot;2016-01-12&quot;,
                },
                Id = &quot;tmp&quot;,
                Prefix = &quot;tmp/&quot;,
            },
        },
    });
    var versioningBucket = new Aws.S3.Bucket(&quot;versioningBucket&quot;, new Aws.S3.BucketArgs
    {
        Acl = &quot;private&quot;,
        LifecycleRules = 
        {
            new Aws.S3.Inputs.BucketLifecycleRuleArgs
            {
                Enabled = true,

                {
                    Days = 90,

                    {
                        { &quot;days&quot;, 30 },

                        { &quot;days&quot;, 60 },

                Prefix = &quot;config/&quot;,

        },
        Versioning = new Aws.S3.Inputs.BucketVersioningArgs
        {
            Enabled = true,
        },
    });

Using replication configuration

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var central = new Aws.Provider(&quot;central&quot;, new Aws.ProviderArgs
    {
        Region = &quot;eu-central-1&quot;,
    });
    var replicationRole = new Aws.Iam.Role(&quot;replicationRole&quot;, new Aws.Iam.RoleArgs
    {
        AssumeRolePolicy = @&quot;{
&quot;&quot;Version&quot;&quot;: &quot;&quot;2012-10-17&quot;&quot;,
&quot;&quot;Statement&quot;&quot;: [
{
  &quot;&quot;Action&quot;&quot;: &quot;&quot;sts:AssumeRole&quot;&quot;,
  &quot;&quot;Principal&quot;&quot;: {
    &quot;&quot;Service&quot;&quot;: &quot;&quot;s3.amazonaws.com&quot;&quot;
  },
  &quot;&quot;Effect&quot;&quot;: &quot;&quot;Allow&quot;&quot;,
  &quot;&quot;Sid&quot;&quot;: &quot;&quot;&quot;&quot;
}
]
}

&quot;,
    });
    var destination = new Aws.S3.Bucket(&quot;destination&quot;, new Aws.S3.BucketArgs
    {
        Region = &quot;eu-west-1&quot;,
        Versioning = new Aws.S3.Inputs.BucketVersioningArgs
        {
            Enabled = true,
        },
    });
    var bucket = new Aws.S3.Bucket(&quot;bucket&quot;, new Aws.S3.BucketArgs
    {
        Acl = &quot;private&quot;,
        Region = &quot;eu-central-1&quot;,
        ReplicationConfiguration = new Aws.S3.Inputs.BucketReplicationConfigurationArgs
        {
            Role = replicationRole.Arn,
            Rules = 
            {
                new Aws.S3.Inputs.BucketReplicationConfigurationRuleArgs
                {
                    Destination = new Aws.S3.Inputs.BucketReplicationConfigurationRuleDestinationArgs
                    {
                        Bucket = destination.Arn,
                        StorageClass = &quot;STANDARD&quot;,
                    },
                    Id = &quot;foobar&quot;,
                    Prefix = &quot;foo&quot;,
                    Status = &quot;Enabled&quot;,
                },
            },
        },
        Versioning = new Aws.S3.Inputs.BucketVersioningArgs
        {
            Enabled = true,
        },
    });
    var replicationPolicy = new Aws.Iam.Policy(&quot;replicationPolicy&quot;, new Aws.Iam.PolicyArgs
    {
        Policy = Output.Tuple(bucket.Arn, bucket.Arn, destination.Arn).Apply(values =>
        {
            var bucketArn = values.Item1;
            var bucketArn1 = values.Item2;
            var destinationArn = values.Item3;
            return @$&quot;{{
&quot;&quot;Version&quot;&quot;: &quot;&quot;2012-10-17&quot;&quot;,
&quot;&quot;Statement&quot;&quot;: [
{{
  &quot;&quot;Action&quot;&quot;: [
    &quot;&quot;s3:GetReplicationConfiguration&quot;&quot;,
    &quot;&quot;s3:ListBucket&quot;&quot;
  ],
  &quot;&quot;Effect&quot;&quot;: &quot;&quot;Allow&quot;&quot;,
  &quot;&quot;Resource&quot;&quot;: [
    &quot;&quot;{bucketArn}&quot;&quot;
  ]
}},
{{
  &quot;&quot;Action&quot;&quot;: [
    &quot;&quot;s3:GetObjectVersion&quot;&quot;,
    &quot;&quot;s3:GetObjectVersionAcl&quot;&quot;
  ],
  &quot;&quot;Effect&quot;&quot;: &quot;&quot;Allow&quot;&quot;,
  &quot;&quot;Resource&quot;&quot;: [
    &quot;&quot;{bucketArn1}/*&quot;&quot;
  ]
}},
{{
  &quot;&quot;Action&quot;&quot;: [
    &quot;&quot;s3:ReplicateObject&quot;&quot;,
    &quot;&quot;s3:ReplicateDelete&quot;&quot;
  ],
  &quot;&quot;Effect&quot;&quot;: &quot;&quot;Allow&quot;&quot;,
  &quot;&quot;Resource&quot;&quot;: &quot;&quot;{destinationArn}/*&quot;&quot;
}}
]
}}

&quot;;
        }),
    });
    var replicationRolePolicyAttachment = new Aws.Iam.RolePolicyAttachment(&quot;replicationRolePolicyAttachment&quot;, new Aws.Iam.RolePolicyAttachmentArgs
    {
        PolicyArn = replicationPolicy.Arn,
        Role = replicationRole.Name,
    });
}

Enable Default Server Side Encryption

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var mykey = new Aws.Kms.Key(&quot;mykey&quot;, new Aws.Kms.KeyArgs
    {
        DeletionWindowInDays = 10,
        Description = &quot;This key is used to encrypt bucket objects&quot;,
    });
    var mybucket = new Aws.S3.Bucket(&quot;mybucket&quot;, new Aws.S3.BucketArgs
    {
        ServerSideEncryptionConfiguration = new Aws.S3.Inputs.BucketServerSideEncryptionConfigurationArgs
        {
            Rule = new Aws.S3.Inputs.BucketServerSideEncryptionConfigurationRuleArgs
            {
                ApplyServerSideEncryptionByDefault = new Aws.S3.Inputs.BucketServerSideEncryptionConfigurationRuleApplyServerSideEncryptionByDefaultArgs
                {
                    KmsMasterKeyId = mykey.Arn,
                    SseAlgorithm = &quot;aws:kms&quot;,
                },
            },
        },
    });
}

}

Using ACL policy grants

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var currentUser = Output.Create(Aws.GetCanonicalUserId.InvokeAsync());
    var bucket = new Aws.S3.Bucket(&quot;bucket&quot;, new Aws.S3.BucketArgs
    {
        Grants = 
        {
            new Aws.S3.Inputs.BucketGrantArgs
            {
                Id = currentUser.Apply(currentUser => currentUser.Id),
                Permissions = 
                {
                    &quot;FULL_CONTROL&quot;,
                },
                Type = &quot;CanonicalUser&quot;,
            },
            new Aws.S3.Inputs.BucketGrantArgs
            {
                Permissions = 
                {
                    &quot;READ&quot;,
                    &quot;WRITE&quot;,
                },
                Type = &quot;Group&quot;,
                Uri = &quot;http://acs.amazonaws.com/groups/s3/LogDelivery&quot;,
            },
        },
    });
}

}
