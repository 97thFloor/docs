Cluster

Provides an Neptune Cluster Resource. A Cluster Resource defines attributes that are applied to the entire cluster of Neptune Cluster Instances.

Changes to a Neptune Cluster can occur when you manually change a parameter, such as backup_retention_period, and are reflected in the next maintenance window. Because of this, this provider may report a difference in its planning phase because a modification has not yet taken place. You can use the apply_immediately flag to instruct the service to apply the change immediately (see documentation below).

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var @default = new Aws.Neptune.Cluster(&quot;default&quot;, new Aws.Neptune.ClusterArgs
    {
        ApplyImmediately = true,
        BackupRetentionPeriod = 5,
        ClusterIdentifier = &quot;neptune-cluster-demo&quot;,
        Engine = &quot;neptune&quot;,
        IamDatabaseAuthenticationEnabled = true,
        PreferredBackupWindow = &quot;07:00-09:00&quot;,
        SkipFinalSnapshot = true,
    });
}

}

ClusterInstance

A Cluster Instance Resource defines attributes that are specific to a single instance in a Neptune Cluster.

You can simply add neptune instances and Neptune manages the replication. You can use the count meta-parameter to make multiple instances and join them all to the same Neptune Cluster, or you may specify different Cluster Instance resources with various instance_class sizes.

Example Usage

using System.Collections.Generic;
using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var @default = new Aws.Neptune.Cluster(&quot;default&quot;, new Aws.Neptune.ClusterArgs
    {
        ApplyImmediately = true,
        BackupRetentionPeriod = 5,
        ClusterIdentifier = &quot;neptune-cluster-demo&quot;,
        Engine = &quot;neptune&quot;,
        IamDatabaseAuthenticationEnabled = true,
        PreferredBackupWindow = &quot;07:00-09:00&quot;,
        SkipFinalSnapshot = true,
    });
    var example = new List&lt;Aws.Neptune.ClusterInstance>();
    for (var rangeIndex = 0; rangeIndex &lt; 2; rangeIndex++)
    {
        var range = new { Value = rangeIndex };
        example.Add(new Aws.Neptune.ClusterInstance($&quot;example-{range.Value}&quot;, new Aws.Neptune.ClusterInstanceArgs
        {
            ApplyImmediately = true,
            ClusterIdentifier = @default.Id,
            Engine = &quot;neptune&quot;,
            InstanceClass = &quot;db.r4.large&quot;,
        }));
    }
}

}

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var defaultCluster = new Aws.Neptune.Cluster(&quot;defaultCluster&quot;, new Aws.Neptune.ClusterArgs
    {
        ApplyImmediately = &quot;true&quot;,
        BackupRetentionPeriod = 5,
        ClusterIdentifier = &quot;neptune-cluster-demo&quot;,
        Engine = &quot;neptune&quot;,
        IamDatabaseAuthenticationEnabled = &quot;true&quot;,
        PreferredBackupWindow = &quot;07:00-09:00&quot;,
        SkipFinalSnapshot = true,
    });
    var example = new Aws.Neptune.ClusterInstance(&quot;example&quot;, new Aws.Neptune.ClusterInstanceArgs
    {
        ApplyImmediately = &quot;true&quot;,
        ClusterIdentifier = defaultCluster.Id,
        Engine = &quot;neptune&quot;,
        InstanceClass = &quot;db.r4.large&quot;,
    });
    var defaultTopic = new Aws.Sns.Topic(&quot;defaultTopic&quot;, new Aws.Sns.TopicArgs
    {
    });
    var defaultEventSubscription = new Aws.Neptune.EventSubscription(&quot;defaultEventSubscription&quot;, new Aws.Neptune.EventSubscriptionArgs
    {
        EventCategories = 
        {
            &quot;maintenance&quot;,
            &quot;availability&quot;,
            &quot;creation&quot;,
            &quot;backup&quot;,
            &quot;restoration&quot;,
            &quot;recovery&quot;,
            &quot;deletion&quot;,
            &quot;failover&quot;,
            &quot;failure&quot;,
            &quot;notification&quot;,
            &quot;configuration change&quot;,
            &quot;read replica&quot;,
        },
        SnsTopicArn = defaultTopic.Arn,
        SourceIds = 
        {
            example.Id,
        },
        SourceType = &quot;db-instance&quot;,
        Tags = 
        {
            { &quot;env&quot;, &quot;test&quot; },
        },
    });
}

}

Attributes

The following additional atttributes are provided:

  • id - The name of the Neptune event notification subscription.
  • arn - The Amazon Resource Name of the Neptune event notification subscription.
  • customer_aws_id - The AWS customer account associated with the Neptune event notification subscription.
