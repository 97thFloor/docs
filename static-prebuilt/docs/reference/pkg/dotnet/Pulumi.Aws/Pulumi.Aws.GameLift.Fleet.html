Class Fleet

Provides a Gamelift Fleet resource.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var example = new Aws.GameLift.Fleet(&quot;example&quot;, new Aws.GameLift.FleetArgs
    {
        BuildId = aws_gamelift_build.Example.Id,
        Ec2InstanceType = &quot;t2.micro&quot;,
        FleetType = &quot;ON_DEMAND&quot;,
        RuntimeConfiguration = new Aws.GameLift.Inputs.FleetRuntimeConfigurationArgs
        {
            ServerProcess = 
            {

                {
                    { &quot;concurrentExecutions&quot;, 1 },
                    { &quot;launchPath&quot;, &quot;C:\\game\\GomokuServer.exe&quot; },
                },
            },
        },
    });
}

}
