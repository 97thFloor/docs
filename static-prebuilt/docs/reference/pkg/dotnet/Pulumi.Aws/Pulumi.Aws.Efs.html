MountTarget

Provides an Elastic File System (EFS) mount target.

Example Usage

using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var foo = new Aws.Ec2.Vpc(&quot;foo&quot;, new Aws.Ec2.VpcArgs
    {
        CidrBlock = &quot;10.0.0.0/16&quot;,
    });
    var alphaSubnet = new Aws.Ec2.Subnet(&quot;alphaSubnet&quot;, new Aws.Ec2.SubnetArgs
    {
        AvailabilityZone = &quot;us-west-2a&quot;,
        CidrBlock = &quot;10.0.1.0/24&quot;,
        VpcId = foo.Id,
    });
    var alphaMountTarget = new Aws.Efs.MountTarget(&quot;alphaMountTarget&quot;, new Aws.Efs.MountTargetArgs
    {
        FileSystemId = aws_efs_file_system.Foo.Id,
        SubnetId = alphaSubnet.Id,
    });
}

}
