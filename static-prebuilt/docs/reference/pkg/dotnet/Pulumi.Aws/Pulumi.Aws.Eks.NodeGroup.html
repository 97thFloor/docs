Class NodeGroup

Manages an EKS Node Group, which can provision and optionally update an Auto Scaling Group of Kubernetes worker nodes compatible with EKS. Additional documentation about this functionality can be found in the EKS User Guide.

Example Usage

Example IAM Role for EKS Node Group

using System.Collections.Generic;
using System.Text.Json;
using Pulumi;
using Aws = Pulumi.Aws;

class MyStack : Stack
{
public MyStack()
{
    var example = new Aws.Iam.Role(&quot;example&quot;, new Aws.Iam.RoleArgs
    {
        AssumeRolePolicy = JsonSerializer.Serialize(new Dictionary&lt;string, object?>
        {
            { &quot;Statement&quot;, new[]
                {
                    new Dictionary&lt;string, object?>
                    {
                        { &quot;Action&quot;, &quot;sts:AssumeRole&quot; },
                        { &quot;Effect&quot;, &quot;Allow&quot; },
                        { &quot;Principal&quot;, new Dictionary&lt;string, object?>
                        {
                            { &quot;Service&quot;, &quot;ec2.amazonaws.com&quot; },
                        } },
                    },
                }
             },
            { &quot;Version&quot;, &quot;2012-10-17&quot; },
        }),
    });
    var example_AmazonEKSWorkerNodePolicy = new Aws.Iam.RolePolicyAttachment(&quot;example-AmazonEKSWorkerNodePolicy&quot;, new Aws.Iam.RolePolicyAttachmentArgs
    {
        PolicyArn = &quot;arn:aws:iam::aws:policy/AmazonEKSWorkerNodePolicy&quot;,
        Role = example.Name,
    });
    var example_AmazonEKSCNIPolicy = new Aws.Iam.RolePolicyAttachment(&quot;example-AmazonEKSCNIPolicy&quot;, new Aws.Iam.RolePolicyAttachmentArgs
    {
        PolicyArn = &quot;arn:aws:iam::aws:policy/AmazonEKS_CNI_Policy&quot;,
        Role = example.Name,
    });
    var example_AmazonEC2ContainerRegistryReadOnly = new Aws.Iam.RolePolicyAttachment(&quot;example-AmazonEC2ContainerRegistryReadOnly&quot;, new Aws.Iam.RolePolicyAttachmentArgs
    {
        PolicyArn = &quot;arn:aws:iam::aws:policy/AmazonEC2ContainerRegistryReadOnly&quot;,
        Role = example.Name,
    });
}
