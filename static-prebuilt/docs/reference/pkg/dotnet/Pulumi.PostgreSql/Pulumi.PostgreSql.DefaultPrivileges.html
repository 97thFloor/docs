Class DefaultPrivileges

The postgresql..DefaultPrivileges resource creates and manages default privileges given to a user for a database schema.

Note: This resource needs Postgresql version 9 or above.

Usage

using Pulumi;
using PostgreSql = Pulumi.PostgreSql;

class MyStack : Stack
{
public MyStack()
{
    var readOnlyTables = new PostgreSql.DefaultPrivileges(&quot;readOnlyTables&quot;, new PostgreSql.DefaultPrivilegesArgs
    {
        Database = &quot;test_db&quot;,
        ObjectType = &quot;table&quot;,
        Owner = &quot;db_owner&quot;,
        Privileges = 
        {
            &quot;SELECT&quot;,
        },
        Role = &quot;test_role&quot;,
        Schema = &quot;public&quot;,
    });
}

}
