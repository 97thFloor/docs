Class ServiceJira

This resource allows you to manage Jira integration.

Example Usage

using Pulumi;
using GitLab = Pulumi.GitLab;

class MyStack : Stack
{
public MyStack()
{
    var awesomeProject = new GitLab.Project(&quot;awesomeProject&quot;, new GitLab.ProjectArgs
    {
        Description = &quot;My awesome project.&quot;,
        VisibilityLevel = &quot;public&quot;,
    });
    var jira = new GitLab.ServiceJira(&quot;jira&quot;, new GitLab.ServiceJiraArgs
    {
        Password = &quot;mypass&quot;,
        Project = awesomeProject.Id,
        Url = &quot;https://jira.example.com&quot;,
        Username = &quot;user&quot;,
    });
}

}
