Class DeployToken

This resource allows you to create and manage deploy token for your GitLab projects and groups.

Example Usage - Project

using Pulumi;
using GitLab = Pulumi.GitLab;

class MyStack : Stack
{
public MyStack()
{
    var example = new GitLab.DeployToken(&quot;example&quot;, new GitLab.DeployTokenArgs
    {
        ExpiresAt = &quot;2020-03-14T00:00:00.000Z&quot;,
        Project = &quot;example/deploying&quot;,
        Scopes = 
        {
            &quot;read_repository&quot;,
            &quot;read_registry&quot;,
        },
        Username = &quot;example-username&quot;,
    });
}

}
