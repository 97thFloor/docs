Binding

The rabbitmq..Binding resource creates and manages a binding relationship between a queue an exchange.

Example Usage

using Pulumi;
using RabbitMQ = Pulumi.RabbitMQ;

class MyStack : Stack
{
public MyStack()
{
    var testVHost = new RabbitMQ.VHost(&quot;testVHost&quot;, new RabbitMQ.VHostArgs
    {
    });
    var guest = new RabbitMQ.Permissions(&quot;guest&quot;, new RabbitMQ.PermissionsArgs
    {
        Permissions = new RabbitMQ.Inputs.PermissionsPermissionsArgs
        {
            Configure = &quot;.*&quot;,
            Read = &quot;.*&quot;,
            Write = &quot;.*&quot;,
        },
        User = &quot;guest&quot;,
        Vhost = testVHost.Name,
    });
    var testExchange = new RabbitMQ.Exchange(&quot;testExchange&quot;, new RabbitMQ.ExchangeArgs
    {
        Settings = new RabbitMQ.Inputs.ExchangeSettingsArgs
        {
            AutoDelete = true,
            Durable = false,
            Type = &quot;fanout&quot;,
        },
        Vhost = guest.Vhost,
    });
    var testQueue = new RabbitMQ.Queue(&quot;testQueue&quot;, new RabbitMQ.QueueArgs
    {
        Settings = new RabbitMQ.Inputs.QueueSettingsArgs
        {
            AutoDelete = false,
            Durable = true,
        },
        Vhost = guest.Vhost,
    });
    var testBinding = new RabbitMQ.Binding(&quot;testBinding&quot;, new RabbitMQ.BindingArgs
    {
        Destination = testQueue.Name,
        DestinationType = &quot;queue&quot;,
        RoutingKey = &quot;#&quot;,
        Source = testExchange.Name,
        Vhost = testVHost.Name,
    });
}

}
