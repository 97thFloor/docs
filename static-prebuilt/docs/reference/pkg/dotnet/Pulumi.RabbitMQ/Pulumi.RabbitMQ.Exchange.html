Class Exchange

The rabbitmq..Exchange resource creates and manages an exchange.

Example Usage

using Pulumi;
using RabbitMQ = Pulumi.RabbitMQ;

class MyStack : Stack
{
public MyStack()
{
    var testVHost = new RabbitMQ.VHost(&quot;testVHost&quot;, new RabbitMQ.VHostArgs
    {
    });
    var guest = new RabbitMQ.Permissions(&quot;guest&quot;, new RabbitMQ.PermissionsArgs
    {
        Permissions = new RabbitMQ.Inputs.PermissionsPermissionsArgs
        {
            Configure = &quot;.*&quot;,
            Read = &quot;.*&quot;,
            Write = &quot;.*&quot;,
        },
        User = &quot;guest&quot;,
        Vhost = testVHost.Name,
    });
    var testExchange = new RabbitMQ.Exchange(&quot;testExchange&quot;, new RabbitMQ.ExchangeArgs
    {
        Settings = new RabbitMQ.Inputs.ExchangeSettingsArgs
        {
            AutoDelete = true,
            Durable = false,
            Type = &quot;fanout&quot;,
        },
        Vhost = guest.Vhost,
    });
}

}
