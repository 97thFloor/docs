Cdn

Provides a DigitalOcean CDN Endpoint resource for use with Spaces.

Example Usage

Basic Example

using Pulumi;
using DigitalOcean = Pulumi.DigitalOcean;

class MyStack : Stack
{
public MyStack()
{
    // Create a new Spaces Bucket
    var mybucket = new DigitalOcean.SpacesBucket(&quot;mybucket&quot;, new DigitalOcean.SpacesBucketArgs
    {
        Region = &quot;sfo2&quot;,
        Acl = &quot;public-read&quot;,
    });
    // Add a CDN endpoint to the Spaces Bucket
    var mycdn = new DigitalOcean.Cdn(&quot;mycdn&quot;, new DigitalOcean.CdnArgs
    {
        Origin = mybucket.BucketDomainName,
    });
    this.Fqdn = mycdn.Endpoint;
}

[Output(&quot;fqdn&quot;)]
public Output&lt;string> Fqdn { get; set; }
}

Custom Sub-Domain Example

using Pulumi;
using DigitalOcean = Pulumi.DigitalOcean;

class MyStack : Stack
{
public MyStack()
{
    // Create a new Spaces Bucket
    var mybucket = new DigitalOcean.SpacesBucket(&quot;mybucket&quot;, new DigitalOcean.SpacesBucketArgs
    {
        Region = &quot;sfo2&quot;,
        Acl = &quot;public-read&quot;,
    });
    // Create a DigitalOcean managed Let&apos;s Encrypt Certificate
    var cert = new DigitalOcean.Certificate(&quot;cert&quot;, new DigitalOcean.CertificateArgs
    {
        Type = &quot;lets_encrypt&quot;,
        Domains = 
        {
            &quot;static.example.com&quot;,
        },
    });
    // Add a CDN endpoint with a custom sub-domain to the Spaces Bucket
    var mycdn = new DigitalOcean.Cdn(&quot;mycdn&quot;, new DigitalOcean.CdnArgs
    {
        Origin = mybucket.BucketDomainName,
        CustomDomain = &quot;static.example.com&quot;,
        CertificateId = cert.Id,
    });
}

}

DnsRecord

Provides a DigitalOcean DNS record resource.

Example Usage

using Pulumi;
using DigitalOcean = Pulumi.DigitalOcean;

class MyStack : Stack
{
public MyStack()
{
    var @default = new DigitalOcean.Domain(&quot;default&quot;, new DigitalOcean.DomainArgs
    {
        Name = &quot;example.com&quot;,
    });
    // Add an A record to the domain for www.example.com.
    var www = new DigitalOcean.DnsRecord(&quot;www&quot;, new DigitalOcean.DnsRecordArgs
    {
        Domain = @default.Name,
        Type = &quot;A&quot;,
        Value = &quot;192.168.0.11&quot;,
    });
    // Add a MX record for the example.com domain itself.
    var mx = new DigitalOcean.DnsRecord(&quot;mx&quot;, new DigitalOcean.DnsRecordArgs
    {
        Domain = @default.Name,
        Type = &quot;MX&quot;,
        Priority = 10,
        Value = &quot;mail.example.com.&quot;,
    });
    this.WwwFqdn = www.Fqdn;
    this.MxFqdn = mx.Fqdn;
}

[Output(&quot;wwwFqdn&quot;)]
public Output&lt;string> WwwFqdn { get; set; }
[Output(&quot;mxFqdn&quot;)]
public Output&lt;string> MxFqdn { get; set; }
}

FloatingIp

Provides a DigitalOcean Floating IP to represent a publicly-accessible static IP addresses that can be mapped to one of your Droplets.

NOTE: Floating IPs can be assigned to a Droplet either directly on the digitalocean..FloatingIp resource by setting a droplet_id or using the digitalocean..FloatingIpAssignment resource, but the two cannot be used together.

Example Usage

using Pulumi;
using DigitalOcean = Pulumi.DigitalOcean;

class MyStack : Stack
{
public MyStack()
{
    var foobarDroplet = new DigitalOcean.Droplet(&quot;foobarDroplet&quot;, new DigitalOcean.DropletArgs
    {
        Size = &quot;s-1vcpu-1gb&quot;,
        Image = &quot;ubuntu-18-04-x64&quot;,
        Region = &quot;sgp1&quot;,
        Ipv6 = true,
        PrivateNetworking = true,
    });
    var foobarFloatingIp = new DigitalOcean.FloatingIp(&quot;foobarFloatingIp&quot;, new DigitalOcean.FloatingIpArgs
    {
        DropletId = foobarDroplet.Id,
        Region = foobarDroplet.Region,
    });
}

}

SpacesBucket

Provides a bucket resource for Spaces, DigitalOcean's object storage product.

The Spaces API was designed to be interoperable with Amazon's AWS S3 API. This allows users to interact with the service while using the tools they already know. Spaces mirrors S3's authentication framework and requests to Spaces require a key pair similar to Amazon's Access ID and Secret Key.

The authentication requirement can be met by either setting the SPACES_ACCESS_KEY_ID and SPACES_SECRET_ACCESS_KEY environment variables or the provider's spaces_access_id and spaces_secret_key arguments to the access ID and secret you generate via the DigitalOcean control panel. For example:

using Pulumi;
using DigitalOcean = Pulumi.DigitalOcean;

class MyStack : Stack
{
public MyStack()
{
    var static_assets = new DigitalOcean.SpacesBucket(&quot;static-assets&quot;, new DigitalOcean.SpacesBucketArgs
    {
    });
    // ...
}

}

For more information, See An Introduction to DigitalOcean Spaces
