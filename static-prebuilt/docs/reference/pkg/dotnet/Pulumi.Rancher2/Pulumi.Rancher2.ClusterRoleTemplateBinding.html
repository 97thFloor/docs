Class ClusterRoleTemplateBinding

Provides a Rancher v2 Cluster Role Template Binding resource. This can be used to create Cluster Role Template Bindings for Rancher v2 environments and retrieve their information.

Example Usage

using Pulumi;
using Rancher2 = Pulumi.Rancher2;

class MyStack : Stack
{
public MyStack()
{
    // Create a new Rancher2 Cluster Role Template Binding
    var foo = new Rancher2.ClusterRoleTemplateBinding(&quot;foo&quot;, new Rancher2.ClusterRoleTemplateBindingArgs
    {
        ClusterId = &quot;&lt;cluster_id>&quot;,
        RoleTemplateId = &quot;&lt;role_template_id>&quot;,
        UserId = &quot;&lt;user_id>&quot;,
    });
}

}
