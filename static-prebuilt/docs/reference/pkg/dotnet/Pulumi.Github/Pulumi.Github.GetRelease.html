InvokeAsync(GetReleaseArgs, InvokeOptions)

Use this data source to retrieve information about a GitHub release in a specific repository.

{{% examples %}}

Example Usage

{{% example %}} To retrieve the latest release that is present in a repository:

using Pulumi;
using Github = Pulumi.Github;

class MyStack : Stack
{
public MyStack()
{
    var example = Output.Create(Github.GetRelease.InvokeAsync(new Github.GetReleaseArgs
    {
        Owner = &quot;example-owner&quot;,
        Repository = &quot;example-repository&quot;,
        RetrieveBy = &quot;latest&quot;,
    }));
}

}

To retrieve a specific release from a repository based on it's ID:

using Pulumi;
using Github = Pulumi.Github;

class MyStack : Stack
{
public MyStack()
{
    var example = Output.Create(Github.GetRelease.InvokeAsync(new Github.GetReleaseArgs
    {
        Id = 12345,
        Owner = &quot;example-owner&quot;,
        Repository = &quot;example-repository&quot;,
        RetrieveBy = &quot;id&quot;,
    }));
}

}

Finally, to retrieve a release based on it's tag:

using Pulumi;
using Github = Pulumi.Github;

class MyStack : Stack
{
public MyStack()
{
    var example = Output.Create(Github.GetRelease.InvokeAsync(new Github.GetReleaseArgs
    {
        Owner = &quot;example-owner&quot;,
        ReleaseTag = &quot;v1.0.0&quot;,
        Repository = &quot;example-repository&quot;,
        RetrieveBy = &quot;tag&quot;,
    }));
}

}
