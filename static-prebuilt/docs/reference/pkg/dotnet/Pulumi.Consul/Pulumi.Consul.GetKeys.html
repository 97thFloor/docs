InvokeAsync(GetKeysArgs, InvokeOptions)

The consul..Keys resource reads values from the Consul key/value store. This is a powerful way dynamically set values in templates.

{{% examples %}}

Example Usage

{{% example %}}

using Pulumi;
using Aws = Pulumi.Aws;
using Consul = Pulumi.Consul;

class MyStack : Stack
{
public MyStack()
{
    var appKeys = Output.Create(Consul.GetKeys.InvokeAsync(new Consul.GetKeysArgs
    {
        Datacenter = &quot;nyc1&quot;,
        Keys = 
        {
            new Consul.Inputs.GetKeysKeyArgs
            {
                Default = &quot;ami-1234&quot;,
                Name = &quot;ami&quot;,
                Path = &quot;service/app/launch_ami&quot;,
            },
        },
        Token = &quot;abcd&quot;,
    }));
    // Start our instance with the dynamic ami value
    var appInstance = new Aws.Ec2.Instance(&quot;appInstance&quot;, new Aws.Ec2.InstanceArgs
    {
        Ami = appKeys.Apply(appKeys => appKeys.Var.Ami),
    });
}

}
