Class Tenant

With this resource, you can manage Auth0 tenants, including setting logos and support contact information, setting error pages, and configuring default tenant behaviors.

Example Usage

using System.IO;
using Pulumi;
using Auth0 = Pulumi.Auth0;

class MyStack : Stack
{
public MyStack()
{
    var tenant = new Auth0.Tenant(&quot;tenant&quot;, new Auth0.TenantArgs
    {
        AllowedLogoutUrls = 
        {
            &quot;http://mysite/logout&quot;,
        },
        ChangePassword = new Auth0.Inputs.TenantChangePasswordArgs
        {
            Enabled = true,
            Html = File.ReadAllText(&quot;./password_reset.html&quot;),
        },
        DefaultAudience = &quot;&lt;client_id>&quot;,
        DefaultDirectory = &quot;Connection-Name&quot;,
        ErrorPage = new Auth0.Inputs.TenantErrorPageArgs
        {
            Html = File.ReadAllText(&quot;./error.html&quot;),
            ShowLogLink = true,
            Url = &quot;http://mysite/errors&quot;,
        },
        FriendlyName = &quot;Tenant Name&quot;,
        GuardianMfaPage = new Auth0.Inputs.TenantGuardianMfaPageArgs
        {
            Enabled = true,
            Html = File.ReadAllText(&quot;./guardian_multifactor.html&quot;),
        },
        PictureUrl = &quot;http://mysite/logo.png&quot;,
        SandboxVersion = &quot;8&quot;,
        SessionLifetime = 46000,
        SupportEmail = &quot;support@mysite&quot;,
        SupportUrl = &quot;http://mysite/support&quot;,
    });
}

}
