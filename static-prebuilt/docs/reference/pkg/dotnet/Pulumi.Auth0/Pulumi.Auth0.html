Client

With this resource, you can set up applications that use Auth0 for authentication and configure allowed callback URLs and secrets for these applications. Depending on your plan, you may also configure add-ons to allow your application to call another application's API (such as Firebase and AWS) on behalf of an authenticated user.

Example Usage

using Pulumi;
using Auth0 = Pulumi.Auth0;

class MyStack : Stack
{
public MyStack()
{
    var myClient = new Auth0.Client(&quot;myClient&quot;, new Auth0.ClientArgs
    {
        Addons = new Auth0.Inputs.ClientAddonsArgs
        {
            Firebase = 
            {
                { &quot;client_email&quot;, &quot;john.doe@example.com&quot; },
                { &quot;lifetime_in_seconds&quot;, 1 },
                { &quot;private_key&quot;, &quot;wer&quot; },
                { &quot;private_key_id&quot;, &quot;qwreerwerwe&quot; },
            },
            Samlp = new Auth0.Inputs.ClientAddonsSamlpArgs
            {
                Audience = &quot;https://example.com/saml&quot;,
                CreateUpnClaim = false,
                MapIdentities = false,
                MapUnknownClaimsAsIs = false,
                Mappings = 
                {
                    { &quot;email&quot;, &quot;http://schemas.xmlsoap.org/ws/2005/05/identity/claims/emailaddress&quot; },
                    { &quot;name&quot;, &quot;http://schemas.xmlsoap.org/ws/2005/05/identity/claims/name&quot; },
                },
                NameIdentifierFormat = &quot;urn:oasis:names:tc:SAML:2.0:nameid-format:persistent&quot;,
                NameIdentifierProbes = 
                {
                    &quot;http://schemas.xmlsoap.org/ws/2005/05/identity/claims/emailaddress&quot;,
                },
                PassthroughClaimsWithNoMapping = false,
            },
        },
        AllowedLogoutUrls = 
        {
            &quot;https://example.com&quot;,
        },
        AllowedOrigins = 
        {
            &quot;https://example.com&quot;,
        },
        AppType = &quot;non_interactive&quot;,
        Callbacks = 
        {
            &quot;https://example.com/callback&quot;,
        },
        ClientMetadata = 
        {
            { &quot;foo&quot;, &quot;zoo&quot; },
        },
        CustomLoginPageOn = true,
        Description = &quot;Test Applications Long Description&quot;,
        GrantTypes = 
        {
            &quot;authorization_code&quot;,
            &quot;http://auth0.com/oauth/grant-type/password-realm&quot;,
            &quot;implicit&quot;,
            &quot;password&quot;,
            &quot;refresh_token&quot;,
        },
        IsFirstParty = true,
        IsTokenEndpointIpHeaderTrusted = true,
        JwtConfiguration = new Auth0.Inputs.ClientJwtConfigurationArgs
        {
            Alg = &quot;RS256&quot;,
            LifetimeInSeconds = 300,
            Scopes = 
            {
                { &quot;foo&quot;, &quot;bar&quot; },
            },
            SecretEncoded = true,
        },
        Mobile = new Auth0.Inputs.ClientMobileArgs
        {
            Ios = new Auth0.Inputs.ClientMobileIosArgs
            {
                AppBundleIdentifier = &quot;com.my.bundle.id&quot;,
                TeamId = &quot;9JA89QQLNQ&quot;,
            },
        },
        OidcConformant = false,
        TokenEndpointAuthMethod = &quot;client_secret_post&quot;,
        WebOrigins = 
        {
            &quot;https://example.com&quot;,
        },
    });
}

}

ClientGrant

Auth0 uses various grant types, or methods by which you grant limited access to your resources to another entity without exposing credentials. The OAuth 2.0 protocol supports several types of grants, which allow different types of access. This resource allows you to create and manage client grants used with configured Auth0 clients.

Example Usage

using Pulumi;
using Auth0 = Pulumi.Auth0;

class MyStack : Stack
{
public MyStack()
{
    var myClient = new Auth0.Client(&quot;myClient&quot;, new Auth0.ClientArgs
    {
    });
    var myResourceServer = new Auth0.ResourceServer(&quot;myResourceServer&quot;, new Auth0.ResourceServerArgs
    {
        Identifier = &quot;https://api.example.com/client-grant&quot;,
        Scopes = 
        {
            new Auth0.Inputs.ResourceServerScopeArgs
            {
                Description = &quot;Create foos&quot;,
                Value = &quot;create:foo&quot;,
            },
            new Auth0.Inputs.ResourceServerScopeArgs
            {
                Description = &quot;Create bars&quot;,
                Value = &quot;create:bar&quot;,
            },
        },
    });
    var myClientGrant = new Auth0.ClientGrant(&quot;myClientGrant&quot;, new Auth0.ClientGrantArgs
    {
        Audience = myResourceServer.Identifier,
        ClientId = myClient.Id,
        Scopes = 
        {
            &quot;create:foo&quot;,
        },
    });
}

}

            Configuration = 
            {

            CustomScripts = 
            {
                { &quot;getUser&quot;, @&quot;function getByEmail (email, callback) {
return callback(new Error(&quot;&quot;Whoops!&quot;&quot;))
}

&quot; },

Hook

Hooks are secure, self-contained functions that allow you to customize the behavior of Auth0 when executed for selected extensibility points of the Auth0 platform. Auth0 invokes Hooks during runtime to execute your custom Node.js code.

Depending on the extensibility point, you can use Hooks with Database Connections and/or Passwordless Connections.

Example Usage

using Pulumi;
using Auth0 = Pulumi.Auth0;

class MyStack : Stack
{
public MyStack()
{
    var myHook = new Auth0.Hook(&quot;myHook&quot;, new Auth0.HookArgs
    {
        Enabled = true,
        Script = @&quot;function (user, context, callback) { 
callback(null, { user }); 
}

&quot;,
        TriggerId = &quot;pre-user-registration&quot;,
    });
}

}

Role

With this resource, you can created and manage collections of permissions that can be assigned to users, which are otherwise known as roles. Permissions (scopes) are created on auth0_resource_server, then associated with roles and optionally, users using this resource.

Example Usage

using Pulumi;
using Auth0 = Pulumi.Auth0;

class MyStack : Stack
{
public MyStack()
{
    var myResourceServer = new Auth0.ResourceServer(&quot;myResourceServer&quot;, new Auth0.ResourceServerArgs
    {
        EnforcePolicies = true,
        Identifier = &quot;my-resource-server-identifier&quot;,
        Scopes = 
        {
            new Auth0.Inputs.ResourceServerScopeArgs
            {
                Description = &quot;read something&quot;,
                Value = &quot;read:something&quot;,
            },
        },
        SigningAlg = &quot;RS256&quot;,
        SkipConsentForVerifiableFirstPartyClients = true,
        TokenLifetime = 86400,
    });
    var myRole = new Auth0.Role(&quot;myRole&quot;, new Auth0.RoleArgs
    {
        Description = &quot;Role Description...&quot;,
        Permissions = 
        {
            new Auth0.Inputs.RolePermissionArgs
            {
                Name = &quot;read:something&quot;,
                ResourceServerIdentifier = myResourceServer.Identifier,
            },
        },
    });
    var myUser = new Auth0.User(&quot;myUser&quot;, new Auth0.UserArgs
    {
        ConnectionName = &quot;Username-Password-Authentication&quot;,
        Email = &quot;test@test.com&quot;,
        Nickname = &quot;testnick&quot;,
        Password = &quot;passpass$$12$$12&quot;,
        Roles = 
        {
            myRole.Id,
        },
        UserId = &quot;auth0|1234567890&quot;,
        Username = &quot;testnick&quot;,
    });
}

}
