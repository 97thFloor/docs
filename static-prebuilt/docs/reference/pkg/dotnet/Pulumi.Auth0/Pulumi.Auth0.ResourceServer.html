Class ResourceServer

With this resource, you can set up APIs that can be consumed from your authorized applications.

Example Usage

using Pulumi;
using Auth0 = Pulumi.Auth0;

class MyStack : Stack
{
public MyStack()
{
    var myResourceServer = new Auth0.ResourceServer(&quot;myResourceServer&quot;, new Auth0.ResourceServerArgs
    {
        AllowOfflineAccess = true,
        Identifier = &quot;https://api.example.com&quot;,
        Scopes = 
        {
            new Auth0.Inputs.ResourceServerScopeArgs
            {
                Description = &quot;Create foos&quot;,
                Value = &quot;create:foo&quot;,
            },
            new Auth0.Inputs.ResourceServerScopeArgs
            {
                Description = &quot;Create bars&quot;,
                Value = &quot;create:bar&quot;,
            },
        },
        SigningAlg = &quot;RS256&quot;,
        SkipConsentForVerifiableFirstPartyClients = true,
        TokenLifetime = 8600,
    });
}

}
