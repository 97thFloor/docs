Class User

With this resource, you can manage user identities, including resetting passwords, and creating, provisioning, blocking, and deleting users.

Example Usage

using Pulumi;
using Auth0 = Pulumi.Auth0;

class MyStack : Stack
{
public MyStack()
{
    var admin = new Auth0.Role(&quot;admin&quot;, new Auth0.RoleArgs
    {
        Description = &quot;Administrator&quot;,
    });
    var user = new Auth0.User(&quot;user&quot;, new Auth0.UserArgs
    {
        ConnectionName = &quot;Username-Password-Authentication&quot;,
        UserId = &quot;12345&quot;,
        Username = &quot;unique_username&quot;,
        GivenName = &quot;Firstname&quot;,
        FamilyName = &quot;Lastname&quot;,
        Nickname = &quot;some.nickname&quot;,
        Email = &quot;test@test.com&quot;,
        EmailVerified = true,
        Password = &quot;passpass$12$12&quot;,
        Roles = 
        {
            admin.Id,
        },
    });
}

}
