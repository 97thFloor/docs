Class Prompt

With this resource, you can manage your Auth0 prompts, including choosing the login experience version.

Example Usage

using Pulumi;
using Auth0 = Pulumi.Auth0;

class MyStack : Stack
{
public MyStack()
{
    var example = new Auth0.Prompt(&quot;example&quot;, new Auth0.PromptArgs
    {
        UniversalLoginExperience = &quot;classic&quot;,
    });
}

}
