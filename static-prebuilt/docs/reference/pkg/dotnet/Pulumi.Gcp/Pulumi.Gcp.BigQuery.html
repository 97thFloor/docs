DatasetAccess

Gives dataset access for a single entity. This resource is intended to be used in cases where it is not possible to compile a full list of access blocks to include in a gcp.bigquery.Dataset resource, to enable them to be added separately.

Note: If this resource is used alongside a gcp.bigquery.Dataset resource, the dataset resource must either have no defined access blocks or a lifecycle block with ignore_changes = [access] so they don't fight over which accesses should be on the dataset.

To get more information about DatasetAccess, see:

  • API documentation
  • How-to Guides
  • Controlling access to datasets

Example Usage - Bigquery Dataset Access Basic User

using Pulumi;
using Gcp = Pulumi.Gcp;

class MyStack : Stack
{
public MyStack()
{
    var dataset = new Gcp.BigQuery.Dataset(&quot;dataset&quot;, new Gcp.BigQuery.DatasetArgs
    {
        DatasetId = &quot;example_dataset&quot;,
    });
    var bqowner = new Gcp.ServiceAccount.Account(&quot;bqowner&quot;, new Gcp.ServiceAccount.AccountArgs
    {
        AccountId = &quot;bqowner&quot;,
    });
    var access = new Gcp.BigQuery.DatasetAccess(&quot;access&quot;, new Gcp.BigQuery.DatasetAccessArgs
    {
        DatasetId = dataset.DatasetId,
        Role = &quot;OWNER&quot;,
        UserByEmail = bqowner.Email,
    });
}

}

Example Usage - Bigquery Dataset Access View

using Pulumi;
using Gcp = Pulumi.Gcp;

class MyStack : Stack
{
public MyStack()
{
    var @private = new Gcp.BigQuery.Dataset(&quot;private&quot;, new Gcp.BigQuery.DatasetArgs
    {
        DatasetId = &quot;example_dataset&quot;,
    });
    var publicDataset = new Gcp.BigQuery.Dataset(&quot;publicDataset&quot;, new Gcp.BigQuery.DatasetArgs
    {
        DatasetId = &quot;example_dataset2&quot;,
    });
    var publicTable = new Gcp.BigQuery.Table(&quot;publicTable&quot;, new Gcp.BigQuery.TableArgs
    {
        DatasetId = publicDataset.DatasetId,
        TableId = &quot;example_table&quot;,
        View = new Gcp.BigQuery.Inputs.TableViewArgs
        {
            Query = &quot;SELECT state FROM [lookerdata:cdc.project_tycho_reports]&quot;,
            UseLegacySql = false,
        },
    });
    var access = new Gcp.BigQuery.DatasetAccess(&quot;access&quot;, new Gcp.BigQuery.DatasetAccessArgs
    {
        DatasetId = @private.DatasetId,
        View = new Gcp.BigQuery.Inputs.DatasetAccessViewArgs
        {
            ProjectId = publicTable.Project,
            DatasetId = publicDataset.DatasetId,
            TableId = publicTable.TableId,
        },
    });
}

}

Example Usage - Scheduled Query

using Pulumi;
using Gcp = Pulumi.Gcp;

class MyStack : Stack
{
public MyStack()
{
    var project = Output.Create(Gcp.Organizations.GetProject.InvokeAsync());
    var permissions = new Gcp.Projects.IAMMember(&quot;permissions&quot;, new Gcp.Projects.IAMMemberArgs
    {
        Role = &quot;roles/iam.serviceAccountShortTermTokenMinter&quot;,
        Member = project.Apply(project => $&quot;serviceAccount:service-{project.Number}@gcp-sa-bigquerydatatransfer.iam.gserviceaccount.com&quot;),
    });
    var myDataset = new Gcp.BigQuery.Dataset(&quot;myDataset&quot;, new Gcp.BigQuery.DatasetArgs
    {
        DatasetId = &quot;my_dataset&quot;,
        FriendlyName = &quot;foo&quot;,
        Description = &quot;bar&quot;,
        Location = &quot;asia-northeast1&quot;,
    });
    var queryConfig = new Gcp.BigQuery.DataTransferConfig(&quot;queryConfig&quot;, new Gcp.BigQuery.DataTransferConfigArgs
    {
        DisplayName = &quot;my-query&quot;,
        Location = &quot;asia-northeast1&quot;,
        DataSourceId = &quot;scheduled_query&quot;,
        Schedule = &quot;first sunday of quarter 00:00&quot;,
        DestinationDatasetId = myDataset.DatasetId,
        Params = 
        {
            { &quot;destination_table_name_template&quot;, &quot;my-table&quot; },
            { &quot;write_disposition&quot;, &quot;WRITE_APPEND&quot; },
            { &quot;query&quot;, &quot;SELECT name FROM tabl WHERE x = &apos;y&apos;&quot; },
        },
    });
}

}

Job

Jobs are actions that BigQuery runs on your behalf to load data, export data, query data, or copy data. Once a BigQuery job is created, it cannot be changed or deleted.

Example Usage - Bigquery Job Query

using Pulumi;
using Gcp = Pulumi.Gcp;

class MyStack : Stack
{
public MyStack()
{
    var bar = new Gcp.BigQuery.Dataset(&quot;bar&quot;, new Gcp.BigQuery.DatasetArgs
    {
        DatasetId = &quot;job_query_dataset&quot;,
        FriendlyName = &quot;test&quot;,
        Description = &quot;This is a test description&quot;,
        Location = &quot;US&quot;,
    });
    var foo = new Gcp.BigQuery.Table(&quot;foo&quot;, new Gcp.BigQuery.TableArgs
    {
        DatasetId = bar.DatasetId,
        TableId = &quot;job_query_table&quot;,
    });
    var job = new Gcp.BigQuery.Job(&quot;job&quot;, new Gcp.BigQuery.JobArgs
    {
        JobId = &quot;job_query&quot;,
        Labels = 
        {
            { &quot;example-label&quot;, &quot;example-value&quot; },
        },
        Query = new Gcp.BigQuery.Inputs.JobQueryArgs
        {
            Query = &quot;SELECT state FROM [lookerdata:cdc.project_tycho_reports]&quot;,
            Destination_table = 
            {
                { &quot;projectId&quot;, foo.Project },
                { &quot;datasetId&quot;, foo.DatasetId },
                { &quot;tableId&quot;, foo.TableId },
            },
            AllowLargeResults = true,
            FlattenResults = true,
            Script_options = 
            {
                { &quot;keyResultStatement&quot;, &quot;LAST&quot; },
            },
        },
    });
}

}

Example Usage - Bigquery Job Extract

using Pulumi;
using Gcp = Pulumi.Gcp;

class MyStack : Stack
{
public MyStack()
{
    var source_oneDataset = new Gcp.BigQuery.Dataset(&quot;source-oneDataset&quot;, new Gcp.BigQuery.DatasetArgs
    {
        DatasetId = &quot;job_extract_dataset&quot;,
        FriendlyName = &quot;test&quot;,
        Description = &quot;This is a test description&quot;,
        Location = &quot;US&quot;,
    });
    var source_oneTable = new Gcp.BigQuery.Table(&quot;source-oneTable&quot;, new Gcp.BigQuery.TableArgs
    {
        DatasetId = source_oneDataset.DatasetId,
        TableId = &quot;job_extract_table&quot;,
        Schema = @&quot;[
{
&quot;&quot;name&quot;&quot;: &quot;&quot;name&quot;&quot;,
&quot;&quot;type&quot;&quot;: &quot;&quot;STRING&quot;&quot;,
&quot;&quot;mode&quot;&quot;: &quot;&quot;NULLABLE&quot;&quot;
},
{
&quot;&quot;name&quot;&quot;: &quot;&quot;post_abbr&quot;&quot;,
&quot;&quot;type&quot;&quot;: &quot;&quot;STRING&quot;&quot;,
&quot;&quot;mode&quot;&quot;: &quot;&quot;NULLABLE&quot;&quot;
},
{
&quot;&quot;name&quot;&quot;: &quot;&quot;date&quot;&quot;,
&quot;&quot;type&quot;&quot;: &quot;&quot;DATE&quot;&quot;,
&quot;&quot;mode&quot;&quot;: &quot;&quot;NULLABLE&quot;&quot;
}
]
&quot;,
    });
    var dest = new Gcp.Storage.Bucket(&quot;dest&quot;, new Gcp.Storage.BucketArgs
    {
        ForceDestroy = true,
    });
    var job = new Gcp.BigQuery.Job(&quot;job&quot;, new Gcp.BigQuery.JobArgs
    {
        JobId = &quot;job_extract&quot;,
        Extract = new Gcp.BigQuery.Inputs.JobExtractArgs
        {
            DestinationUris = 
            {
                dest.Url.Apply(url => $&quot;{url}/extract&quot;),
            },
            Source_table = 
            {
                { &quot;projectId&quot;, source_oneTable.Project },
                { &quot;datasetId&quot;, source_oneTable.DatasetId },
                { &quot;tableId&quot;, source_oneTable.TableId },
            },
            DestinationFormat = &quot;NEWLINE_DELIMITED_JSON&quot;,
            Compression = &quot;GZIP&quot;,
        },
    });
}
