Agent

A Dialogflow agent is a virtual agent that handles conversations with your end-users. It is a natural language understanding module that understands the nuances of human language. Dialogflow translates end-user text or audio during a conversation to structured data that your apps and services can understand. You design and build a Dialogflow agent to handle the types of conversations required for your system.

To get more information about Agent, see:

  • API documentation
  • How-to Guides
  • Official Documentation

Example Usage - Dialogflow Agent Full

using Pulumi;
using Gcp = Pulumi.Gcp;

class MyStack : Stack
{
public MyStack()
{
    var fullAgent = new Gcp.Diagflow.Agent(&quot;fullAgent&quot;, new Gcp.Diagflow.AgentArgs
    {
        ApiVersion = &quot;API_VERSION_V2_BETA_1&quot;,
        AvatarUri = &quot;https://cloud.google.com/_static/images/cloud/icons/favicons/onecloud/super_cloud.png&quot;,
        ClassificationThreshold = 0.3,
        DefaultLanguageCode = &quot;en&quot;,
        Description = &quot;Example description.&quot;,
        DisplayName = &quot;dialogflow-agent&quot;,
        EnableLogging = true,
        MatchMode = &quot;MATCH_MODE_ML_ONLY&quot;,
        SupportedLanguageCodes = 
        {
            &quot;fr&quot;,
            &quot;de&quot;,
            &quot;es&quot;,
        },
        Tier = &quot;TIER_STANDARD&quot;,
        TimeZone = &quot;America/New_York&quot;,
    });
}

}

Example Usage - Dialogflow Entity Type Basic

using Pulumi;
using Gcp = Pulumi.Gcp;

class MyStack : Stack
{
public MyStack()
{
    var basicAgent = new Gcp.Diagflow.Agent(&quot;basicAgent&quot;, new Gcp.Diagflow.AgentArgs
    {
        DisplayName = &quot;example_agent&quot;,
        DefaultLanguageCode = &quot;en&quot;,
        TimeZone = &quot;America/New_York&quot;,
    });
    var basicEntityType = new Gcp.Diagflow.EntityType(&quot;basicEntityType&quot;, new Gcp.Diagflow.EntityTypeArgs
    {
        DisplayName = &quot;&quot;,
        Kind = &quot;KIND_MAP&quot;,
        Entities = 
        {
            new Gcp.Diagflow.Inputs.EntityTypeEntityArgs
            {
                Value = &quot;value1&quot;,
                Synonyms = 
                {
                    &quot;synonym1&quot;,
                    &quot;synonym2&quot;,
                },
            },
            new Gcp.Diagflow.Inputs.EntityTypeEntityArgs
            {
                Value = &quot;value2&quot;,
                Synonyms = 
                {
                    &quot;synonym3&quot;,
                    &quot;synonym4&quot;,
                },
            },
        },
    });
}
