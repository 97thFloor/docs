Class Dashboard

A Google Stackdriver dashboard. Dashboards define the content and layout of pages in the Stackdriver web application.

To get more information about Dashboards, see:

  • API documentation
  • How-to Guides
  • Official Documentation

Example Usage - Monitoring Dashboard Basic

using Pulumi;
using Gcp = Pulumi.Gcp;

class MyStack : Stack
{
public MyStack()
{
    var dashboard = new Gcp.Monitoring.Dashboard(&quot;dashboard&quot;, new Gcp.Monitoring.DashboardArgs
    {
        DashboardJson = @&quot;{
&quot;&quot;displayName&quot;&quot;: &quot;&quot;Demo Dashboard&quot;&quot;,
&quot;&quot;gridLayout&quot;&quot;: {
&quot;&quot;widgets&quot;&quot;: [
  {
    &quot;&quot;blank&quot;&quot;: {}
  }
]
}
}


&quot;,
    });
}

}
