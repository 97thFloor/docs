BillingAccountBucketConfig

Manages a billing account level logging bucket config. For more information see the official logging documentation and Storing Logs.

Note: Logging buckets are automatically created for a given folder, project, organization, billingAccount and cannot be deleted. Creating a resource of this type will acquire and update the resource that already exists at the desired location. These buckets cannot be removed so deleting this resource will remove the bucket config from your state but will leave the logging bucket unchanged. The buckets that are currently automatically created are "_Default" and "_Required".

Example Usage

using Pulumi;
using Gcp = Pulumi.Gcp;

class MyStack : Stack
{
public MyStack()
{
    var @default = Output.Create(Gcp.Organizations.GetBillingAccount.InvokeAsync(new Gcp.Organizations.GetBillingAccountArgs
    {
        BillingAccount = &quot;00AA00-000AAA-00AA0A&quot;,
    }));
    var basic = new Gcp.Logging.BillingAccountBucketConfig(&quot;basic&quot;, new Gcp.Logging.BillingAccountBucketConfigArgs
    {
        BillingAccount = @default.Apply(@default => @default.BillingAccount),
        Location = &quot;global&quot;,
        RetentionDays = 30,
        BucketId = &quot;_Default&quot;,
    });
}

}

BillingAccountSink

Manages a billing account logging sink. For more information see the official documentation and Exporting Logs in the API.

Note You must have the "Logs Configuration Writer" IAM role (roles/logging.configWriter) granted on the billing account to the credentials used with this provider. IAM roles granted on a billing account are separate from the typical IAM roles granted on a project.

Example Usage

using Pulumi;
using Gcp = Pulumi.Gcp;

class MyStack : Stack
{
public MyStack()
{
    var log_bucket = new Gcp.Storage.Bucket(&quot;log-bucket&quot;, new Gcp.Storage.BucketArgs
    {
    });
    var my_sink = new Gcp.Logging.BillingAccountSink(&quot;my-sink&quot;, new Gcp.Logging.BillingAccountSinkArgs
    {
        BillingAccount = &quot;ABCDEF-012345-GHIJKL&quot;,
        Destination = log_bucket.Name.Apply(name => $&quot;storage.googleapis.com/{name}&quot;),
    });
    var log_writer = new Gcp.Projects.IAMBinding(&quot;log-writer&quot;, new Gcp.Projects.IAMBindingArgs
    {
        Role = &quot;roles/storage.objectCreator&quot;,
        Members = 
        {
            my_sink.WriterIdentity,
        },
    });
}

}

Metric

Logs-based metric can also be used to extract values from logs and create a a distribution of the values. The distribution records the statistics of the extracted values along with an optional histogram of the values as specified by the bucket options.

To get more information about Metric, see:

  • API documentation
  • How-to Guides
  • Official Documentation

Example Usage - Logging Metric Basic

using Pulumi;
using Gcp = Pulumi.Gcp;

class MyStack : Stack
{
public MyStack()
{
    var loggingMetric = new Gcp.Logging.Metric(&quot;loggingMetric&quot;, new Gcp.Logging.MetricArgs
    {
        BucketOptions = new Gcp.Logging.Inputs.MetricBucketOptionsArgs
        {
            LinearBuckets = new Gcp.Logging.Inputs.MetricBucketOptionsLinearBucketsArgs
            {
                NumFiniteBuckets = 3,
                Offset = 1,
                Width = 1,
            },
        },
        Filter = &quot;resource.type=gae_app AND severity>=ERROR&quot;,
        LabelExtractors = 
        {
            { &quot;mass&quot;, &quot;EXTRACT(jsonPayload.request)&quot; },
            { &quot;sku&quot;, &quot;EXTRACT(jsonPayload.id)&quot; },
        },
        MetricDescriptor = new Gcp.Logging.Inputs.MetricMetricDescriptorArgs
        {
            DisplayName = &quot;My metric&quot;,
            Labels = 
            {
                new Gcp.Logging.Inputs.MetricMetricDescriptorLabelArgs
                {
                    Description = &quot;amount of matter&quot;,
                    Key = &quot;mass&quot;,
                    ValueType = &quot;STRING&quot;,
                },
                new Gcp.Logging.Inputs.MetricMetricDescriptorLabelArgs
                {
                    Description = &quot;Identifying number for item&quot;,
                    Key = &quot;sku&quot;,
                    ValueType = &quot;INT64&quot;,
                },
            },
            MetricKind = &quot;DELTA&quot;,
            Unit = &quot;1&quot;,
            ValueType = &quot;DISTRIBUTION&quot;,
        },
        ValueExtractor = &quot;EXTRACT(jsonPayload.request)&quot;,
    });
}

}
