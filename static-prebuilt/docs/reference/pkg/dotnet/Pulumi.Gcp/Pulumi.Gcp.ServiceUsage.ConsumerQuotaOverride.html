Class ConsumerQuotaOverride

A consumer override is applied to the consumer on its own authority to limit its own quota usage. Consumer overrides cannot be used to grant more quota than would be allowed by admin overrides, producer overrides, or the default limit of the service.

Example Usage - Consumer Quota Override

using Pulumi;
using Gcp = Pulumi.Gcp;

class MyStack : Stack
{
public MyStack()
{
    var myProject = new Gcp.Organizations.Project(&quot;myProject&quot;, new Gcp.Organizations.ProjectArgs
    {
        ProjectId = &quot;quota&quot;,
        OrgId = &quot;123456789&quot;,
    });
    var @override = new Gcp.ServiceUsage.ConsumerQuotaOverride(&quot;override&quot;, new Gcp.ServiceUsage.ConsumerQuotaOverrideArgs
    {
        Project = myProject.ProjectId,
        Service = &quot;servicemanagement.googleapis.com&quot;,
        Metric = &quot;servicemanagement.googleapis.com%2Fdefault_requests&quot;,
        Limit = &quot;%2Fmin%2Fproject&quot;,
        OverrideValue = &quot;95&quot;,
        Force = true,
    });
}

}
