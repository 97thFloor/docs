Class MetricMetadata

Provides a Datadog metric_metadata resource. This can be used to manage a metric's metadata.

Example Usage

using Pulumi;
using Datadog = Pulumi.Datadog;

class MyStack : Stack
{
public MyStack()
{
    // Manage a Datadog metric&apos;s metadata
    var requestTime = new Datadog.MetricMetadata(&quot;requestTime&quot;, new Datadog.MetricMetadataArgs
    {
        Description = &quot;99th percentile request time in millseconds&quot;,
        Metric = &quot;request.time&quot;,
        ShortName = &quot;Request time&quot;,
        Type = &quot;gauge&quot;,
        Unit = &quot;millisecond&quot;,
    });
}

}
