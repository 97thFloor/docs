Class AuthBackendConfig

Manages an Kubernetes auth backend config in a Vault server. See the Vault documentation for more information.

Example Usage

using Pulumi;
using Vault = Pulumi.Vault;

class MyStack : Stack
{
public MyStack()
{
    var kubernetes = new Vault.AuthBackend(&quot;kubernetes&quot;, new Vault.AuthBackendArgs
    {
        Type = &quot;kubernetes&quot;,
    });
    var example = new Vault.Kubernetes.AuthBackendConfig(&quot;example&quot;, new Vault.Kubernetes.AuthBackendConfigArgs
    {
        Backend = kubernetes.Path,
        Issuer = &quot;api&quot;,
        KubernetesCaCert = @&quot;-----BEGIN CERTIFICATE-----
example
-----END CERTIFICATE-----
&quot;,
        KubernetesHost = &quot;http://example.com:443&quot;,
        TokenReviewerJwt = &quot;ZXhhbXBsZQo=&quot;,
    });
}

}
