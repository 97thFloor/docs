Class SecretBackendRole

Provides a resource to manage roles in an SSH secret backend SSH secret backend within Vault.

Example Usage

using Pulumi;
using Vault = Pulumi.Vault;

class MyStack : Stack
{
public MyStack()
{
    var example = new Vault.Mount(&quot;example&quot;, new Vault.MountArgs
    {
        Type = &quot;ssh&quot;,
    });
    var foo = new Vault.Ssh.SecretBackendRole(&quot;foo&quot;, new Vault.Ssh.SecretBackendRoleArgs
    {
        AllowUserCertificates = true,
        Backend = example.Path,
        KeyType = &quot;ca&quot;,
    });
    var bar = new Vault.Ssh.SecretBackendRole(&quot;bar&quot;, new Vault.Ssh.SecretBackendRoleArgs
    {
        AllowedUsers = &quot;default,baz&quot;,
        Backend = example.Path,
        CidrList = &quot;0.0.0.0/0&quot;,
        DefaultUser = &quot;default&quot;,
        KeyType = &quot;otp&quot;,
    });
}

}
