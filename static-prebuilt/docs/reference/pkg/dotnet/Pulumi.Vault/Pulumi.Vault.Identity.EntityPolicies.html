Class EntityPolicies

Manages policies for an Identity Entity for Vault. The Identity secrets engine is the identity management solution for Vault.

Example Usage

Exclusive Policies

using Pulumi;
using Vault = Pulumi.Vault;

class MyStack : Stack
{
public MyStack()
{
    var entity = new Vault.Identity.Entity(&quot;entity&quot;, new Vault.Identity.EntityArgs
    {
        ExternalPolicies = true,
    });
    var policies = new Vault.Identity.EntityPolicies(&quot;policies&quot;, new Vault.Identity.EntityPoliciesArgs
    {
        Policies = 
        {
            &quot;default&quot;,
            &quot;test&quot;,
        },
        Exclusive = true,
        EntityId = entity.Id,
    });
}

}

Non-exclusive Policies

using Pulumi;
using Vault = Pulumi.Vault;

    {
        Policies = 
        {
            &quot;default&quot;,
            &quot;test&quot;,
        },
        Exclusive = false,

    });
    var others = new Vault.Identity.EntityPolicies(&quot;others&quot;, new Vault.Identity.EntityPoliciesArgs
    {
        Policies = 
        {
            &quot;others&quot;,
        },
        Exclusive = false,
        EntityId = entity.Id,
    });
}

}
