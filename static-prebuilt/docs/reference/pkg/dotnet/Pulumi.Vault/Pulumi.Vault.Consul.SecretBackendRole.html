Class SecretBackendRole

Manages a Consul secrets role for a Consul secrets engine in Vault. Consul secret backends can then issue Consul tokens.

Example Usage

using Pulumi;
using Vault = Pulumi.Vault;

class MyStack : Stack
{
public MyStack()
{
    var test = new Vault.Consul.SecretBackend(&quot;test&quot;, new Vault.Consul.SecretBackendArgs
    {
        Path = &quot;consul&quot;,
        Description = &quot;Manages the Consul backend&quot;,
        Address = &quot;127.0.0.1:8500&quot;,
        Token = &quot;4240861b-ce3d-8530-115a-521ff070dd29&quot;,
    });
    var example = new Vault.Consul.SecretBackendRole(&quot;example&quot;, new Vault.Consul.SecretBackendRoleArgs
    {
        Backend = test.Path,
        Policies = 
        {
            &quot;example-policy&quot;,
        },
    });
}

}
