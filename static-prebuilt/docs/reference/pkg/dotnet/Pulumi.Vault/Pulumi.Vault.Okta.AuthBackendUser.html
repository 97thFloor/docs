Class AuthBackendUser

Provides a resource to create a user in an Okta auth backend within Vault.

Example Usage

using Pulumi;
using Vault = Pulumi.Vault;

class MyStack : Stack
{
public MyStack()
{
    var example = new Vault.Okta.AuthBackend(&quot;example&quot;, new Vault.Okta.AuthBackendArgs
    {
        Organization = &quot;dummy&quot;,
        Path = &quot;user_okta&quot;,
    });
    var foo = new Vault.Okta.AuthBackendUser(&quot;foo&quot;, new Vault.Okta.AuthBackendUserArgs
    {
        Groups = 
        {
            &quot;one&quot;,
            &quot;two&quot;,
        },
        Path = example.Path,
        Username = &quot;foo&quot;,
    });
}

}
