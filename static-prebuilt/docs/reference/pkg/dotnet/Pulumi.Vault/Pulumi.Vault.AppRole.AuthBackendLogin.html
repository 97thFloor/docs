Class AuthBackendLogin

Logs into Vault using the AppRole auth backend. See the Vault documentation for more information.

Example Usage

using Pulumi;
using Vault = Pulumi.Vault;

class MyStack : Stack
{
public MyStack()
{
    var approle = new Vault.AuthBackend(&quot;approle&quot;, new Vault.AuthBackendArgs
    {
        Type = &quot;approle&quot;,
    });
    var example = new Vault.AppRole.AuthBackendRole(&quot;example&quot;, new Vault.AppRole.AuthBackendRoleArgs
    {
        Backend = approle.Path,
        Policies = 
        {
            &quot;default&quot;,
            &quot;dev&quot;,
            &quot;prod&quot;,
        },
        RoleName = &quot;test-role&quot;,
    });
    var id = new Vault.AppRole.AuthBackendRoleSecretID(&quot;id&quot;, new Vault.AppRole.AuthBackendRoleSecretIDArgs
    {
        Backend = approle.Path,
        RoleName = example.RoleName,
    });
    var login = new Vault.AppRole.AuthBackendLogin(&quot;login&quot;, new Vault.AppRole.AuthBackendLoginArgs
    {
        Backend = approle.Path,
        RoleId = example.RoleId,
        SecretId = id.SecretId,
    });
}

}
