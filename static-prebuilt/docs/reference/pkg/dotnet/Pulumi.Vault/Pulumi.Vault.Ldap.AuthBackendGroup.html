Class AuthBackendGroup

Provides a resource to create a group in an LDAP auth backend within Vault.

Example Usage

using Pulumi;
using Vault = Pulumi.Vault;

class MyStack : Stack
{
public MyStack()
{
    var ldap = new Vault.Ldap.AuthBackend(&quot;ldap&quot;, new Vault.Ldap.AuthBackendArgs
    {
        Discoverdn = false,
        Groupdn = &quot;OU=Groups,DC=example,DC=org&quot;,
        Groupfilter = &quot;(&amp;(objectClass=group)(member:1.2.840.113556.1.4.1941:={{.UserDN}}))&quot;,
        Path = &quot;ldap&quot;,
        Upndomain = &quot;EXAMPLE.ORG&quot;,
        Url = &quot;ldaps://dc-01.example.org&quot;,
        Userattr = &quot;sAMAccountName&quot;,
        Userdn = &quot;OU=Users,OU=Accounts,DC=example,DC=org&quot;,
    });
    var @group = new Vault.Ldap.AuthBackendGroup(&quot;group&quot;, new Vault.Ldap.AuthBackendGroupArgs
    {
        Backend = ldap.Path,
        Groupname = &quot;dba&quot;,
        Policies = 
        {
            &quot;dba&quot;,
        },
    });
}

}
