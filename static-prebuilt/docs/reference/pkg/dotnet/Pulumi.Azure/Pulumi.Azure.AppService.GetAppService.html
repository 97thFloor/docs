InvokeAsync(GetAppServiceArgs, InvokeOptions)

Use this data source to access information about an existing App Service.

{{% examples %}}

Example Usage

{{% example %}}

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var example = Output.Create(Azure.AppService.GetAppService.InvokeAsync(new Azure.AppService.GetAppServiceArgs
    {
        Name = &quot;search-app-service&quot;,
        ResourceGroupName = &quot;search-service&quot;,
    }));
    this.AppServiceId = example.Apply(example => example.Id);
}

[Output(&quot;appServiceId&quot;)]
public Output&lt;string> AppServiceId { get; set; }
}
