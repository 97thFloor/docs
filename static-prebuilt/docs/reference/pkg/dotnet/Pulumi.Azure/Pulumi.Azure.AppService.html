ActiveSlot

Promotes an App Service Slot to Production within an App Service.

Note: When using Slots - the app_settings, connection_string and site_config blocks on the azure.appservice.AppService resource will be overwritten when promoting a Slot using the azure.appservice.ActiveSlot resource.

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;
using Random = Pulumi.Random;

class MyStack : Stack
{
public MyStack()
{
    var server = new Random.RandomId(&quot;server&quot;, new Random.RandomIdArgs
    {
    });
    // ...
    var exampleResourceGroup = new Azure.Core.ResourceGroup(&quot;exampleResourceGroup&quot;, new Azure.Core.ResourceGroupArgs
    {
    });
    // ...
    var examplePlan = new Azure.AppService.Plan(&quot;examplePlan&quot;, new Azure.AppService.PlanArgs
    {
    });
    // ...
    var exampleAppService = new Azure.AppService.AppService(&quot;exampleAppService&quot;, new Azure.AppService.AppServiceArgs
    {
    });
    // ...
    var exampleSlot = new Azure.AppService.Slot(&quot;exampleSlot&quot;, new Azure.AppService.SlotArgs
    {
    });
    // ...
    var exampleActiveSlot = new Azure.AppService.ActiveSlot(&quot;exampleActiveSlot&quot;, new Azure.AppService.ActiveSlotArgs
    {
        ResourceGroupName = exampleResourceGroup.Name,
        AppServiceName = exampleAppService.Name,
        AppServiceSlotName = exampleSlot.Name,
    });
}

}

CustomHostnameBinding

Manages a Hostname Binding within an App Service.

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;
using Random = Pulumi.Random;

class MyStack : Stack
{
public MyStack()
{
    var server = new Random.RandomId(&quot;server&quot;, new Random.RandomIdArgs
    {
        Keepers = 
        {
            { &quot;azi_id&quot;, 1 },
        },
        ByteLength = 8,
    });
    var exampleResourceGroup = new Azure.Core.ResourceGroup(&quot;exampleResourceGroup&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;West Europe&quot;,
    });
    var examplePlan = new Azure.AppService.Plan(&quot;examplePlan&quot;, new Azure.AppService.PlanArgs
    {
        Location = exampleResourceGroup.Location,
        ResourceGroupName = exampleResourceGroup.Name,
        Sku = new Azure.AppService.Inputs.PlanSkuArgs
        {
            Tier = &quot;Standard&quot;,
            Size = &quot;S1&quot;,
        },
    });
    var exampleAppService = new Azure.AppService.AppService(&quot;exampleAppService&quot;, new Azure.AppService.AppServiceArgs
    {
        Location = exampleResourceGroup.Location,
        ResourceGroupName = exampleResourceGroup.Name,
        AppServicePlanId = examplePlan.Id,
    });
    var exampleCustomHostnameBinding = new Azure.AppService.CustomHostnameBinding(&quot;exampleCustomHostnameBinding&quot;, new Azure.AppService.CustomHostnameBindingArgs
    {
        Hostname = &quot;www.mywebsite.com&quot;,
        AppServiceName = exampleAppService.Name,
        ResourceGroupName = exampleResourceGroup.Name,
    });
}

}

Environment

Manages an App Service Environment.

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var exampleResourceGroup = new Azure.Core.ResourceGroup(&quot;exampleResourceGroup&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;westeurope&quot;,
    });
    var exampleVirtualNetwork = new Azure.Network.VirtualNetwork(&quot;exampleVirtualNetwork&quot;, new Azure.Network.VirtualNetworkArgs
    {
        Location = exampleResourceGroup.Location,
        ResourceGroupName = exampleResourceGroup.Name,
        AddressSpaces = 
        {
            &quot;10.0.0.0/16&quot;,
        },
    });
    var ase = new Azure.Network.Subnet(&quot;ase&quot;, new Azure.Network.SubnetArgs
    {
        ResourceGroupName = exampleResourceGroup.Name,
        VirtualNetworkName = exampleVirtualNetwork.Name,
        AddressPrefix = &quot;10.0.1.0/24&quot;,
    });
    var gateway = new Azure.Network.Subnet(&quot;gateway&quot;, new Azure.Network.SubnetArgs
    {
        ResourceGroupName = exampleResourceGroup.Name,
        VirtualNetworkName = exampleVirtualNetwork.Name,
        AddressPrefix = &quot;10.0.2.0/24&quot;,
    });
    var exampleEnvironment = new Azure.AppService.Environment(&quot;exampleEnvironment&quot;, new Azure.AppService.EnvironmentArgs
    {
        SubnetId = ase.Id,
        PricingTier = &quot;I2&quot;,
        FrontEndScaleFactor = 10,
    });
}

}

SourceCodeToken

Manages an App Service source control token.

NOTE: Source Control Tokens are configured at the subscription level, not on each App Service - as such this can only be configured Subscription-wide

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var example = new Azure.AppService.SourceCodeToken(&quot;example&quot;, new Azure.AppService.SourceCodeTokenArgs
    {
        Token = &quot;7e57735e77e577e57&quot;,
        Type = &quot;GitHub&quot;,
    });
}
