Class Environment

Manages an App Service Environment.

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var exampleResourceGroup = new Azure.Core.ResourceGroup(&quot;exampleResourceGroup&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;westeurope&quot;,
    });
    var exampleVirtualNetwork = new Azure.Network.VirtualNetwork(&quot;exampleVirtualNetwork&quot;, new Azure.Network.VirtualNetworkArgs
    {
        Location = exampleResourceGroup.Location,
        ResourceGroupName = exampleResourceGroup.Name,
        AddressSpaces = 
        {
            &quot;10.0.0.0/16&quot;,
        },
    });
    var ase = new Azure.Network.Subnet(&quot;ase&quot;, new Azure.Network.SubnetArgs
    {
        ResourceGroupName = exampleResourceGroup.Name,
        VirtualNetworkName = exampleVirtualNetwork.Name,
        AddressPrefix = &quot;10.0.1.0/24&quot;,
    });
    var gateway = new Azure.Network.Subnet(&quot;gateway&quot;, new Azure.Network.SubnetArgs
    {
        ResourceGroupName = exampleResourceGroup.Name,
        VirtualNetworkName = exampleVirtualNetwork.Name,
        AddressPrefix = &quot;10.0.2.0/24&quot;,
    });
    var exampleEnvironment = new Azure.AppService.Environment(&quot;exampleEnvironment&quot;, new Azure.AppService.EnvironmentArgs
    {
        SubnetId = ase.Id,
        PricingTier = &quot;I2&quot;,
        FrontEndScaleFactor = 10,
    });
}

}
