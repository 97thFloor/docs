AaaaRecord

Enables you to manage DNS AAAA Records within Azure DNS.

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var exampleResourceGroup = new Azure.Core.ResourceGroup(&quot;exampleResourceGroup&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;West US&quot;,
    });
    var exampleZone = new Azure.Dns.Zone(&quot;exampleZone&quot;, new Azure.Dns.ZoneArgs
    {
        ResourceGroupName = exampleResourceGroup.Name,
    });
    var exampleAaaaRecord = new Azure.Dns.AaaaRecord(&quot;exampleAaaaRecord&quot;, new Azure.Dns.AaaaRecordArgs
    {
        ZoneName = exampleZone.Name,
        ResourceGroupName = exampleResourceGroup.Name,
        Ttl = 300,
    });
}

}

Zone

Enables you to manage DNS zones within Azure DNS. These zones are hosted on Azure's name servers to which you can delegate the zone from the parent domain.

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var example = new Azure.Core.ResourceGroup(&quot;example&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;West US&quot;,
    });
    var example_public = new Azure.Dns.Zone(&quot;example-public&quot;, new Azure.Dns.ZoneArgs
    {
        ResourceGroupName = example.Name,
    });
    var example_private = new Azure.PrivateDns.Zone(&quot;example-private&quot;, new Azure.PrivateDns.ZoneArgs
    {
        ResourceGroupName = example.Name,
    });
}

}
