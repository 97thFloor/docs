Database

Manages a MariaDB Database within a MariaDB Server

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var exampleResourceGroup = new Azure.Core.ResourceGroup(&quot;exampleResourceGroup&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;westeurope&quot;,
    });
    var exampleServer = new Azure.MariaDB.Server(&quot;exampleServer&quot;, new Azure.MariaDB.ServerArgs
    {
        Location = exampleResourceGroup.Location,
        ResourceGroupName = exampleResourceGroup.Name,
        SkuName = &quot;B_Gen5_2&quot;,
        StorageMb = 51200,
        BackupRetentionDays = 7,
        GeoRedundantBackupEnabled = false,
        AdministratorLogin = &quot;acctestun&quot;,
        AdministratorLoginPassword = &quot;H@Sh1CoR3!&quot;,
        Version = &quot;10.2&quot;,
        SslEnforcementEnabled = true,
    });
    var exampleDatabase = new Azure.MariaDB.Database(&quot;exampleDatabase&quot;, new Azure.MariaDB.DatabaseArgs
    {
        ResourceGroupName = exampleResourceGroup.Name,
        ServerName = exampleServer.Name,
        Charset = &quot;utf8&quot;,
        Collation = &quot;utf8_general_ci&quot;,
    });
}

}

FirewallRule

Manages a Firewall Rule for a MariaDB Server

Example Usage (Single IP Address)

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var example = new Azure.MariaDB.FirewallRule(&quot;example&quot;, new Azure.MariaDB.FirewallRuleArgs
    {
        EndIpAddress = &quot;40.112.8.12&quot;,
        ResourceGroupName = &quot;test-rg&quot;,
        ServerName = &quot;test-server&quot;,
        StartIpAddress = &quot;40.112.8.12&quot;,
    });
}

}
