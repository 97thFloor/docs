ActionCustom

Manages a Custom Action within a Logic App Workflow

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var exampleResourceGroup = new Azure.Core.ResourceGroup(&quot;exampleResourceGroup&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;East US&quot;,
    });
    var exampleWorkflow = new Azure.LogicApps.Workflow(&quot;exampleWorkflow&quot;, new Azure.LogicApps.WorkflowArgs
    {
        Location = exampleResourceGroup.Location,
        ResourceGroupName = exampleResourceGroup.Name,
    });
    var exampleActionCustom = new Azure.LogicApps.ActionCustom(&quot;exampleActionCustom&quot;, new Azure.LogicApps.ActionCustomArgs
    {
        LogicAppId = exampleWorkflow.Id,
        Body = @&quot;{
&quot;&quot;description&quot;&quot;: &quot;&quot;A variable to configure the auto expiration age in days. Configured in negative number. Default is -30 (30 days old).&quot;&quot;,
&quot;&quot;inputs&quot;&quot;: {
    &quot;&quot;variables&quot;&quot;: [
        {
            &quot;&quot;name&quot;&quot;: &quot;&quot;ExpirationAgeInDays&quot;&quot;,
            &quot;&quot;type&quot;&quot;: &quot;&quot;Integer&quot;&quot;,
            &quot;&quot;value&quot;&quot;: -30
        }
    ]
},
&quot;&quot;runAfter&quot;&quot;: {},
&quot;&quot;type&quot;&quot;: &quot;&quot;InitializeVariable&quot;&quot;
}
&quot;,
    });
}

}
