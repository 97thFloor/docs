ServerSecurityAlertPolicy

Manages a Security Alert Policy for a MSSQL Server.

NOTE Security Alert Policy is currently only available for MS SQL databases.

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var exampleResourceGroup = new Azure.Core.ResourceGroup(&quot;exampleResourceGroup&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;West US&quot;,
    });
    var exampleSqlServer = new Azure.Sql.SqlServer(&quot;exampleSqlServer&quot;, new Azure.Sql.SqlServerArgs
    {
        ResourceGroupName = exampleResourceGroup.Name,
        Location = exampleResourceGroup.Location,
        Version = &quot;12.0&quot;,
        AdministratorLogin = &quot;4dm1n157r470r&quot;,
        AdministratorLoginPassword = &quot;4-v3ry-53cr37-p455w0rd&quot;,
    });
    var exampleAccount = new Azure.Storage.Account(&quot;exampleAccount&quot;, new Azure.Storage.AccountArgs
    {
        ResourceGroupName = exampleResourceGroup.Name,
        Location = exampleResourceGroup.Location,
        AccountTier = &quot;Standard&quot;,
        AccountReplicationType = &quot;GRS&quot;,
    });
    var exampleServerSecurityAlertPolicy = new Azure.MSSql.ServerSecurityAlertPolicy(&quot;exampleServerSecurityAlertPolicy&quot;, new Azure.MSSql.ServerSecurityAlertPolicyArgs
    {
        ResourceGroupName = exampleResourceGroup.Name,
        ServerName = exampleSqlServer.Name,
        State = &quot;Enabled&quot;,
        StorageEndpoint = exampleAccount.PrimaryBlobEndpoint,
        StorageAccountAccessKey = exampleAccount.PrimaryAccessKey,
        DisabledAlerts = 
        {
            &quot;Sql_Injection&quot;,
            &quot;Data_Exfiltration&quot;,
        },
        RetentionDays = 20,
    });
}

}
