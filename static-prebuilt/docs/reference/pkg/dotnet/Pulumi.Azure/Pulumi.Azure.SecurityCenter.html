AdvancedThreatProtection

Manages a resources Advanced Threat Protection setting.

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var rg = new Azure.Core.ResourceGroup(&quot;rg&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;northeurope&quot;,
    });
    var exampleAccount = new Azure.Storage.Account(&quot;exampleAccount&quot;, new Azure.Storage.AccountArgs
    {
        ResourceGroupName = azurerm_resource_group.Example.Name,
        Location = azurerm_resource_group.Example.Location,
        AccountTier = &quot;Standard&quot;,
        AccountReplicationType = &quot;LRS&quot;,
        Tags = 
        {
            { &quot;environment&quot;, &quot;example&quot; },
        },
    });
    var exampleAdvancedThreatProtection = new Azure.SecurityCenter.AdvancedThreatProtection(&quot;exampleAdvancedThreatProtection&quot;, new Azure.SecurityCenter.AdvancedThreatProtectionArgs
    {
        TargetResourceId = exampleAccount.Id,
        Enabled = true,
    });
}

}

SubscriptionPricing

Manages the Pricing Tier for Azure Security Center in the current subscription.

NOTE: This resource requires the Owner permission on the Subscription.

NOTE: Deletion of this resource does not change or reset the pricing tier to Free

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var example = new Azure.SecurityCenter.SubscriptionPricing(&quot;example&quot;, new Azure.SecurityCenter.SubscriptionPricingArgs
    {
        Tier = &quot;Standard&quot;,
    });
}

}

Workspace

Manages the subscription's Security Center Workspace.

NOTE: Owner access permission is required.

NOTE: The subscription's pricing model can not be Free for this to have any affect.

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var exampleResourceGroup = new Azure.Core.ResourceGroup(&quot;exampleResourceGroup&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;westus&quot;,
    });
    var exampleAnalyticsWorkspace = new Azure.OperationalInsights.AnalyticsWorkspace(&quot;exampleAnalyticsWorkspace&quot;, new Azure.OperationalInsights.AnalyticsWorkspaceArgs
    {
        Location = exampleResourceGroup.Location,
        ResourceGroupName = exampleResourceGroup.Name,
        Sku = &quot;PerGB2018&quot;,
    });
    var exampleWorkspace = new Azure.SecurityCenter.Workspace(&quot;exampleWorkspace&quot;, new Azure.SecurityCenter.WorkspaceArgs
    {
        Scope = &quot;/subscriptions/00000000-0000-0000-0000-000000000000&quot;,
        WorkspaceId = exampleAnalyticsWorkspace.Id,
    });
}

}
