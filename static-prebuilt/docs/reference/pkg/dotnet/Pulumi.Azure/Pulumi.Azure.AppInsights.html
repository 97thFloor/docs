Manages an Application Insights Analytics Item component.

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var exampleResourceGroup = new Azure.Core.ResourceGroup(&quot;exampleResourceGroup&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;West Europe&quot;,
    });
    var exampleInsights = new Azure.AppInsights.Insights(&quot;exampleInsights&quot;, new Azure.AppInsights.InsightsArgs
    {
        Location = &quot;West Europe&quot;,
        ResourceGroupName = exampleResourceGroup.Name,
        ApplicationType = &quot;web&quot;,
    });
    var exampleAnalyticsItem = new Azure.AppInsights.AnalyticsItem(&quot;exampleAnalyticsItem&quot;, new Azure.AppInsights.AnalyticsItemArgs
    {
        ApplicationInsightsId = exampleInsights.Id,
        Content = &quot;requests //simple example query&quot;,
        Scope = &quot;shared&quot;,
        Type = &quot;query&quot;,
    });
}

ApiKey

Manages an Application Insights API key.

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var exampleResourceGroup = new Azure.Core.ResourceGroup(&quot;exampleResourceGroup&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;West Europe&quot;,
    });
    var exampleInsights = new Azure.AppInsights.Insights(&quot;exampleInsights&quot;, new Azure.AppInsights.InsightsArgs
    {
        Location = &quot;West Europe&quot;,
        ResourceGroupName = exampleResourceGroup.Name,
        ApplicationType = &quot;web&quot;,
    });
    var readTelemetry = new Azure.AppInsights.ApiKey(&quot;readTelemetry&quot;, new Azure.AppInsights.ApiKeyArgs
    {
        ApplicationInsightsId = exampleInsights.Id,
        ReadPermissions = 
        {
            &quot;aggregate&quot;,
            &quot;api&quot;,
            &quot;draft&quot;,
            &quot;extendqueries&quot;,
            &quot;search&quot;,
        },
    });
    var writeAnnotations = new Azure.AppInsights.ApiKey(&quot;writeAnnotations&quot;, new Azure.AppInsights.ApiKeyArgs
    {
        ApplicationInsightsId = exampleInsights.Id,
        WritePermissions = 
        {
            &quot;annotations&quot;,
        },
    });
    var authenticateSdkControlChannelApiKey = new Azure.AppInsights.ApiKey(&quot;authenticateSdkControlChannelApiKey&quot;, new Azure.AppInsights.ApiKeyArgs
    {
        ApplicationInsightsId = exampleInsights.Id,
        ReadPermissions = 
        {
            &quot;agentconfig&quot;,
        },
    });
    var fullPermissions = new Azure.AppInsights.ApiKey(&quot;fullPermissions&quot;, new Azure.AppInsights.ApiKeyArgs
    {
        ApplicationInsightsId = exampleInsights.Id,
        ReadPermissions = 
        {
            &quot;agentconfig&quot;,
            &quot;aggregate&quot;,
            &quot;api&quot;,
            &quot;draft&quot;,
            &quot;extendqueries&quot;,
            &quot;search&quot;,
        },
        WritePermissions = 
        {
            &quot;annotations&quot;,
        },
    });
    this.ReadTelemetryApiKey = readTelemetry.ApiKey;
    this.WriteAnnotationsApiKey = writeAnnotations.ApiKey;
    this.AuthenticateSdkControlChannel = authenticateSdkControlChannelApiKey.ApiKey;
    this.FullPermissionsApiKey = fullPermissions.ApiKey;
}

[Output(&quot;readTelemetryApiKey&quot;)]
public Output&lt;string> ReadTelemetryApiKey { get; set; }
[Output(&quot;writeAnnotationsApiKey&quot;)]
public Output&lt;string> WriteAnnotationsApiKey { get; set; }
[Output(&quot;authenticateSdkControlChannel&quot;)]
public Output&lt;string> AuthenticateSdkControlChannel { get; set; }
[Output(&quot;fullPermissionsApiKey&quot;)]
public Output&lt;string> FullPermissionsApiKey { get; set; }
}

Insights

Manages an Application Insights component.

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var exampleResourceGroup = new Azure.Core.ResourceGroup(&quot;exampleResourceGroup&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;West Europe&quot;,
    });
    var exampleInsights = new Azure.AppInsights.Insights(&quot;exampleInsights&quot;, new Azure.AppInsights.InsightsArgs
    {
        Location = exampleResourceGroup.Location,
        ResourceGroupName = exampleResourceGroup.Name,
        ApplicationType = &quot;web&quot;,
    });
    this.InstrumentationKey = exampleInsights.InstrumentationKey;
    this.AppId = exampleInsights.AppId;
}

[Output(&quot;instrumentationKey&quot;)]
public Output&lt;string> InstrumentationKey { get; set; }
[Output(&quot;appId&quot;)]
public Output&lt;string> AppId { get; set; }
}
