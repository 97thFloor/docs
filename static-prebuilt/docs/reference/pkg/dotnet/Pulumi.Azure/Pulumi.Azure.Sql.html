ActiveDirectoryAdministrator

Allows you to set a user or group as the AD administrator for an Azure SQL server

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var current = Output.Create(Azure.Core.GetClientConfig.InvokeAsync());
    var exampleResourceGroup = new Azure.Core.ResourceGroup(&quot;exampleResourceGroup&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;West US&quot;,
    });
    var exampleSqlServer = new Azure.Sql.SqlServer(&quot;exampleSqlServer&quot;, new Azure.Sql.SqlServerArgs
    {
        ResourceGroupName = exampleResourceGroup.Name,
        Location = exampleResourceGroup.Location,
        Version = &quot;12.0&quot;,
        AdministratorLogin = &quot;4dm1n157r470r&quot;,
        AdministratorLoginPassword = &quot;4-v3ry-53cr37-p455w0rd&quot;,
    });
    var exampleActiveDirectoryAdministrator = new Azure.Sql.ActiveDirectoryAdministrator(&quot;exampleActiveDirectoryAdministrator&quot;, new Azure.Sql.ActiveDirectoryAdministratorArgs
    {
        ServerName = exampleSqlServer.Name,
        ResourceGroupName = exampleResourceGroup.Name,
        Login = &quot;sqladmin&quot;,
        TenantId = current.Apply(current => current.TenantId),
        ObjectId = current.Apply(current => current.ObjectId),
    });
}

}

VirtualNetworkRule

Allows you to add, update, or remove an Azure SQL server to a subnet of a virtual network.

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var example = new Azure.Core.ResourceGroup(&quot;example&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;West US&quot;,
    });
    var vnet = new Azure.Network.VirtualNetwork(&quot;vnet&quot;, new Azure.Network.VirtualNetworkArgs
    {
        AddressSpaces = 
        {
            &quot;10.7.29.0/29&quot;,
        },
        Location = example.Location,
        ResourceGroupName = example.Name,
    });
    var subnet = new Azure.Network.Subnet(&quot;subnet&quot;, new Azure.Network.SubnetArgs
    {
        ResourceGroupName = example.Name,
        VirtualNetworkName = vnet.Name,
        AddressPrefix = &quot;10.7.29.0/29&quot;,
        ServiceEndpoints = 
        {
            &quot;Microsoft.Sql&quot;,
        },
    });
    var sqlserver = new Azure.Sql.SqlServer(&quot;sqlserver&quot;, new Azure.Sql.SqlServerArgs
    {
        ResourceGroupName = example.Name,
        Location = example.Location,
        Version = &quot;12.0&quot;,
        AdministratorLogin = &quot;4dm1n157r470r&quot;,
        AdministratorLoginPassword = &quot;4-v3ry-53cr37-p455w0rd&quot;,
    });
    var sqlvnetrule = new Azure.Sql.VirtualNetworkRule(&quot;sqlvnetrule&quot;, new Azure.Sql.VirtualNetworkRuleArgs
    {
        ResourceGroupName = example.Name,
        ServerName = sqlserver.Name,
        SubnetId = subnet.Id,
    });
}

}
