Configuration

Sets a PostgreSQL Configuration value on a PostgreSQL Server.

Disclaimers

Note: Since this resource is provisioned by default, the Azure Provider will not check for the presence of an existing resource prior to attempting to create it.

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var exampleResourceGroup = new Azure.Core.ResourceGroup(&quot;exampleResourceGroup&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;West Europe&quot;,
    });
    var exampleServer = new Azure.PostgreSql.Server(&quot;exampleServer&quot;, new Azure.PostgreSql.ServerArgs
    {
        Location = exampleResourceGroup.Location,
        ResourceGroupName = exampleResourceGroup.Name,
        SkuName = &quot;B_Gen5_2&quot;,
        StorageMb = 5120,
        BackupRetentionDays = 7,
        GeoRedundantBackupEnabled = false,
        AutoGrowEnabled = true,
        AdministratorLogin = &quot;psqladminun&quot;,
        AdministratorLoginPassword = &quot;H@Sh1CoR3!&quot;,
        Version = &quot;9.5&quot;,
        SslEnforcementEnabled = true,
    });
    var exampleConfiguration = new Azure.PostgreSql.Configuration(&quot;exampleConfiguration&quot;, new Azure.PostgreSql.ConfigurationArgs
    {
        ResourceGroupName = exampleResourceGroup.Name,
        ServerName = exampleServer.Name,
        Value = &quot;on&quot;,
    });
}

}
