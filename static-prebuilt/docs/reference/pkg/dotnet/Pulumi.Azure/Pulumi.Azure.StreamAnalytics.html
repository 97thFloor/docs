Job

Manages a Stream Analytics Job.

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var exampleResourceGroup = new Azure.Core.ResourceGroup(&quot;exampleResourceGroup&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;West Europe&quot;,
    });
    var exampleJob = new Azure.StreamAnalytics.Job(&quot;exampleJob&quot;, new Azure.StreamAnalytics.JobArgs
    {
        ResourceGroupName = exampleResourceGroup.Name,
        Location = exampleResourceGroup.Location,
        CompatibilityLevel = &quot;1.1&quot;,
        DataLocale = &quot;en-GB&quot;,
        EventsLateArrivalMaxDelayInSeconds = 60,
        EventsOutOfOrderMaxDelayInSeconds = 50,
        EventsOutOfOrderPolicy = &quot;Adjust&quot;,
        OutputErrorPolicy = &quot;Drop&quot;,
        StreamingUnits = 3,
        Tags = 
        {
            { &quot;environment&quot;, &quot;Example&quot; },
        },
        TransformationQuery = @&quot;    SELECT *
INTO [YourOutputAlias]
FROM [YourInputAlias]
&quot;,
    });
}

}

OutputEventHub

Manages a Stream Analytics Output to an EventHub.

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var exampleResourceGroup = Output.Create(Azure.Core.GetResourceGroup.InvokeAsync(new Azure.Core.GetResourceGroupArgs
    {
        Name = &quot;example-resources&quot;,
    }));
    var exampleJob = Output.Create(Azure.StreamAnalytics.GetJob.InvokeAsync(new Azure.StreamAnalytics.GetJobArgs
    {
        Name = &quot;example-job&quot;,
        ResourceGroupName = azurerm_resource_group.Example.Name,
    }));
    var exampleEventHubNamespace = new Azure.EventHub.EventHubNamespace(&quot;exampleEventHubNamespace&quot;, new Azure.EventHub.EventHubNamespaceArgs
    {
        Location = exampleResourceGroup.Apply(exampleResourceGroup => exampleResourceGroup.Location),
        ResourceGroupName = exampleResourceGroup.Apply(exampleResourceGroup => exampleResourceGroup.Name),
        Sku = &quot;Standard&quot;,
        Capacity = 1,
    });
    var exampleEventHub = new Azure.EventHub.EventHub(&quot;exampleEventHub&quot;, new Azure.EventHub.EventHubArgs
    {
        NamespaceName = exampleEventHubNamespace.Name,
        ResourceGroupName = exampleResourceGroup.Apply(exampleResourceGroup => exampleResourceGroup.Name),
        PartitionCount = 2,
        MessageRetention = 1,
    });
    var exampleOutputEventHub = new Azure.StreamAnalytics.OutputEventHub(&quot;exampleOutputEventHub&quot;, new Azure.StreamAnalytics.OutputEventHubArgs
    {
        StreamAnalyticsJobName = exampleJob.Apply(exampleJob => exampleJob.Name),
        ResourceGroupName = exampleJob.Apply(exampleJob => exampleJob.ResourceGroupName),
        EventhubName = exampleEventHub.Name,
        ServicebusNamespace = exampleEventHubNamespace.Name,
        SharedAccessPolicyKey = exampleEventHubNamespace.DefaultPrimaryKey,
        SharedAccessPolicyName = &quot;RootManageSharedAccessKey&quot;,
        Serialization = new Azure.StreamAnalytics.Inputs.OutputEventHubSerializationArgs
        {
            Type = &quot;Avro&quot;,
        },
    });
}

}
