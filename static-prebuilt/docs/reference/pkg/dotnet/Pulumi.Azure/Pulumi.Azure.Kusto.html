Cluster

Manages a Kusto (also known as Azure Data Explorer) Cluster

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var rg = new Azure.Core.ResourceGroup(&quot;rg&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;East US&quot;,
    });
    var example = new Azure.Kusto.Cluster(&quot;example&quot;, new Azure.Kusto.ClusterArgs
    {
        Location = rg.Location,
        ResourceGroupName = rg.Name,
        Sku = new Azure.Kusto.Inputs.ClusterSkuArgs
        {
            Name = &quot;Standard_D13_v2&quot;,
            Capacity = 2,
        },
        Tags = 
        {
            { &quot;Environment&quot;, &quot;Production&quot; },
        },
    });
}

}

EventhubDataConnection

Manages a Kusto (also known as Azure Data Explorer) EventHub Data Connection

Example Usage

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var rg = new Azure.Core.ResourceGroup(&quot;rg&quot;, new Azure.Core.ResourceGroupArgs
    {
        Location = &quot;East US&quot;,
    });
    var cluster = new Azure.Kusto.Cluster(&quot;cluster&quot;, new Azure.Kusto.ClusterArgs
    {
        Location = rg.Location,
        ResourceGroupName = rg.Name,
        Sku = new Azure.Kusto.Inputs.ClusterSkuArgs
        {
            Name = &quot;Standard_D13_v2&quot;,
            Capacity = 2,
        },
    });
    var database = new Azure.Kusto.Database(&quot;database&quot;, new Azure.Kusto.DatabaseArgs
    {
        ResourceGroupName = rg.Name,
        Location = rg.Location,
        ClusterName = cluster.Name,
        HotCachePeriod = &quot;P7D&quot;,
        SoftDeletePeriod = &quot;P31D&quot;,
    });
    var eventhubNs = new Azure.EventHub.EventHubNamespace(&quot;eventhubNs&quot;, new Azure.EventHub.EventHubNamespaceArgs
    {
        Location = rg.Location,
        ResourceGroupName = rg.Name,
        Sku = &quot;Standard&quot;,
    });
    var eventhub = new Azure.EventHub.EventHub(&quot;eventhub&quot;, new Azure.EventHub.EventHubArgs
    {
        NamespaceName = eventhubNs.Name,
        ResourceGroupName = rg.Name,
        PartitionCount = 1,
        MessageRetention = 1,
    });
    var consumerGroup = new Azure.EventHub.ConsumerGroup(&quot;consumerGroup&quot;, new Azure.EventHub.ConsumerGroupArgs
    {
        NamespaceName = eventhubNs.Name,
        EventhubName = eventhub.Name,
        ResourceGroupName = rg.Name,
    });
    var eventhubConnection = new Azure.Kusto.EventhubDataConnection(&quot;eventhubConnection&quot;, new Azure.Kusto.EventhubDataConnectionArgs
    {
        ResourceGroupName = rg.Name,
        Location = rg.Location,
        ClusterName = cluster.Name,
        DatabaseName = database.Name,
        EventhubId = azurerm_eventhub.Evenhub.Id,
        ConsumerGroup = consumerGroup.Name,
        TableName = &quot;my-table&quot;,
        MappingRuleName = &quot;my-table-mapping&quot;,
        DataFormat = &quot;JSON&quot;,
    });
    //(Optional)
}

}
