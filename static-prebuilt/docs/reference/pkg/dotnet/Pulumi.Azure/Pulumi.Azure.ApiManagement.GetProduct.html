Use this data source to access information about an existing API Management Product.

{{% examples %}}

Example Usage

{{% example %}}

using Pulumi;
using Azure = Pulumi.Azure;

class MyStack : Stack
{
public MyStack()
{
    var example = Output.Create(Azure.ApiManagement.GetProduct.InvokeAsync(new Azure.ApiManagement.GetProductArgs
    {
        ProductId = &quot;my-product&quot;,
        ApiManagementName = &quot;example-apim&quot;,
        ResourceGroupName = &quot;search-service&quot;,
    }));
    this.ProductTerms = example.Apply(example => example.Terms);
}

[Output(&quot;productTerms&quot;)]
public Output&lt;string> ProductTerms { get; set; }
}
