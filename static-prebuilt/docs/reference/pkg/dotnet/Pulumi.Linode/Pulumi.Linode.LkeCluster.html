Class LkeCluster

Manages an LKE cluster.

Example Usage

using Pulumi;
using Linode = Pulumi.Linode;

class MyStack : Stack
{
public MyStack()
{
    var my_cluster = new Linode.LkeCluster(&quot;my-cluster&quot;, new Linode.LkeClusterArgs
    {
        K8sVersion = &quot;1.17&quot;,
        Label = &quot;my-cluster&quot;,
        Pools = 
        {
            new Linode.Inputs.LkeClusterPoolArgs
            {
                Count = 3,
                Type = &quot;g6-standard-2&quot;,
            },
        },
        Region = &quot;us-central&quot;,
        Tags = 
        {
            &quot;prod&quot;,
        },
    });
}

}
