AutoLogin

Creates an Auto Login Okta Application.

This resource allows you to create and configure an Auto Login Okta Application.

Example Usage

using Pulumi;
using Okta = Pulumi.Okta;

class MyStack : Stack
{
public MyStack()
{
    var example = new Okta.App.AutoLogin(&quot;example&quot;, new Okta.App.AutoLoginArgs
    {
        CredentialsScheme = &quot;EDIT_USERNAME_AND_PASSWORD&quot;,
        Label = &quot;Example App&quot;,
        RevealPassword = true,
        SignOnRedirectUrl = &quot;https://example.com&quot;,
        SignOnUrl = &quot;https://example.com/login.html&quot;,
    });
}

}

Saml

Creates an SAML Application.

This resource allows you to create and configure an SAML Application.

Example Usage

using Pulumi;
using Okta = Pulumi.Okta;

class MyStack : Stack
{
public MyStack()
{
    var example = new Okta.App.Saml(&quot;example&quot;, new Okta.App.SamlArgs
    {
        AttributeStatements = 
        {
            new Okta.App.Inputs.SamlAttributeStatementArgs
            {
                FilterType = &quot;REGEX&quot;,
                FilterValue = &quot;.*&quot;,
                Name = &quot;groups&quot;,
                Type = &quot;GROUP&quot;,
            },
        },
        Audience = &quot;http://example.com/audience&quot;,
        AuthnContextClassRef = &quot;urn:oasis:names:tc:SAML:2.0:ac:classes:PasswordProtectedTransport&quot;,
        Destination = &quot;http://example.com&quot;,
        DigestAlgorithm = &quot;SHA256&quot;,
        HonorForceAuthn = false,
        Label = &quot;example&quot;,
        Recipient = &quot;http://example.com&quot;,
        ResponseSigned = true,
        SignatureAlgorithm = &quot;RSA_SHA256&quot;,
        SsoUrl = &quot;http://example.com&quot;,
        SubjectNameIdFormat = &quot;urn:oasis:names:tc:SAML:1.1:nameid-format:emailAddress&quot;,
        SubjectNameIdTemplate = user.UserName,
    });
}
