Oidc

Creates an OIDC Identity Provider.

This resource allows you to create and configure an OIDC Identity Provider.

Example Usage

using Pulumi;
using Okta = Pulumi.Okta;

class MyStack : Stack
{
public MyStack()
{
    var example = new Okta.Idp.Oidc(&quot;example&quot;, new Okta.Idp.OidcArgs
    {
        AcsBinding = &quot;HTTP-POST&quot;,
        AcsType = &quot;INSTANCE&quot;,
        AuthorizationBinding = &quot;HTTP-REDIRECT&quot;,
        AuthorizationUrl = &quot;https://idp.example.com/authorize&quot;,
        ClientId = &quot;efg456&quot;,
        ClientSecret = &quot;efg456&quot;,
        IssuerUrl = &quot;https://id.example.com&quot;,
        JwksBinding = &quot;HTTP-REDIRECT&quot;,
        JwksUrl = &quot;https://idp.example.com/keys&quot;,
        Scopes = 
        {
            &quot;openid&quot;,
        },
        TokenBinding = &quot;HTTP-POST&quot;,
        TokenUrl = &quot;https://idp.example.com/token&quot;,
        UserInfoBinding = &quot;HTTP-REDIRECT&quot;,
        UserInfoUrl = &quot;https://idp.example.com/userinfo&quot;,
        UsernameTemplate = &quot;idpuser.email&quot;,
    });
}

}
