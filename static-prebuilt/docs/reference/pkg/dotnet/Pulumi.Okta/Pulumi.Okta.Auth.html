Server

Creates an Authorization Server.

This resource allows you to create and configure an Authorization Server.

Example Usage

using Pulumi;
using Okta = Pulumi.Okta;

class MyStack : Stack
{
public MyStack()
{
    var example = new Okta.Auth.Server(&quot;example&quot;, new Okta.Auth.ServerArgs
    {
        Audiences = 
        {
            &quot;api://example&quot;,
        },
        Description = &quot;My Example Auth Server&quot;,
        IssuerMode = &quot;CUSTOM_URL&quot;,
        Status = &quot;ACTIVE&quot;,
    });
}

}
