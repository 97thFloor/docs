Class Password

Creates a Password Policy.

This resource allows you to create and configure a Password Policy.

Example Usage

using Pulumi;
using Okta = Pulumi.Okta;

class MyStack : Stack
{
public MyStack()
{
    var example = new Okta.Policy.Password(&quot;example&quot;, new Okta.Policy.PasswordArgs
    {
        Description = &quot;Example&quot;,
        GroupsIncludeds = 
        {
            data.Okta_group.Everyone.Id,
        },
        PasswordHistoryCount = 4,
        Status = &quot;ACTIVE&quot;,
    });
}

}
