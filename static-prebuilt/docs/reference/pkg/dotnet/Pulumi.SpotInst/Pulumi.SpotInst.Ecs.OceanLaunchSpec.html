Class OceanLaunchSpec

Provides a custom Spotinst Ocean ECS Launch Spec resource.

Example Usage

using Pulumi;
using SpotInst = Pulumi.SpotInst;

class MyStack : Stack
{
public MyStack()
{
    var example = new SpotInst.Ecs.OceanLaunchSpec(&quot;example&quot;, new SpotInst.Ecs.OceanLaunchSpecArgs
    {
        Attributes = 
        {
            new SpotInst.Ecs.Inputs.OceanLaunchSpecAttributeArgs
            {
                Key = &quot;fakeKey&quot;,
                Value = &quot;fakeValue&quot;,
            },
        },
        AutoscaleHeadrooms = 
        {
            new SpotInst.Ecs.Inputs.OceanLaunchSpecAutoscaleHeadroomArgs
            {
                CpuPerUnit = 1000,
                MemoryPerUnit = 2048,
                NumOfUnits = 5,
            },
        },
        IamInstanceProfile = &quot;iam-profile&quot;,
        ImageId = &quot;ami-123456&quot;,
        OceanId = &quot;o-123456&quot;,
        SecurityGroupIds = 
        {
            &quot;awseb-12345&quot;,
        },
        Tags = 
        {
            new SpotInst.Ecs.Inputs.OceanLaunchSpecTagArgs
            {
                Key = &quot;Env&quot;,
                Value = &quot;production&quot;,
            },
        },
        UserData = &quot;echo hello world&quot;,
    });
}

}
