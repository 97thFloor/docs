Balancer

Provides a Spotinst Multai Balancer.

Example Usage

using Pulumi;
using SpotInst = Pulumi.SpotInst;

class MyStack : Stack
{
public MyStack()
{
    var myBalancer = new SpotInst.Multai.Balancer(&quot;myBalancer&quot;, new SpotInst.Multai.BalancerArgs
    {
        ConnectionTimeouts = new SpotInst.Multai.Inputs.BalancerConnectionTimeoutsArgs
        {
            Draining = 10,
            Idle = 10,
        },
        Scheme = &quot;internal&quot;,
        Tags = 
        {
            new SpotInst.Multai.Inputs.BalancerTagArgs
            {
                Key = &quot;env&quot;,
                Value = &quot;prod&quot;,
            },
        },
    });
}

}
