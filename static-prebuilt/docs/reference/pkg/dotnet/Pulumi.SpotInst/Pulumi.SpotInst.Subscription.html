Class Subscription

Provides a Spotinst subscription resource.

Example Usage

using Pulumi;
using SpotInst = Pulumi.SpotInst;

class MyStack : Stack
{
public MyStack()
{
    // Create a Subscription
    var default_subscription = new SpotInst.Subscription(&quot;default-subscription&quot;, new SpotInst.SubscriptionArgs
    {
        Endpoint = &quot;http://endpoint.com&quot;,
        EventType = &quot;AWS_EC2_INSTANCE_LAUNCH&quot;,
        Format = 
        {
            { &quot;event&quot;, &quot;%event%&quot; },
            { &quot;instance_id&quot;, &quot;%instance-id%&quot; },
            { &quot;resource_id&quot;, &quot;%resource-id%&quot; },
            { &quot;resource_name&quot;, &quot;%resource-name%&quot; },
            { &quot;tags&quot;, &quot;foo,baz,baz&quot; },
        },
        Protocol = &quot;http&quot;,
        ResourceId = spotinst_elastigroup_aws.My_eg.Id,
    });
}

}
