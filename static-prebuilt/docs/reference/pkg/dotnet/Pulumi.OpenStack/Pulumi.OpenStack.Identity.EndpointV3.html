Class EndpointV3

Manages a V3 Endpoint resource within OpenStack Keystone.

Note: This usually requires admin privileges.

Example Usage

using Pulumi;
using OpenStack = Pulumi.OpenStack;

class MyStack : Stack
{
public MyStack()
{
    var service1 = new OpenStack.Identity.ServiceV3(&quot;service1&quot;, new OpenStack.Identity.ServiceV3Args
    {
        Type = &quot;my-service-type&quot;,
    });
    var endpoint1 = new OpenStack.Identity.EndpointV3(&quot;endpoint1&quot;, new OpenStack.Identity.EndpointV3Args
    {
        EndpointRegion = service1.Region,
        ServiceId = service1.Id,
        Url = &quot;http://my-endpoint&quot;,
    });
}

}
