Class Firewall

Manages a v1 firewall resource within OpenStack.

Example Usage

using Pulumi;
using OpenStack = Pulumi.OpenStack;

class MyStack : Stack
{
public MyStack()
{
    var rule1 = new OpenStack.Firewall.Rule(&quot;rule1&quot;, new OpenStack.Firewall.RuleArgs
    {
        Action = &quot;deny&quot;,
        Description = &quot;drop TELNET traffic&quot;,
        DestinationPort = &quot;23&quot;,
        Enabled = &quot;true&quot;,
        Protocol = &quot;tcp&quot;,
    });
    var rule2 = new OpenStack.Firewall.Rule(&quot;rule2&quot;, new OpenStack.Firewall.RuleArgs
    {
        Action = &quot;deny&quot;,
        Description = &quot;drop NTP traffic&quot;,
        DestinationPort = &quot;123&quot;,
        Enabled = &quot;false&quot;,
        Protocol = &quot;udp&quot;,
    });
    var policy1 = new OpenStack.Firewall.Policy(&quot;policy1&quot;, new OpenStack.Firewall.PolicyArgs
    {
        Rules = 
        {
            rule1.Id,
            rule2.Id,
        },
    });
    var firewall1 = new OpenStack.Firewall.Firewall(&quot;firewall1&quot;, new OpenStack.Firewall.FirewallArgs
    {
        PolicyId = policy1.Id,
    });
}

}
