Network

Manages a V2 Neutron network resource within OpenStack.

Example Usage

using Pulumi;
using OpenStack = Pulumi.OpenStack;

class MyStack : Stack
{
public MyStack()
{
    var network1 = new OpenStack.Networking.Network(&quot;network1&quot;, new OpenStack.Networking.NetworkArgs
    {
        AdminStateUp = &quot;true&quot;,
    });
    var subnet1 = new OpenStack.Networking.Subnet(&quot;subnet1&quot;, new OpenStack.Networking.SubnetArgs
    {
        Cidr = &quot;192.168.199.0/24&quot;,
        IpVersion = 4,
        NetworkId = network1.Id,
    });
    var secgroup1 = new OpenStack.Compute.SecGroup(&quot;secgroup1&quot;, new OpenStack.Compute.SecGroupArgs
    {
        Description = &quot;a security group&quot;,
        Rules = 
        {
            new OpenStack.Compute.Inputs.SecGroupRuleArgs
            {
                Cidr = &quot;0.0.0.0/0&quot;,
                FromPort = 22,
                IpProtocol = &quot;tcp&quot;,
                ToPort = 22,
            },
        },
    });
    var port1 = new OpenStack.Networking.Port(&quot;port1&quot;, new OpenStack.Networking.PortArgs
    {
        AdminStateUp = &quot;true&quot;,
        FixedIps = 
        {
            new OpenStack.Networking.Inputs.PortFixedIpArgs
            {
                IpAddress = &quot;192.168.199.10&quot;,
                SubnetId = subnet1.Id,
            },
        },
        NetworkId = network1.Id,
        SecurityGroupIds = 
        {
            secgroup1.Id,
        },
    });
    var instance1 = new OpenStack.Compute.Instance(&quot;instance1&quot;, new OpenStack.Compute.InstanceArgs
    {
        Networks = 
        {
            new OpenStack.Compute.Inputs.InstanceNetworkArgs
            {
                Port = port1.Id,
            },
        },
        SecurityGroups = 
        {
            secgroup1.Name,
        },
    });
}

RbacPolicyV2

The RBAC policy resource contains functionality for working with Neutron RBAC Policies. Role-Based Access Control (RBAC) policy framework enables both operators and users to grant access to resources for specific projects.

Sharing an object with a specific project is accomplished by creating a policy entry that permits the target project the access_as_shared action on that object.

To make a network available as an external network for specific projects rather than all projects, use the access_as_external action. If a network is marked as external during creation, it now implicitly creates a wildcard RBAC policy granting everyone access to preserve previous behavior before this feature was added.

Example Usage

using Pulumi;
using OpenStack = Pulumi.OpenStack;

class MyStack : Stack
{
public MyStack()
{
    var network1 = new OpenStack.Networking.Network(&quot;network1&quot;, new OpenStack.Networking.NetworkArgs
    {
        AdminStateUp = &quot;true&quot;,
    });
    var rbacPolicy1 = new OpenStack.Networking.RbacPolicyV2(&quot;rbacPolicy1&quot;, new OpenStack.Networking.RbacPolicyV2Args
    {
        Action = &quot;access_as_shared&quot;,
        ObjectId = network1.Id,
        ObjectType = &quot;network&quot;,
        TargetTenant = &quot;20415a973c9e45d3917f078950644697&quot;,
    });
}

}

RouterRoute

Creates a routing entry on a OpenStack V2 router.

Example Usage

using Pulumi;
using OpenStack = Pulumi.OpenStack;

class MyStack : Stack
{
public MyStack()
{
    var router1 = new OpenStack.Networking.Router(&quot;router1&quot;, new OpenStack.Networking.RouterArgs
    {
        AdminStateUp = &quot;true&quot;,
    });
    var network1 = new OpenStack.Networking.Network(&quot;network1&quot;, new OpenStack.Networking.NetworkArgs
    {
        AdminStateUp = &quot;true&quot;,
    });
    var subnet1 = new OpenStack.Networking.Subnet(&quot;subnet1&quot;, new OpenStack.Networking.SubnetArgs
    {
        Cidr = &quot;192.168.199.0/24&quot;,
        IpVersion = 4,
        NetworkId = network1.Id,
    });
    var int1 = new OpenStack.Networking.RouterInterface(&quot;int1&quot;, new OpenStack.Networking.RouterInterfaceArgs
    {
        RouterId = router1.Id,
        SubnetId = subnet1.Id,
    });
    var routerRoute1 = new OpenStack.Networking.RouterRoute(&quot;routerRoute1&quot;, new OpenStack.Networking.RouterRouteArgs
    {
        DestinationCidr = &quot;10.0.1.0/24&quot;,
        NextHop = &quot;192.168.199.254&quot;,
        RouterId = router1.Id,
    });
}

}

Notes

The next_hop IP address must be directly reachable from the router at the openstack.networking.RouterRoute resource creation time. You can ensure that by explicitly specifying a dependency on the openstack.networking.RouterInterface resource that connects the next hop to the router, as in the example above.

SecGroupRule

Manages a V2 neutron security group rule resource within OpenStack. Unlike Nova security groups, neutron separates the group from the rules and also allows an admin to target a specific tenant_id.

        Description = &quot;My neutron security group&quot;,
    });
    var secgroupRule1 = new OpenStack.Networking.SecGroupRule(&quot;secgroupRule1&quot;, new OpenStack.Networking.SecGroupRuleArgs
    {
        Direction = &quot;ingress&quot;,
        Ethertype = &quot;IPv4&quot;,
        PortRangeMax = 22,
        PortRangeMin = 22,
        Protocol = &quot;tcp&quot;,
        RemoteIpPrefix = &quot;0.0.0.0/0&quot;,
