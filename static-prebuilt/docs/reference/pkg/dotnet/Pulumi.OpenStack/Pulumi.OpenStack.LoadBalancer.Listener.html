Class Listener

Manages a V2 listener resource within OpenStack.

Example Usage

using Pulumi;
using OpenStack = Pulumi.OpenStack;

class MyStack : Stack
{
public MyStack()
{
    var listener1 = new OpenStack.LoadBalancer.Listener(&quot;listener1&quot;, new OpenStack.LoadBalancer.ListenerArgs
    {
        InsertHeaders = 
        {
            { &quot;X-Forwarded-For&quot;, &quot;true&quot; },
        },
        LoadbalancerId = &quot;d9415786-5f1a-428b-b35f-2f1523e146d2&quot;,
        Protocol = &quot;HTTP&quot;,
        ProtocolPort = 8080,
    });
}

}
