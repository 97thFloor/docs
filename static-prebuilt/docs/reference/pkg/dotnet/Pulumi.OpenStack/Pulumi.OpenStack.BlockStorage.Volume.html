Class Volume

Manages a V3 volume resource within OpenStack.

Example Usage

using Pulumi;
using OpenStack = Pulumi.OpenStack;

class MyStack : Stack
{
public MyStack()
{
    var volume1 = new OpenStack.BlockStorage.Volume(&quot;volume1&quot;, new OpenStack.BlockStorage.VolumeArgs
    {
        Description = &quot;first test volume&quot;,
        Region = &quot;RegionOne&quot;,
        Size = 3,
    });
}

}
