Share

Use this resource to configure a share.

Example Usage

using Pulumi;
using OpenStack = Pulumi.OpenStack;

class MyStack : Stack
{
public MyStack()
{
    var network1 = new OpenStack.Networking.Network(&quot;network1&quot;, new OpenStack.Networking.NetworkArgs
    {
        AdminStateUp = &quot;true&quot;,
    });
    var subnet1 = new OpenStack.Networking.Subnet(&quot;subnet1&quot;, new OpenStack.Networking.SubnetArgs
    {
        Cidr = &quot;192.168.199.0/24&quot;,
        IpVersion = 4,
        NetworkId = network1.Id,
    });
    var sharenetwork1 = new OpenStack.SharedFileSystem.ShareNetwork(&quot;sharenetwork1&quot;, new OpenStack.SharedFileSystem.ShareNetworkArgs
    {
        Description = &quot;test share network with security services&quot;,
        NeutronNetId = network1.Id,
        NeutronSubnetId = subnet1.Id,
    });
    var share1 = new OpenStack.SharedFileSystem.Share(&quot;share1&quot;, new OpenStack.SharedFileSystem.ShareArgs
    {
        Description = &quot;test share description&quot;,
        ShareNetworkId = sharenetwork1.Id,
        ShareProto = &quot;NFS&quot;,
        Size = 1,
    });
}

}
