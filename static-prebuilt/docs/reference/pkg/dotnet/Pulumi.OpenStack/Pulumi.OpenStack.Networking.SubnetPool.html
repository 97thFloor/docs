Class SubnetPool

Manages a V2 Neutron subnetpool resource within OpenStack.

Example Usage

Create a Subnet Pool

using Pulumi;
using OpenStack = Pulumi.OpenStack;

class MyStack : Stack
{
public MyStack()
{
    var subnetpool1 = new OpenStack.Networking.SubnetPool(&quot;subnetpool1&quot;, new OpenStack.Networking.SubnetPoolArgs
    {
        IpVersion = 6,
        Prefixes = 
        {
            &quot;fdf7:b13d:dead:beef::/64&quot;,
            &quot;fd65:86cc:a334:39b7::/64&quot;,
        },
    });
}

}

Create a Subnet from a Subnet Pool

using Pulumi;
using OpenStack = Pulumi.OpenStack;

class MyStack : Stack
{
public MyStack()
{
    var network1 = new OpenStack.Networking.Network(&quot;network1&quot;, new OpenStack.Networking.NetworkArgs
    {
        AdminStateUp = &quot;true&quot;,
    });
    var subnetpool1 = new OpenStack.Networking.SubnetPool(&quot;subnetpool1&quot;, new OpenStack.Networking.SubnetPoolArgs
    {
        Prefixes = 
        {
            &quot;10.11.12.0/24&quot;,
        },
    });
    var subnet1 = new OpenStack.Networking.Subnet(&quot;subnet1&quot;, new OpenStack.Networking.SubnetArgs
    {
        Cidr = &quot;10.11.12.0/25&quot;,
        NetworkId = network1.Id,
        SubnetpoolId = subnetpool1.Id,
    });
}

}
