Class Healthcheck

Standalone Health Checks provide a way to monitor origin servers without needing a Cloudflare Load Balancer.

Example Usage

TCP Monitor

using Pulumi;
using Cloudflare = Pulumi.Cloudflare;

class MyStack : Stack
{
public MyStack()
{
    var tcpHealthCheck = new Cloudflare.Healthcheck(&quot;tcpHealthCheck&quot;, new Cloudflare.HealthcheckArgs
    {
        ZoneId = @var.Cloudflare_zone_id,
        Name = &quot;tcp-health-check&quot;,
        Description = &quot;example tcp health check&quot;,
        Address = &quot;example.com&quot;,
        Suspended = false,
        CheckRegions = 
        {
            &quot;WEU&quot;,
            &quot;EEU&quot;,
        },
        NotificationSuspended = false,
        NotificationEmailAddresses = 
        {
            &quot;hostmaster@example.com&quot;,
        },
        Type = &quot;TCP&quot;,
        Port = &quot;22&quot;,
        Method = &quot;connection_established&quot;,
        Timeout = 10,
        Retries = 2,
        Interval = 60,
        ConsecutiveFails = 3,
        ConsecutiveSuccesses = 2,
    });
}

}
