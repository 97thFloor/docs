Class AccessIdentityProvider

Provides a Cloudflare Access Identity Provider resource. Identity Providers are used as an authentication or authorisation source within Access.

Example Usage

using Pulumi;
using Cloudflare = Pulumi.Cloudflare;

class MyStack : Stack
{
public MyStack()
{
    // one time pin
    var pinLogin = new Cloudflare.AccessIdentityProvider(&quot;pinLogin&quot;, new Cloudflare.AccessIdentityProviderArgs
    {
        AccountId = &quot;1d5fdc9e88c8a8c4518b068cd94331fe&quot;,
        Name = &quot;PIN login&quot;,
        Type = &quot;onetimepin&quot;,
    });
    // oauth
    var githubOauth = new Cloudflare.AccessIdentityProvider(&quot;githubOauth&quot;, new Cloudflare.AccessIdentityProviderArgs
    {
        AccountId = &quot;1d5fdc9e88c8a8c4518b068cd94331fe&quot;,
        Configs = 
        {
            new Cloudflare.Inputs.AccessIdentityProviderConfigArgs
            {
                ClientId = &quot;example&quot;,
                ClientSecret = &quot;secret_key&quot;,
            },
        },
        Name = &quot;GitHub OAuth&quot;,
        Type = &quot;github&quot;,
    });
    // saml
    var jumpcloudSaml = new Cloudflare.AccessIdentityProvider(&quot;jumpcloudSaml&quot;, new Cloudflare.AccessIdentityProviderArgs
    {
        AccountId = &quot;1d5fdc9e88c8a8c4518b068cd94331fe&quot;,
        Configs = 
        {
            new Cloudflare.Inputs.AccessIdentityProviderConfigArgs
            {
                Attributes = 
                {
                    &quot;email&quot;,
                    &quot;username&quot;,
                },
                IdpPublicCert = @&quot;MIIDpDCCAoygAwIBAgIGAV2ka+55MA0GCSqGSIb3DQEBCwUAMIGSMQswCQ...GF/Q2/MHadws97cZg
uTnQyuOqPuHbnN83d/2l1NSYKCbHt24o
&quot;,
                IssuerUrl = &quot;jumpcloud&quot;,
                SignRequest = false,
                SsoTargetUrl = &quot;https://sso.myexample.jumpcloud.com/saml2/cloudflareaccess&quot;,
            },
        },
        Name = &quot;JumpCloud SAML&quot;,
        Type = &quot;saml&quot;,
    });
}

}
