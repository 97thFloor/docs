Class SpectrumApplication

Provides a Cloudflare Spectrum Application. You can extend the power of Cloudflare's DDoS, TLS, and IP Firewall to your other TCP-based services.

Example Usage

using Pulumi;
using Cloudflare = Pulumi.Cloudflare;

class MyStack : Stack
{
public MyStack()
{
    // Define a spectrum application proxies ssh traffic
    var sshProxy = new Cloudflare.SpectrumApplication(&quot;sshProxy&quot;, new Cloudflare.SpectrumApplicationArgs
    {
        ZoneId = @var.Cloudflare_zone_id,
        Protocol = &quot;tcp/22&quot;,
        TrafficType = &quot;direct&quot;,
        Dns = new Cloudflare.Inputs.SpectrumApplicationDnsArgs
        {
            Type = &quot;CNAME&quot;,
            Name = &quot;ssh.example.com&quot;,
        },
        OriginDirects = 
        {
            &quot;tcp://109.151.40.129:22&quot;,
        },
    });
}

}
