Class LoadBalancerMonitor

If you're using Cloudflare's Load Balancing to load-balance across multiple origin servers or data centers, you configure one of these Monitors to actively check the availability of those servers over HTTP(S) or TCP.

Example Usage

HTTP Monitor

using Pulumi;
using Cloudflare = Pulumi.Cloudflare;

class MyStack : Stack
{
public MyStack()
{
    var httpMonitor = new Cloudflare.LoadBalancerMonitor(&quot;httpMonitor&quot;, new Cloudflare.LoadBalancerMonitorArgs
    {
        AllowInsecure = false,
        Description = &quot;example http load balancer&quot;,
        ExpectedBody = &quot;alive&quot;,
        ExpectedCodes = &quot;2xx&quot;,
        FollowRedirects = true,
        Headers = 
        {
            new Cloudflare.Inputs.LoadBalancerMonitorHeaderArgs
            {
                Header = &quot;Host&quot;,
                Values = 
                {
                    &quot;example.com&quot;,
                },
            },
        },
        Interval = 60,
        Method = &quot;GET&quot;,
        Path = &quot;/health&quot;,
        Retries = 5,
        Timeout = 7,
        Type = &quot;http&quot;,
    });
}

}
