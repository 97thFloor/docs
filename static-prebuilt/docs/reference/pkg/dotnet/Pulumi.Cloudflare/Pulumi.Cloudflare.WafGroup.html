Class WafGroup

Provides a Cloudflare WAF rule group resource for a particular zone. This can be used to configure firewall behaviour for pre-defined firewall groups.

Example Usage

using Pulumi;
using Cloudflare = Pulumi.Cloudflare;

class MyStack : Stack
{
public MyStack()
{
    var honeyPot = new Cloudflare.WafGroup(&quot;honeyPot&quot;, new Cloudflare.WafGroupArgs
    {
        GroupId = &quot;de677e5818985db1285d0e80225f06e5&quot;,
        Mode = &quot;on&quot;,
        ZoneId = &quot;ae36f999674d196762efcc5abb06b345&quot;,
    });
}

}
