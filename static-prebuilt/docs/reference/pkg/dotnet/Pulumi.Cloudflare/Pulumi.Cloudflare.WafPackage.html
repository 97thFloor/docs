Class WafPackage

Provides a Cloudflare WAF rule package resource for a particular zone. This can be used to configure firewall behaviour for pre-defined firewall packages.

Example Usage

using Pulumi;
using Cloudflare = Pulumi.Cloudflare;

class MyStack : Stack
{
public MyStack()
{
    var owasp = new Cloudflare.WafPackage(&quot;owasp&quot;, new Cloudflare.WafPackageArgs
    {
        ActionMode = &quot;simulate&quot;,
        PackageId = &quot;a25a9a7e9c00afc1fb2e0245519d725b&quot;,
        Sensitivity = &quot;medium&quot;,
        ZoneId = &quot;ae36f999674d196762efcc5abb06b345&quot;,
    });
}

}
