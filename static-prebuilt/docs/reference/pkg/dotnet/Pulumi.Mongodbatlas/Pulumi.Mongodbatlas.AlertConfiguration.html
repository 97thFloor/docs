Class AlertConfiguration

mongodbatlas..AlertConfiguration provides an Alert Configuration resource to define the conditions that trigger an alert and the methods of notification within a MongoDB Atlas project.

NOTE: Groups and projects are synonymous terms. You may find groupId in the official documentation.

Example Usage

using Pulumi;
using Mongodbatlas = Pulumi.Mongodbatlas;

class MyStack : Stack
{
public MyStack()
{
    var test = new Mongodbatlas.AlertConfiguration(&quot;test&quot;, new Mongodbatlas.AlertConfigurationArgs
    {
        Enabled = true,
        EventType = &quot;OUTSIDE_METRIC_THRESHOLD&quot;,
        Matchers = 
        {
            new Mongodbatlas.Inputs.AlertConfigurationMatcherArgs
            {
                FieldName = &quot;HOSTNAME_AND_PORT&quot;,
                Operator = &quot;EQUALS&quot;,
                Value = &quot;SECONDARY&quot;,
            },
        },
        MetricThreshold = new Mongodbatlas.Inputs.AlertConfigurationMetricThresholdArgs
        {
            Metric_name = &quot;ASSERT_REGULAR&quot;,
            Mode = &quot;AVERAGE&quot;,
            Operator = &quot;LESS_THAN&quot;,
            Threshold = 99,
            Units = &quot;RAW&quot;,
        },
        Notifications = 
        {
            new Mongodbatlas.Inputs.AlertConfigurationNotificationArgs
            {
                DelayMin = 0,
                EmailEnabled = true,
                IntervalMin = 5,
                Roles = 
                {
                    &quot;GROUP_CHARTS_ADMIN&quot;,
                    &quot;GROUP_CLUSTER_MANAGER&quot;,
                },
                SmsEnabled = false,
                TypeName = &quot;GROUP&quot;,
            },
        },
        ProjectId = &quot;&lt;PROJECT-ID>&quot;,
    });
}

}
