Class Monitor

Use this resource to create, update, and delete a synthetics monitor in New Relic.

Example Usage

Type: SIMPLE

using Pulumi;
using NewRelic = Pulumi.NewRelic;

class MyStack : Stack
{
public MyStack()
{
    var foo = new NewRelic.Synthetics.Monitor(&quot;foo&quot;, new NewRelic.Synthetics.MonitorArgs
    {
        Frequency = 5,
        Locations = 
        {
            &quot;AWS_US_EAST_1&quot;,
            &quot;AWS_US_EAST_2&quot;,
        },
        Status = &quot;ENABLED&quot;,
        Type = &quot;SIMPLE&quot;,
        Uri = &quot;https://example.com&quot;,
        ValidationString = &quot;add example validation check here&quot;,
        VerifySsl = true,
    });
    // Optional for type &quot;SIMPLE&quot; and &quot;BROWSER&quot;
}

}
