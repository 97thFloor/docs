Namespace Pulumi.AliCloud.Waf

Classes

Domain

Provides a WAF Domain resource to create domain in the Web Application Firewall.

For information about WAF and how to use it, see What is Alibaba Cloud WAF.

NOTE: Available in 1.82.0+ .

Example Usage

using Pulumi;
using AliCloud = Pulumi.AliCloud;

class MyStack : Stack
{
public MyStack()
{
    var domain = new AliCloud.Waf.Domain(&quot;domain&quot;, new AliCloud.Waf.DomainArgs
    {
        ClusterType = &quot;PhysicalCluster&quot;,
        Domain = &quot;www.aliyun.com&quot;,
        Http2Ports = 443,
        HttpPorts = 80,
        HttpToUserIp = &quot;Off&quot;,
        HttpsPorts = 443,
        HttpsRedirect = &quot;Off&quot;,
        InstanceId = &quot;waf-123455&quot;,
        IsAccessProduct = &quot;On&quot;,
        LoadBalancing = &quot;IpHash&quot;,
        LogHeaders = 
        {
            new AliCloud.Waf.Inputs.DomainLogHeaderArgs
            {
                Key = &quot;foo&quot;,
                Value = &quot;http&quot;,
            },
        },
        SourceIps = 
        {
            &quot;1.1.1.1&quot;,
        },
    });
}

}

Instance

Provides a WAF Instance resource to create instance in the Web Application Firewall.

For information about WAF and how to use it, see What is Alibaba Cloud WAF.

NOTE: Available in 1.83.0+ .

Example Usage

using Pulumi;
using AliCloud = Pulumi.AliCloud;

class MyStack : Stack
{
public MyStack()
{
    var @default = new AliCloud.Waf.Instance(&quot;default&quot;, new AliCloud.Waf.InstanceArgs
    {
        BigScreen = &quot;0&quot;,
        ExclusiveIpPackage = &quot;1&quot;,
        ExtBandwidth = &quot;50&quot;,
        ExtDomainPackage = &quot;1&quot;,
        LogStorage = &quot;3&quot;,
        LogTime = &quot;180&quot;,
        PackageCode = &quot;version_3&quot;,
        Period = 1,
        PrefessionalService = &quot;false&quot;,
        ResourceGroupId = &quot;rs-abc12345&quot;,
        SubscriptionType = &quot;Subscription&quot;,
        WafLog = &quot;false&quot;,
    });
}

}
