Class AlidnsRecord

Provides a Alidns Record resource. For information about Alidns Domain Record and how to use it, see What is Resource Alidns Record.

NOTE: Available in v1.85.0+.

NOTE: When the site is an international site, the type neither supports REDIRECT_URL nor REDIRECT_URL

Example Usage

using Pulumi;
using AliCloud = Pulumi.AliCloud;

class MyStack : Stack
{
public MyStack()
{
    // Create a new Domain Record
    var record = new AliCloud.Dns.AlidnsRecord(&quot;record&quot;, new AliCloud.Dns.AlidnsRecordArgs
    {
        DomainName = &quot;domainname&quot;,
        Remark = &quot;Test new alidns record.&quot;,
        Rr = &quot;@&quot;,
        Status = &quot;ENABLE&quot;,
        Type = &quot;A&quot;,
        Value = &quot;192.168.99.99&quot;,
    });
}

}
