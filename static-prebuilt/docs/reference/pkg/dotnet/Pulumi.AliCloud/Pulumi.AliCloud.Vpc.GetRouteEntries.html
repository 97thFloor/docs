InvokeAsync(GetRouteEntriesArgs, InvokeOptions)

This data source provides a list of Route Entries owned by an Alibaba Cloud account.

NOTE: Available in 1.37.0+.

{{% examples %}}

Example Usage

{{% example %}}

using Pulumi;
using AliCloud = Pulumi.AliCloud;

class MyStack : Stack
{
public MyStack()
{
    var defaultZones = Output.Create(AliCloud.GetZones.InvokeAsync(new AliCloud.GetZonesArgs
    {
        AvailableResourceCreation = &quot;VSwitch&quot;,
    }));
    var defaultInstanceTypes = defaultZones.Apply(defaultZones => Output.Create(AliCloud.Ecs.GetInstanceTypes.InvokeAsync(new AliCloud.Ecs.GetInstanceTypesArgs
    {
        AvailabilityZone = defaultZones.Zones[0].Id,
        CpuCoreCount = 1,
        MemorySize = 2,
    })));
    var defaultImages = Output.Create(AliCloud.Ecs.GetImages.InvokeAsync(new AliCloud.Ecs.GetImagesArgs
    {
        MostRecent = true,
        NameRegex = &quot;^ubuntu_18.*64&quot;,
        Owners = &quot;system&quot;,
    }));
    var config = new Config();
    var name = config.Get(&quot;name&quot;) ?? &quot;tf-testAccRouteEntryConfig&quot;;
    var fooNetwork = new AliCloud.Vpc.Network(&quot;fooNetwork&quot;, new AliCloud.Vpc.NetworkArgs
    {
        CidrBlock = &quot;10.1.0.0/21&quot;,
    });
    var fooSwitch = new AliCloud.Vpc.Switch(&quot;fooSwitch&quot;, new AliCloud.Vpc.SwitchArgs
    {
        AvailabilityZone = defaultZones.Apply(defaultZones => defaultZones.Zones[0].Id),
        CidrBlock = &quot;10.1.1.0/24&quot;,
        VpcId = fooNetwork.Id,
    });
    var tfTestFoo = new AliCloud.Ecs.SecurityGroup(&quot;tfTestFoo&quot;, new AliCloud.Ecs.SecurityGroupArgs
    {
        Description = &quot;foo&quot;,
        VpcId = fooNetwork.Id,
    });
    var fooInstance = new AliCloud.Ecs.Instance(&quot;fooInstance&quot;, new AliCloud.Ecs.InstanceArgs
    {
        AllocatePublicIp = true,
        ImageId = defaultImages.Apply(defaultImages => defaultImages.Images[0].Id),
        InstanceChargeType = &quot;PostPaid&quot;,
        InstanceName = name,
        InstanceType = defaultInstanceTypes.Apply(defaultInstanceTypes => defaultInstanceTypes.InstanceTypes[0].Id),
        InternetChargeType = &quot;PayByTraffic&quot;,
        InternetMaxBandwidthOut = 5,
        SecurityGroups = 
        {
            tfTestFoo.Id,
        },
        SystemDiskCategory = &quot;cloud_efficiency&quot;,
        VswitchId = fooSwitch.Id,
    });
    var fooRouteEntry = new AliCloud.Vpc.RouteEntry(&quot;fooRouteEntry&quot;, new AliCloud.Vpc.RouteEntryArgs
    {
        DestinationCidrblock = &quot;172.11.1.1/32&quot;,
        NexthopId = fooInstance.Id,
        NexthopType = &quot;Instance&quot;,
        RouteTableId = fooNetwork.RouteTableId,
    });
    var ingress = new AliCloud.Ecs.SecurityGroupRule(&quot;ingress&quot;, new AliCloud.Ecs.SecurityGroupRuleArgs
    {
        CidrIp = &quot;0.0.0.0/0&quot;,
        IpProtocol = &quot;tcp&quot;,
        NicType = &quot;intranet&quot;,
        Policy = &quot;accept&quot;,
        PortRange = &quot;22/22&quot;,
        Priority = 1,
        SecurityGroupId = tfTestFoo.Id,
        Type = &quot;ingress&quot;,
    });
    var fooRouteEntries = fooRouteEntry.RouteTableId.Apply(routeTableId => AliCloud.Vpc.GetRouteEntries.InvokeAsync(new AliCloud.Vpc.GetRouteEntriesArgs
    {
        RouteTableId = routeTableId,
    }));
}

}
