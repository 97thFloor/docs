Provides a CDN Accelerated Domain resource.

For information about domain config and how to use it, see Batch set config

NOTE: Available in v1.34.0+.

Example Usage

using Pulumi;
using AliCloud = Pulumi.AliCloud;

class MyStack : Stack
{
public MyStack()
{
    // Create a new Domain config.
    var domain = new AliCloud.Cdn.DomainNew(&quot;domain&quot;, new AliCloud.Cdn.DomainNewArgs
    {
        CdnType = &quot;web&quot;,
        DomainName = &quot;tf-testacc%d.xiaozhu.com&quot;,
        Scope = &quot;overseas&quot;,
        Sources = new AliCloud.Cdn.Inputs.DomainNewSourcesArgs
        {
            Content = &quot;1.1.1.1&quot;,
            Port = 80,
            Priority = &quot;20&quot;,
            Type = &quot;ipaddr&quot;,
            Weight = &quot;15&quot;,
        },
    });
    var config = new AliCloud.Cdn.DomainConfig(&quot;config&quot;, new AliCloud.Cdn.DomainConfigArgs
    {
        DomainName = domain.DomainName,
        FunctionArgs = 
        {
            new AliCloud.Cdn.Inputs.DomainConfigFunctionArgArgs
            {
                ArgName = &quot;ip_list&quot;,
                ArgValue = &quot;110.110.110.110&quot;,
            },
        },
        FunctionName = &quot;ip_allow_list_set&quot;,
    });
}

}
