Class Instance

Create an DNS Instance resource.

NOTE: Available in v1.80.0+.

Example Usage

using Pulumi;
using AliCloud = Pulumi.AliCloud;

class MyStack : Stack
{
public MyStack()
{
    var @this = new AliCloud.Dns.Instance(&quot;this&quot;, new AliCloud.Dns.InstanceArgs
    {
        DnsSecurity = &quot;no&quot;,
        DomainNumbers = &quot;2&quot;,
        Period = 1,
        RenewPeriod = 1,
        RenewalStatus = &quot;ManualRenewal&quot;,
        VersionCode = &quot;version_personal&quot;,
    });
}

}
