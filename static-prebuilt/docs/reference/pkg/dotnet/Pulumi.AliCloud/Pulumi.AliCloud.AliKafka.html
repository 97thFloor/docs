ConsumerGroup

Provides an ALIKAFKA consumer group resource.

NOTE: Available in 1.56.0+

NOTE: Only the following regions support create alikafka consumer group. [cn-hangzhou,cn-beijing,cn-shenzhen,cn-shanghai,cn-qingdao,cn-hongkong,cn-huhehaote,cn-zhangjiakou,ap-southeast-1,ap-south-1,ap-southeast-5]

Example Usage

using Pulumi;
using AliCloud = Pulumi.AliCloud;

class MyStack : Stack
{
public MyStack()
{
    var config = new Config();
    var consumerId = config.Get(&quot;consumerId&quot;) ?? &quot;CID-alikafkaGroupDatasourceName&quot;;
    var defaultZones = Output.Create(AliCloud.GetZones.InvokeAsync(new AliCloud.GetZonesArgs
    {
        AvailableResourceCreation = &quot;VSwitch&quot;,
    }));
    var defaultNetwork = new AliCloud.Vpc.Network(&quot;defaultNetwork&quot;, new AliCloud.Vpc.NetworkArgs
    {
        CidrBlock = &quot;172.16.0.0/12&quot;,
    });
    var defaultSwitch = new AliCloud.Vpc.Switch(&quot;defaultSwitch&quot;, new AliCloud.Vpc.SwitchArgs
    {
        AvailabilityZone = defaultZones.Apply(defaultZones => defaultZones.Zones[0].Id),
        CidrBlock = &quot;172.16.0.0/24&quot;,
        VpcId = defaultNetwork.Id,
    });
    var defaultInstance = new AliCloud.AliKafka.Instance(&quot;defaultInstance&quot;, new AliCloud.AliKafka.InstanceArgs
    {
        DeployType = &quot;5&quot;,
        DiskSize = &quot;500&quot;,
        DiskType = &quot;1&quot;,
        IoMax = &quot;20&quot;,
        TopicQuota = &quot;50&quot;,
        VswitchId = defaultSwitch.Id,
    });
    var defaultConsumerGroup = new AliCloud.AliKafka.ConsumerGroup(&quot;defaultConsumerGroup&quot;, new AliCloud.AliKafka.ConsumerGroupArgs
    {
        ConsumerId = consumerId,
        InstanceId = defaultInstance.Id,
    });
}

}
