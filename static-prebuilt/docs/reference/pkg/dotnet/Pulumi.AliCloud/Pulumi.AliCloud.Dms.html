Namespace Pulumi.AliCloud.Dms

Classes

EnterpriseInstance

Provides a DMS Enterprise Instance resource.

NOTE: API users must first register in DMS. NOTE: Available in 1.81.0+.

Example Usage

using Pulumi;
using AliCloud = Pulumi.AliCloud;

class MyStack : Stack
{
public MyStack()
{
    var @default = new AliCloud.Dms.EnterpriseInstance(&quot;default&quot;, new AliCloud.Dms.EnterpriseInstanceArgs
    {
        DatabasePassword = &quot;Yourpassword123&quot;,
        DatabaseUser = &quot;your_user_name&quot;,
        DbaUid = &quot;1182725234xxxxxxx&quot;,
        EcsRegion = &quot;cn-shanghai&quot;,
        EnvType = &quot;test&quot;,
        ExportTimeout = 600,
        Host = &quot;rm-uf648hgsxxxxxx.mysql.rds.aliyuncs.com&quot;,
        InstanceAlias = &quot;your_alias_name&quot;,
        InstanceSource = &quot;RDS&quot;,
        InstanceType = &quot;MySQL&quot;,
        NetworkType = &quot;VPC&quot;,
        Port = 3306,
        QueryTimeout = 60,
        SafeRule = &quot;自由操作&quot;,
        Tid = &quot;12345&quot;,
    });
}

}
