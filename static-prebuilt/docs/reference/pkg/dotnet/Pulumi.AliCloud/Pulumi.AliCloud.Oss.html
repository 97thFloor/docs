Namespace Pulumi.AliCloud.Oss

Classes

Bucket

Provides a resource to create a oss bucket and set its attribution.

NOTE: The bucket namespace is shared by all users of the OSS system. Please set bucket name as unique as possible.

Example Usage

using Pulumi;
using AliCloud = Pulumi.AliCloud;

class MyStack : Stack
{
public MyStack()
{
    var bucket_acl = new AliCloud.Oss.Bucket(&quot;bucket-acl&quot;, new AliCloud.Oss.BucketArgs
    {
        Acl = &quot;private&quot;,
        Bucket = &quot;bucket-170309-acl&quot;,
    });
}

}

BucketObject

Provides a resource to put a object(content or file) to a oss bucket.

Example Usage

Uploading a file to a bucket

using Pulumi;
using AliCloud = Pulumi.AliCloud;

class MyStack : Stack
{
public MyStack()
{
    var object_source = new AliCloud.Oss.BucketObject(&quot;object-source&quot;, new AliCloud.Oss.BucketObjectArgs
    {
        Bucket = &quot;your_bucket_name&quot;,
        Key = &quot;new_object_key&quot;,
        Source = &quot;path/to/file&quot;,
    });
}

}

Uploading a content to a bucket

using Pulumi;
using AliCloud = Pulumi.AliCloud;

class MyStack : Stack
{
public MyStack()
{
    var example = new AliCloud.Oss.Bucket(&quot;example&quot;, new AliCloud.Oss.BucketArgs
    {
        Acl = &quot;public-read&quot;,
        Bucket = &quot;your_bucket_name&quot;,
    });
    var object_content = new AliCloud.Oss.BucketObject(&quot;object-content&quot;, new AliCloud.Oss.BucketObjectArgs
    {
        Bucket = example.BucketName,
        Content = &quot;the content that you want to upload.&quot;,
        Key = &quot;new_object_key&quot;,
    });
}

}
