Class HeatmapChart

This chart type displays the specified plot in a heatmap fashion. This format is similar to the Infrastructure Navigator, with squares representing each source for the selected metric, and the color of each square representing the value range of the metric.

Example Usage

using Pulumi;
using SignalFx = Pulumi.SignalFx;

class MyStack : Stack
{
public MyStack()
{
    var myheatmapchart0 = new SignalFx.HeatmapChart(&quot;myheatmapchart0&quot;, new SignalFx.HeatmapChartArgs
    {
        ColorRange = new SignalFx.Inputs.HeatmapChartColorRangeArgs
        {
            Color = &quot;#ff0000&quot;,
            MaxValue = 100,
            MinValue = 0,
        },
        ColorScales = 
        {
            new SignalFx.Inputs.HeatmapChartColorScaleArgs
            {
                Color = &quot;green&quot;,
                Gte = 99,
            },
            new SignalFx.Inputs.HeatmapChartColorScaleArgs
            {
                Color = &quot;yellow&quot;,
                Gte = 95,
                Lt = 99,
            },
            new SignalFx.Inputs.HeatmapChartColorScaleArgs
            {
                Color = &quot;red&quot;,
                Lt = 95,
            },
        },
        Description = &quot;Very cool Heatmap&quot;,
        DisableSampling = true,
        GroupBies = 
        {
            &quot;hostname&quot;,
            &quot;host&quot;,
        },
        HideTimestamp = true,
        ProgramText = @&quot;myfilters = filter(&quot;&quot;cluster_name&quot;&quot;, &quot;&quot;prod&quot;&quot;) and filter(&quot;&quot;role&quot;&quot;, &quot;&quot;search&quot;&quot;)
data(&quot;&quot;cpu.total.idle&quot;&quot;, filter=myfilters).publish()

&quot;,
        SortBy = &quot;+host&quot;,
    });
}

}
